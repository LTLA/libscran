309
310 ChooseOutlierFilters filter;

312 filter.set_min_diff(-std::log(1 - min_detected_drop));
313 filter.set_upper(false);
314 auto detected_filt = filter.run(std::move(detected_res));
315
316 output.detected = std::move(detected_filt.lower);
317 }
318

321 ComputeMedianMad meddler;
