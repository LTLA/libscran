147 template<typename Data_, typename Index_, typename Block_>
148 void run_sparse_simple(
149 const tatami::Matrix<Data_, Index_>* mat,
150 const Block_* block,
151 const pca_utils::BlockingDetails<true> block_details,
152 const irlba::Irlba& irb,
153 Eigen::MatrixXd& pcs,
154 Eigen::MatrixXd& rotation,
155 Eigen::VectorXd& variance_explained,
156 double& total_var)
157 const {
158 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
159 pca_utils::SparseMatrix emat(mat->ncol(), mat->nrow(), std::move(extracted.values), std::move(extracted.indices), std::move(extracted.ptrs), nthreads); // CSC with genes in columns.
160
161 size_t ngenes = emat.cols();
162 Eigen::VectorXd center_v(ngenes);
163 Eigen::VectorXd scale_v(ngenes);
164
165 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
166 const auto& values = emat.get_values();
167 const auto& indices = emat.get_indices();
168 const auto& ptrs = emat.get_pointers();
169
170 const auto& block_size = block_details.block_size;
171 size_t nblocks = block_size.size();
172 std::vector<int> block_count(nblocks);
173 const auto& block_weight = block_details.per_element_weight;
174
175 for (size_t r = start, end = start + length; r < end; ++r) {
176 auto offset = ptrs[r];
177 size_t num_entries = ptrs[r+1] - offset;
178 auto value_ptr = values.data() + offset;
179 auto index_ptr = indices.data() + offset;
180
181 std::fill(block_count.begin(), block_count.end(), 0);
182
183 // Computing the grand mean across all blocks.
184 double& grand_mean = center_v[r];
185 grand_mean = 0;
186 for (size_t i = 0; i < num_entries; ++i) {
187 auto b = block[index_ptr[i]];
188 grand_mean += value_ptr[i] * block_weight[b];
189 ++(block_count[b]);
190 }
191 grand_mean /= block_details.total_block_weight;
192
193 // Computing pseudo-variances where each block's contribution
194 // is weighted inversely proportional to its size. This aims to
195 // match up with the variances used in the PCA but not the
196 // variances of the output components (where weightings are not used).
197 double& proxyvar = scale_v[r];
198 proxyvar = 0;
199 for (size_t b = 0; b < nblocks; ++b) {
200 double zero_sum = (block_size[b] - block_count[b]) * grand_mean * grand_mean;
201 proxyvar += zero_sum * block_weight[b];
202 }
203
204 for (size_t i = 0; i < num_entries; ++i) {
205 double diff = value_ptr[i] - grand_mean;
206 proxyvar += diff * diff * block_weight[block[index_ptr[i]]];
207 }
208
209 proxyvar /= emat.rows() - 1;
210 }
211 }, ngenes, nthreads);
212
213 total_var = pca_utils::process_scale_vector(scale, scale_v);
214
215 // Now actually performing the PCA.
216 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
217 if (scale) {
218 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
219 pca_utils::SampleScaledWrapper<decltype(scaled)> weighted(&scaled, &(block_details.expanded_weights));
220 irb.run(weighted, pcs, rotation, variance_explained);
221 } else {
222 pca_utils::SampleScaledWrapper<decltype(centered)> weighted(&centered, &(block_details.expanded_weights));
223 irb.run(weighted, pcs, rotation, variance_explained);
224 }
225
226 // This transposes 'pcs' to be a NDIM * NCELLS matrix.
227 pca_utils::project_sparse_matrix(emat, pcs, rotation, scale, scale_v, nthreads);
228
229 pca_utils::clean_up_projected<true>(pcs, variance_explained);
230 if (!transpose) {
231 pcs.adjointInPlace();
232 }
233 }

235 template<typename Data_, typename Index_, typename Block_>
236 void run_dense_simple(
237 const tatami::Matrix<Data_, Index_>* mat,
238 const Block_* block,
239 const pca_utils::BlockingDetails<true>& block_details,
240 const irlba::Irlba& irb,
241 Eigen::MatrixXd& pcs,
242 Eigen::MatrixXd& rotation,
243 Eigen::VectorXd& variance_explained,
244 double& total_var)
245 const {
246 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads); // row-major extraction.
247
248 size_t ngenes = emat.cols();
249 Eigen::VectorXd center_v(ngenes);
250 Eigen::VectorXd scale_v(ngenes);
251
252 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
253 size_t nblocks = block_details.num_blocks();
254 std::vector<double> mean_buffer(nblocks);
255 const auto& block_weight = block_details.per_element_weight;
256 size_t ncells = emat.rows();
257
258 for (size_t c = start, end = start + length; c < end; ++c) {
259 auto ptr = emat.data() + c * ncells;
260
261 double& grand_mean = center_v[c];
262 grand_mean = 0;
263 for (size_t r = 0; r < ncells; ++r) {
264 grand_mean += ptr[r] * block_weight[block[r]];
265 }
266 grand_mean /= block_details.total_block_weight;
267
268 // We don't actually compute the batchwise variance, but instead
269 // the weighted sum of squared deltas, which is what PCA actually sees.
270 double& proxyvar = scale_v[c];
271 proxyvar = 0;
272 for (size_t r = 0; r < ncells; ++r) {
273 double diff = ptr[r] - grand_mean;
274 proxyvar += diff * diff * block_weight[block[r]];
275 }
276
277 proxyvar /= emat.rows() - 1;
278 }
279 }, ngenes, nthreads);
280
281 total_var = pca_utils::process_scale_vector(scale, scale_v);
282
283 // Applying the centering and scaling now so we can do the PCA with fewer wrappers.
284 pca_utils::apply_center_and_scale_to_dense_matrix(emat, center_v, scale, scale_v, nthreads);
285
286 pca_utils::SampleScaledWrapper<decltype(emat)> weighted(&emat, &(block_details.expanded_weights));
287 irb.run(weighted, pcs, rotation, variance_explained);
288
289 pcs.noalias() = emat * rotation;
290 pca_utils::clean_up_projected<false>(pcs, variance_explained);
291 if (transpose) {
292 pcs.adjointInPlace();
293 }
294 }

417 template<typename Data_, typename Index_, typename Block_>
418 void run(const tatami::Matrix<Data_, Index_>* mat, const Block_* block, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
419 irlba::EigenThreadScope t(nthreads);
420 irlba::Irlba irb;
421 irb.set_number(rank);
422
423 if (block_policy == BlockPolicy::RESIDUAL_ONLY) {
424 auto bdetails = pca_utils::compute_blocking_details<false>(mat->ncol(), block);
425 if (mat->sparse()) {
426 run_sparse_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, total_var);
427 } else {
428 run_dense_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, total_var);
429 }
430
431 } else {
432 auto bdetails = pca_utils::compute_blocking_details<true>(mat->ncol(), block);
433
434 if (block_policy == BlockPolicy::WEIGHTED_ONLY) {
435 if (mat->sparse()) {
436 run_sparse_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, total_var);
437 } else {
438 run_dense_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, total_var);
439 }
440
441 } else {
442 if (mat->sparse()) {
443 run_sparse_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, total_var);
444 } else {
445 run_dense_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, total_var);
446 }
447 }
448 }
