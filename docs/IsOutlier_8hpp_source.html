318 void compute_thresholds(size_t n, double* ptr, double& lower_threshold, double& upper_threshold) {
319 if (!n) {
320 lower_threshold = std::numeric_limits<double>::quiet_NaN();
321 upper_threshold = std::numeric_limits<double>::quiet_NaN();
322 return;
323 }
324
325 if (log) {
326 auto copy = ptr;
327 for (size_t i = 0; i < n; ++i, ++copy) {
328 if (*copy > 0) {
329 *copy = std::log(*copy);
330 } else if (*copy == 0) {
331 *copy = -std::numeric_limits<double>::infinity();
332 } else {
333 throw std::runtime_error("cannot log-transform negative values");
334 }
335 }
336 }
337
338 // First, getting the median.
339 size_t halfway = n / 2;
340 double medtmp = compute_median(ptr, halfway, n);
341
342 if (log && std::isinf(medtmp)) {
343 if (lower) {
344 lower_threshold = 0;
345 } else {
346 lower_threshold = -std::numeric_limits<double>::infinity();
347 }
348
349 if (upper) {
350 upper_threshold = 0;
351 } else {
352 upper_threshold = std::numeric_limits<double>::infinity();
353 }
354
355 return;
356 }
357
358 // Now getting the MAD. No need to protect against -inf here;
359 // if the median is finite, the MAD must also be finite.
360 auto copy = ptr;
361 for (size_t i = 0; i < n; ++i, ++copy) {
362 *copy = std::abs(*copy - medtmp);
363 }
364 double madtmp = compute_median(ptr, halfway, n);
365
366 madtmp *= 1.4826; // for equivalence with the standard deviation under normality.
367
368 // Computing the outputs.
369 if (lower) {
370 lower_threshold = medtmp - nmads * madtmp;
371 if (log) {
372 lower_threshold = std::exp(lower_threshold);
373 }
374 } else {
375 lower_threshold = -std::numeric_limits<double>::infinity();
376 }
377
378 if (upper) {
379 upper_threshold = medtmp + nmads * madtmp;
380 if (log) {
381 upper_threshold = std::exp(upper_threshold);
382 }
383 } else {
384 upper_threshold = std::numeric_limits<double>::infinity();
385 }
386
387 return;
388 }

390 static double compute_median (double* ptr, size_t halfway, size_t n) {
391 std::nth_element(ptr, ptr + halfway, ptr + n);
392 double medtmp = *(ptr + halfway);
393
394 if (n % 2 == 0) {
395 std::nth_element(ptr, ptr + halfway - 1, ptr + n);
396 medtmp = (medtmp + *(ptr + halfway - 1))/2;
397 }
398
399 return medtmp;
400 }
