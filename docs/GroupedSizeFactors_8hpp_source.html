145 template<typename T, typename IDX, typename Group, typename Ref, typename Out>
146 void run_internal(const tatami::Matrix<T, IDX>* mat, const Group* group, Ref reference, Out* output) {
147 size_t NR = mat->nrow(), NC = mat->ncol();
148 if (!NC) {
149 return;
150 }
151
152 // Aggregating each group to get a pseudo-bulk sample.
153 auto ngroups = *std::max_element(group, group + NC) + 1;
154 std::vector<double> combined(ngroups * NR);
155 {
156 std::vector<double*> sums(ngroups);
157 for (size_t i = 0; i < ngroups; ++i) {
158 sums[i] = combined.data() + i * NR;
159 }
160 AggregateAcrossCells aggregator;
161 aggregator.set_num_threads(num_threads).run(mat, group, std::move(sums), std::vector<int*>());
162 }
163
164 size_t ref = 0;
165 if constexpr(std::is_same<Ref, size_t>::value) {
166 ref = reference;
167 } else {
168 // Choosing one of them to be the reference. Here, we borrow some logic
169 // from edgeR and use the one with the largest sum of square roots,
170 // which provides a compromise between transcriptome coverage and
171 // complexity. The root ensures that we don't pick a sample that just
172 // has very high expression in a small subset of genes, while still
173 // remaining responsive to the overall coverage level.
174 double best = 0;
175
176 for (size_t i = 0; i < ngroups; ++i) {
177 auto start = combined.data() + i * NR;
178 double current = 0;
179 for (size_t j = 0; j < NR; ++j) {
180 current += std::sqrt(start[j]);
181 }
182 if (current > best) {
183 ref = i;
184 best = current;
185 }
186 }
187 }
188
189 // Computing median-based size factors. No need to center
190 // here as we'll be recentering afterwards anyway.
191 tatami::ArrayView view(combined.data(), combined.size());
192 tatami::DenseColumnMatrix<T, IDX, decltype(view)> aggmat(NR, ngroups, std::move(view));
193 MedianSizeFactors med;
194 med.set_num_threads(num_threads).set_center(false).set_prior_count(prior_count);
195 auto mres = med.run(&aggmat, combined.data() + ref * NR);
196
197 // Propagating to each cell via library size-based normalization.
198 auto aggcolsums = tatami::column_sums(&aggmat, num_threads);
199 auto colsums = tatami::column_sums(mat, num_threads);
200 for (size_t i = 0; i < NC; ++i) {
201 auto curgroup = group[i];
202 auto scale = static_cast<double>(colsums[i])/static_cast<double>(aggcolsums[curgroup]);
203 output[i] = scale * mres.factors[curgroup];
204 }
205
206 // Throwing in some centering.
207 if (center) {
208 CenterSizeFactors centerer;
209 centerer.run(NC, output);
210 }
211 }
