163 void run_blocked(const MAT* mat, const B* block, std::vector<Stat*> means, std::vector<Stat*> variances, std::vector<Stat*> fitted, std::vector<Stat*> residuals) const {
164 size_t NR = mat->nrow(), NC = mat->ncol();
165 std::vector<int> block_size(means.size());
166
167 if (block) {
168 auto copy = block;
169 for (size_t j = 0; j < NC; ++j, ++copy) {
170 ++block_size[*copy];
171 }
172 feature_selection::BlockedVarianceFactory<true, Stat, B, decltype(block_size)> fact(NR, NC, means, variances, block, &block_size);
173 tatami::apply<0>(mat, fact, num_threads);
174 } else {
175 block_size[0] = NC;
176 feature_selection::BlockedVarianceFactory<false, Stat, B, decltype(block_size)> fact(NR, NC, means, variances, block, &block_size);
177 tatami::apply<0>(mat, fact, num_threads);
178 }
179
180 // Applying the trend fit to each block.
181 FitTrendVar fit;
182 fit.set_span(span);
183 fit.set_minimum_mean(min_mean);
184 fit.set_use_fixed_width(use_fixed_width);
185 fit.set_fixed_width(fixed_width);
186 fit.set_minimum_window_count(minimum_window_count);
187
188 for (size_t b = 0; b < block_size.size(); ++b) {
189 if (block_size[b] >= 2) {
190 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
191 } else {
192 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
193 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
194 }
195 }
196
197 return;
198 }
