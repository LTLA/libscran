103 // It's because size_factors need to be moved into the Delayed operation
104 // anyway, so we might as well ask the user to construct a vector for us.
105 if (size_factors.size() != mat->ncol()) {
106 throw std::runtime_error("number of size factors and columns are not equal");
107 }
108
109 if (center) {
110#ifdef SCRAN_LOGGER
111 SCRAN_LOGGER("scran::LogNormCounts", "Centering size factors to unity");
112#endif
113
114 if (block) {
115 auto by_group = block_indices(mat->ncol(), block);
116 for (const auto& g : by_group) {
117 if (g.size()) {
118 double mean = 0;
119 for (auto i : g) {
120 mean += size_factors[i];
121 }
122 mean /= g.size();
123
124 if (mean > 0) {
125 for (auto i : g) {
126 size_factors[i] /= mean;
127 }
128 }
129 }
130 }
131 } else {
132 if (size_factors.size()) { // avoid division by zero
133 double mean = std::accumulate(size_factors.begin(), size_factors.end(), static_cast<double>(0)) / size_factors.size();

135 for (auto& x : size_factors) {

137 }

139 }
140 }
141 }
