541 void run_internal(const MAT* p, const G* group, int ngroups,
542 std::vector<Stat*>& means,
543 std::vector<Stat*>& detected,

556 template<class MAT, typename G, typename B, typename Stat>
557 void run_blocked_internal(const MAT* p, const G* group, const B* block, int ngroups, int nblocks,
558 std::vector<std::vector<Stat*> >& means,
559 std::vector<std::vector<Stat*> >& detected,
560 std::vector<std::vector<Stat*> >& cohen,
561 std::vector<std::vector<Stat*> >& auc,
562 std::vector<std::vector<Stat*> >& lfc,
563 std::vector<std::vector<Stat*> >& delta_detected)
564 {
565 int ncombos = ngroups * nblocks;
566 std::vector<int> combos(p->ncol());
567 std::vector<int> combo_size(ncombos);
568
569 for (size_t i = 0; i < combos.size(); ++i) {
570 combos[i] = group[i] * nblocks + block[i];
571 ++(combo_size[combos[i]]);
572 }
573
574 std::vector<Stat*> means2(ncombos), detected2(ncombos);
575 auto mIt = means2.begin(), dIt = detected2.begin();
576 for (int g = 0; g < ngroups; ++g) {
577 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
578 *mIt = means[g][b];
579 *dIt = detected[g][b];
580 }
581 }
582
583 core(p, combos.data(), combo_size, group, ngroups, block, nblocks, means2, detected2, cohen, auc, lfc, delta_detected);
584 }

586 template<class MAT, typename L, class Ls, typename G, typename B, typename Stat>
587 void core(const MAT* p,
588 const L* level, const Ls& level_size,
589 const G* group, int ngroups,
590 const B* block, int nblocks,
591 std::vector<Stat*>& means,
592 std::vector<Stat*>& detected,
593 std::vector<std::vector<Stat*> >& cohen,
594 std::vector<std::vector<Stat*> >& auc,
595 std::vector<std::vector<Stat*> >& lfc,
596 std::vector<std::vector<Stat*> >& delta_detected)
597 {
598 size_t buffer_size = p->nrow() * ngroups * ngroups;
599
600 const bool do_cohen = !cohen.empty();
601 std::vector<Stat> cohen_buffer(do_cohen ? buffer_size : 0);
602 Stat* cohen_ptr = do_cohen ? cohen_buffer.data() : NULL;
603
604 const bool do_auc = !auc.empty();
605 std::vector<Stat> auc_buffer(do_auc ? buffer_size : 0);
606 Stat* auc_ptr = do_auc ? auc_buffer.data() : NULL;
607
608 const bool do_lfc = !lfc.empty();
609 std::vector<Stat> lfc_buffer(do_lfc ? buffer_size : 0);
610 Stat* lfc_ptr = do_lfc ? lfc_buffer.data() : NULL;
611
612 const bool do_delta = !delta_detected.empty();
613 std::vector<Stat> delta_buffer(do_delta ? buffer_size : 0);
614 Stat* delta_ptr = do_delta ? delta_buffer.data() : NULL;
615
616 std::vector<Stat*> effects { cohen_ptr, auc_ptr, lfc_ptr, delta_ptr };
617
618#ifdef SCRAN_LOGGER
619 SCRAN_LOGGER("scran::ScoreMarkers", "Performing pairwise comparisons between groups of cells");
620#endif
621 if (!do_auc) {
622 differential_analysis::BidimensionalFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, ngroups, nblocks, threshold);
623 tatami::apply<0>(p, fact);
624
625 } else {
626 // Need to remake this, as there's no guarantee that 'blocks' exists.
627 std::vector<B> tmp_blocks;
628 if (!block) {
629 tmp_blocks.resize(p->ncol());
630 block = tmp_blocks.data();
631 }
632
633 differential_analysis::PerRowFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, group, ngroups, block, nblocks, threshold);
634 tatami::apply<0>(p, fact);
635 }

637 auto summarize = [&](Stat* ptr, std::vector<std::vector<Stat*> >& output) -> void {
638 auto& min_rank = output[scran::differential_analysis::MIN_RANK];
639 if (min_rank.size()) {

641 }
642 differential_analysis::summarize_comparisons(p->nrow(), ngroups, ptr, output); // non-const w.r.t. ptr's values, so this is done after min-rank calculations.
643 };
644

652 summarize(lfc_ptr, lfc);
653 }
654 if (do_delta) {
655 summarize(delta_ptr, delta_detected);
656 }

658 return;

672 struct Results {

678 Results(
679 size_t ngenes,
680 int ngroups,
681 int nblocks,
682 const ComputeSummaries& do_cohen,
683 const ComputeSummaries& do_auc,
684 const ComputeSummaries& do_lfc,
685 const ComputeSummaries& do_delta_detected)
686 {
687 auto fill_inner = [&](int N, auto& type) {
688 type.reserve(N);
689 for (int n = 0; n < N; ++n) {
690 type.emplace_back(ngenes);
691 }
692 };
693
694 means.resize(ngroups);
695 detected.resize(ngroups);
696 for (int g = 0; g < ngroups; ++g) {
697 fill_inner(nblocks, means[g]);
698 fill_inner(nblocks, detected[g]);
699 }
700
701 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
702 bool has_any = false;
703 for (size_t i = 0; i < do_this.size(); ++i) {
704 if (do_this[i]) {
705 has_any = true;
706 break;
707 }
708 }
709
710 if (has_any) {
711 effect.resize(differential_analysis::n_summaries);
712 if (do_this[differential_analysis::MIN]) {
713 fill_inner(ngroups, effect[differential_analysis::MIN]);
714 }
715 if (do_this[differential_analysis::MEAN]) {
716 fill_inner(ngroups, effect[differential_analysis::MEAN]);
717 }
718 if (do_this[differential_analysis::MEDIAN]) {
719 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
720 }
721 if (do_this[differential_analysis::MAX]) {
722 fill_inner(ngroups, effect[differential_analysis::MAX]);
723 }
724 if (do_this[differential_analysis::MIN_RANK]) {
725 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
726 }
727 }
728 return;
729 };
730
731 fill_effect(do_cohen, cohen);
732 fill_effect(do_auc, auc);
733 fill_effect(do_lfc, lfc);
734 fill_effect(do_delta_detected, delta_detected);
735 return;
736 }
