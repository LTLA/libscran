438 igraph::Vector membership_holder;
439 auto& membership = membership_holder.vector;
440 igraph_integer_t nb_clusters;
441 igraph_real_t quality;
442
443 igraph::RNGScope rngs(seed);
444 Results output;
445
446 // No need to free this, as it's just a view.
447 igraph_vector_t weight_view;
448 size_t nedges = igraph_ecount(graph.get_graph());
449 igraph_vector_view(&weight_view, weights, nedges);
450
451 if (!modularity) {
452 for (int i = 0; i < iterations; ++i) {
453 output.status = igraph_community_leiden(graph.get_graph(), &weight_view, NULL, resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
454 if (output.status) {
455 break;
456 }
457 }
458 } else {
459 // Based on https://igraph.org/c/doc/igraph-Community.html#igraph_community_leiden.
460 // Copying the adjustment of igraph::cluster_leiden in the R package.
461 igraph::Vector strength_holder(igraph_vcount(graph.get_graph()));
462 auto& strength = strength_holder.vector;
463 igraph_strength(graph.get_graph(), &strength, igraph_vss_all(), IGRAPH_ALL, 1, &weight_view);
464
465 double total_weights = std::accumulate(weights, weights + nedges, 0.0);
466 double mod_resolution = resolution / (2 * total_weights);
467
468 for (int i = 0; i < iterations; ++i) {
469 output.status = igraph_community_leiden(graph.get_graph(), &weight_view, &strength, mod_resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
470 if (output.status) {
471 break;
472 }
473 }
474 }
475
476 if (!output.status) {
477 size_t ncells = igraph_vcount(graph.get_graph());
478 output.membership.resize(ncells);
479 for (size_t i = 0; i < ncells; ++i) {
480 output.membership[i] = VECTOR(membership)[i];
481 }
482 output.quality = quality;
483 }
