159 std::vector<Stat_> tmp_vars(nblocks);
160
161 std::vector<Data_> buffer(NC);
162 auto ext = tatami::consecutive_extractor<true, false>(mat, start, length);
163
164 for (Index_ r = start, end = start + length; r < end; ++r) {
165 auto ptr = ext->fetch(r, buffer.data());

168 means[b][r] = tmp_means[b];

203 void compute_dense_column(const tatami::Matrix<Data_, Index_>* mat, std::vector<Stat_*>& means, std::vector<Stat_*>& variances, const Block_* block, const std::vector<Index_>& block_size) const {
204 auto nblocks = block_size.size();
205 auto NR = mat->nrow(), NC = mat->ncol();
206
207 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
208 std::vector<Data_> buffer(length);
209 auto ext = tatami::consecutive_extractor<false, false>(mat, 0, NC, start, length);
210
211 // Shifting pointers to account for the new start point.
212 auto mcopy = means;
213 auto vcopy = variances;
214 for (Index_ b = 0; b < nblocks; ++b) {
215 mcopy[b] += start;
216 vcopy[b] += start;
217 }
218
219 std::vector<Index_> counts(nblocks);
220 for (Index_ c = 0; c < NC; ++c) {
221 auto ptr = ext->fetch(c, buffer.data());
222 auto b = feature_selection::get_block<blocked_>(c, block);
223 tatami::stats::variances::compute_running(ptr, length, mcopy[b], vcopy[b], counts[b]);
224 }
225
226 for (size_t b = 0; b < nblocks; ++b) {
227 tatami::stats::variances::finish_running(length, mcopy[b], vcopy[b], counts[b]);
228 }
229 }, NR, num_threads);
230 }

259 template<bool blocked_, typename Data_, typename Index_, typename Stat_, typename Block_>
260 void compute(const tatami::Matrix<Data_, Index_>* mat, std::vector<Stat_*>& means, std::vector<Stat_*>& variances, const Block_* block, const std::vector<Index_>& block_size) const {
261 if (mat->prefer_rows()) {
262 if (mat->sparse()) {
263 compute_sparse_row<blocked_>(mat, means, variances, block, block_size);
264 } else {
265 compute_dense_row<blocked_>(mat, means, variances, block, block_size);
266 }
267
268 } else {
269 // Set everything to zero before computing the running statistics.
270 auto NR = mat->nrow();
271 for (auto& mptr : means) {
272 std::fill(mptr, mptr + NR, 0);
273 }
274 for (auto& vptr : variances) {
275 std::fill(vptr, vptr + NR, 0);
276 }
277
278 if (mat->sparse()) {
279 compute_sparse_column<blocked_>(mat, means, variances, block, block_size);
280 } else {
281 compute_dense_column<blocked_>(mat, means, variances, block, block_size);
282 }
283 }

340 template<typename Value_, typename Index_, typename Block_, typename Stat_>
341 void run_blocked_with_average(
342 const tatami::Matrix<Value_, Index_>* mat,
343 const Block_* block,
344 std::vector<Stat_*> means,
345 std::vector<Stat_*> variances,
346 std::vector<Stat_*> fitted,
347 std::vector<Stat_*> residuals,
348 Stat_* ave_means,
349 Stat_* ave_variances,
350 Stat_* ave_fitted,
351 Stat_* ave_residuals)
352 const {
353 Index_ NR = mat->nrow(), NC = mat->ncol();
354 std::vector<Index_> block_size;
355
356 if (block) {
357 block_size = tabulate_ids(NC, block);
358 compute<true>(mat, means, variances, block, block_size);
359 } else {
360 block_size.push_back(NC); // everything is one big block.
361 compute<false>(mat, means, variances, block, block_size);
362 }
363
364 // Applying the trend fit to each block.
365 FitVarianceTrend fit;
366 fit.set_span(span);
367 fit.set_minimum_mean(min_mean);
368 fit.set_use_fixed_width(use_fixed_width);
369 fit.set_fixed_width(fixed_width);
370 fit.set_minimum_window_count(minimum_window_count);
371
372 for (size_t b = 0; b < block_size.size(); ++b) {
373 if (block_size[b] >= 2) {
374 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
375 } else {
376 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
377 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
378 }
379 }
380
381 // Computing averages under different policies.
382 if (ave_means || ave_variances || ave_fitted || ave_residuals) {
383 std::vector<double> block_weight = compute_block_weights(block_size, block_weight_policy, variable_block_weight_parameters);
384 if (ave_means) {
385 average_vectors_weighted(NR, means, block_weight.data(), ave_means);
386 }
387 if (ave_variances) {
388 average_vectors_weighted(NR, variances, block_weight.data(), ave_variances);
389 }
390 if (ave_fitted) {
391 average_vectors_weighted(NR, fitted, block_weight.data(), ave_fitted);
392 }
393 if (ave_residuals) {
394 average_vectors_weighted(NR, residuals, block_weight.data(), ave_residuals);
395 }
396 }
397
398 return;
399 }
