187 void compute_dense_column(const tatami::Matrix<Data_, Index_>* mat, std::vector<Stat_*>& means, std::vector<Stat_*>& variances, const Block_* block, const std::vector<Index_>& block_size) const {
188 auto nblocks = block_size.size();
189 auto NR = mat->nrow(), NC = mat->ncol();
190
191 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
192 std::vector<Data_> buffer(length);
193 auto ext = tatami::consecutive_extractor<false, false>(mat, 0, NC, start, length);
194
195 // Shifting pointers to account for the new start point.
196 auto mcopy = means;
197 auto vcopy = variances;
198 for (Index_ b = 0; b < nblocks; ++b) {
199 mcopy[b] += start;
200 vcopy[b] += start;
201 }
202
203 std::vector<Index_> counts(nblocks);
204 for (Index_ c = 0; c < NC; ++c) {
205 auto ptr = ext->fetch(c, buffer.data());
206 auto b = feature_selection::get_block<blocked_>(c, block);
207 tatami::stats::variances::compute_running(ptr, length, mcopy[b], vcopy[b], counts[b]);
208 }
209
210 for (size_t b = 0; b < nblocks; ++b) {
211 tatami::stats::variances::finish_running(length, mcopy[b], vcopy[b], counts[b]);
212 }
213 }, NR, num_threads);

243 template<bool blocked_, typename Data_, typename Index_, typename Stat_, typename Block_>
244 void compute(const tatami::Matrix<Data_, Index_>* mat, std::vector<Stat_*>& means, std::vector<Stat_*>& variances, const Block_* block, const std::vector<Index_>& block_size) const {
245 if (mat->prefer_rows()) {
246 if (mat->sparse()) {
247 compute_sparse_row<blocked_>(mat, means, variances, block, block_size);
248 } else {
249 compute_dense_row<blocked_>(mat, means, variances, block, block_size);
250 }
251
252 } else {
253 // Set everything to zero before computing the running statistics.
254 auto NR = mat->nrow();
255 for (auto& mptr : means) {
256 std::fill(mptr, mptr + NR, 0);
257 }
258 for (auto& vptr : variances) {
259 std::fill(vptr, vptr + NR, 0);
260 }
261
262 if (mat->sparse()) {
263 compute_sparse_column<blocked_>(mat, means, variances, block, block_size);
264 } else {
265 compute_dense_column<blocked_>(mat, means, variances, block, block_size);
266 }
267 }

325 template<typename Value_, typename Index_, typename Block_, typename Stat_>
326 void run_blocked_with_average(
327 const tatami::Matrix<Value_, Index_>* mat,
328 const Block_* block,
329 std::vector<Stat_*> means,
330 std::vector<Stat_*> variances,
331 std::vector<Stat_*> fitted,
332 std::vector<Stat_*> residuals,
333 Stat_* ave_means,
334 Stat_* ave_variances,
335 Stat_* ave_fitted,
336 Stat_* ave_residuals)
337 const {
338 Index_ NR = mat->nrow(), NC = mat->ncol();
339 std::vector<Index_> block_size;
340
341 if (block) {
342 block_size = count_blocks(NC, block);
343 compute<true>(mat, means, variances, block, block_size);
344 } else {
345 block_size.push_back(NC); // everything is one big block.
346 compute<false>(mat, means, variances, block, block_size);
347 }
348
349 // Applying the trend fit to each block.
350 FitTrendVar fit;
351 fit.set_span(span);
352 fit.set_minimum_mean(min_mean);
353 fit.set_use_fixed_width(use_fixed_width);
354 fit.set_fixed_width(fixed_width);
355 fit.set_minimum_window_count(minimum_window_count);
356
357 for (size_t b = 0; b < block_size.size(); ++b) {
358 if (block_size[b] >= 2) {
359 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
360 } else {
361 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
362 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
363 }
364 }
365
366 // Computing averages.
367 if (ave_means || ave_variances || ave_fitted || ave_residuals) {
368 std::vector<double> block_weight;
369 block_weight.reserve(block_size.size());
370 for (auto bs : block_size) {
371 block_weight.push_back(weight_block(bs, weight_size_cap));
372 }
373
374 if (ave_means) {
375 average_vectors_weighted(NR, means, block_weight.data(), ave_means);
376 }
377 if (ave_variances) {
378 average_vectors_weighted(NR, variances, block_weight.data(), ave_variances);
379 }
380 if (ave_fitted) {
381 average_vectors_weighted(NR, fitted, block_weight.data(), ave_fitted);
382 }
383 if (ave_residuals) {
384 average_vectors_weighted(NR, residuals, block_weight.data(), ave_residuals);
385 }
386 }
387
388 return;
389 }
