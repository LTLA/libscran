29struct BlockedEigenMatrix {
30 BlockedEigenMatrix(const Matrix* m, const Block* b, const Eigen::MatrixXd* mx) : mat(m), block(b), means(mx) {}

36 struct Workspace {
37 Workspace(size_t nblocks, irlba::WrappedWorkspace<Matrix> c) : sub(nblocks), child(std::move(c)) {}
38 Eigen::VectorXd sub;
39 irlba::WrappedWorkspace<Matrix> child;
40 };
41
42 Workspace workspace() const {
43 return Workspace(means->rows(), irlba::wrapped_workspace(mat));
44 }
45
46 template<class Right>
47 void multiply(const Right& rhs, Workspace& work, Eigen::VectorXd& output) const {
48 irlba::wrapped_multiply(mat, rhs, work.child, output);
49
50 work.sub.noalias() = (*means) * rhs;
51 for (Eigen::Index i = 0; i < output.size(); ++i) {
52 output.coeffRef(i) -= work.sub.coeff(block[i]);
53 }
54 return;

61 irlba::WrappedWorkspace<Matrix> child;
62 };

68 template<class Right>
69 void adjoint_multiply(const Right& rhs, AdjointWorkspace& work, Eigen::VectorXd& output) const {
70 irlba::wrapped_adjoint_multiply(mat, rhs, work.child, output);
71
72 work.aggr.setZero();
73 for (Eigen::Index i = 0; i < rhs.size(); ++i) {
74 work.aggr.coeffRef(block[i]) += rhs.coeff(i);
75 }
76
77 output.noalias() -= means->adjoint() * work.aggr;
78 return;
79 }

81public:
82 Eigen::MatrixXd realize() const {
83 Eigen::MatrixXd output = irlba::wrapped_realize(mat);
84
85 for (Eigen::Index c = 0; c < output.cols(); ++c) {
86 for (Eigen::Index r = 0; r < output.rows(); ++r) {
87 output.coeffRef(r, c) -= means->coeff(block[r], c);
88 }
89 }
90 return output;

185 template<typename T, typename IDX, typename Block>
186 void run(const tatami::Matrix<T, IDX>* mat, const Block* block, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
187 const size_t NC = mat->ncol();
188 const int nblocks = (NC ? *std::max_element(block, block + NC) + 1 : 1);
189 std::vector<int> block_size(nblocks);
190 for (size_t j = 0; j < NC; ++j) {
191 ++block_size[block[j]];
192 }
193
194 irlba::EigenThreadScope t(nthreads);
195 irlba::Irlba irb;
196 irb.set_number(rank);
197
198 if (mat->sparse()) {
199 Eigen::MatrixXd center_m(nblocks, mat->nrow());
200 Eigen::VectorXd scale_v(mat->nrow());
201 auto emat = create_custom_sparse_matrix(mat, center_m, scale_v, block, block_size, total_var);
202
203 BlockedEigenMatrix<decltype(emat), Block> centered(&emat, block, &center_m);
204 if (scale) {
205 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
206 irb.run(scaled, pcs, rotation, variance_explained);
207 } else {
208 irb.run(centered, pcs, rotation, variance_explained);
209 }
210 } else {
211 auto emat = create_eigen_matrix_dense(mat, block, block_size, total_var);
212 irb.run(emat, pcs, rotation, variance_explained); // already centered and scaled, if relevant.
213 }
214
215 pca_utils::clean_up(mat->ncol(), pcs, variance_explained);
216 if (transpose) {
217 pcs.adjointInPlace();
218 }
219
220 return;
221 }

314 template<typename T, typename IDX, typename Block>
315 pca_utils::SparseMatrix create_custom_sparse_matrix(const tatami::Matrix<T, IDX>* mat,
316 Eigen::MatrixXd& center_m,
317 Eigen::VectorXd& scale_v,
318 const Block* block,
319 const std::vector<int>& block_size,
320 double& total_var)
321 const {
322
323 size_t NR = mat->nrow(), NC = mat->ncol();
324 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major filling.
325 auto& ptrs = extracted.ptrs;
326 auto& values = extracted.values;
327 auto& indices = extracted.indices;
328
329 // Computing block-specific means and variances.
330 {
331 const size_t nblocks = block_size.size();
332
333#ifndef SCRAN_CUSTOM_PARALLEL
334 #pragma omp parallel for num_threads(nthreads)
335 for (size_t r = 0; r < NR; ++r) {
336#else
337 SCRAN_CUSTOM_PARALLEL(NR, [&](size_t first, size_t last) -> void {
338 for (size_t r = first; r < last; ++r) {
339#endif
340
341 auto offset = ptrs[r];
342 size_t num_entries = ptrs[r+1] - offset;
343 auto value_ptr = values.data() + offset;
344 auto index_ptr = indices.data() + offset;
345
346 // Computing the block-wise means.
347 auto mbuffer = center_m.data() + nblocks * r;
348 std::fill(mbuffer, mbuffer + nblocks, 0);
349 for (size_t i = 0; i < num_entries; ++i) {
350 mbuffer[block[index_ptr[i]]] += value_ptr[i];
351 }
352 for (size_t b = 0; b < nblocks; ++b) {
353 mbuffer[b] /= block_size[b];
354 }
355
356 // Computing the variance from the sum of squared differences.
357 // This is technically not the correct variance estimate if we
358 // were to consider the loss of residual d.f. from estimating
359 // the block means, but it's what the PCA sees, so whatever.
360 double& proxyvar = scale_v[r];
361 proxyvar = 0;
362 {
363 auto block_copy = block_size;
364 for (size_t i = 0; i < num_entries; ++i) {
365 Block curb = block[index_ptr[i]];
366 double diff = value_ptr[i] - mbuffer[curb];
367 proxyvar += diff * diff;
368 --block_copy[curb];
369 }
370
371 for (size_t b = 0; b < nblocks; ++b) {
372 proxyvar += mbuffer[b] * mbuffer[b] * block_copy[b];
373 }
374
375 proxyvar /= NC - 1;
376 }
377
378#ifndef SCRAN_CUSTOM_PARALLEL
379 }
380#else
381 }
382 }, nthreads);
383#endif
384
385 total_var = pca_utils::process_scale_vector(scale, scale_v);
386 }
387
388 return pca_utils::SparseMatrix(
389 NC, // NC => number of rows, i.e., it's transposed as we want genes in the columns.
390 NR,
391 std::move(values),
392 std::move(indices),
393 std::move(ptrs),
394 nthreads
395 );
396 }

399 template<typename T, typename IDX, typename Block>
400 Eigen::MatrixXd create_eigen_matrix_dense(
401 const tatami::Matrix<T, IDX>* mat,
402 const Block* block,
403 const std::vector<int>& block_size,
404 double& total_var)
405 const {
406
407 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads); // get a column-major matrix with genes in columns.
408
409 // Compute mean and variance of each block.
410 size_t NR = emat.rows(), NC = emat.cols();
411 int nblocks = block_size.size();
412 Eigen::VectorXd scale_v(NC);
413
414#ifndef SCRAN_CUSTOM_PARALLEL
415 #pragma omp parallel num_threads(nthreads)
416 {
417 std::vector<double> mean_buffer(nblocks);
418 #pragma omp for
419 for (size_t c = 0; c < NC; ++c) {
420#else
421 SCRAN_CUSTOM_PARALLEL(NC, [&](size_t first, size_t last) -> void {
422 std::vector<double> mean_buffer(nblocks);
423 for (size_t c = first; c < last; ++c) {
424#endif
425
426 auto ptr = emat.data() + c * NR;
427 std::fill(mean_buffer.begin(), mean_buffer.end(), 0);
428 for (size_t r = 0; r < NR; ++r) {
429 mean_buffer[block[r]] += ptr[r];
430 }
431
432 for (int b = 0; b < nblocks; ++b) {
433 const auto& bsize = block_size[b];
434 if (bsize) {
435 mean_buffer[b] /= bsize;
436 }
437 }
438
439 // We don't actually compute the blockwise variance, but
440 // instead the squared sum of deltas from each block's means,
441 // divided by the degrees of freedom as if there weren't any
442 // blocks... as this is what PCA actually sees.
443 double& proxyvar = scale_v[c];
444 proxyvar = 0;
445 for (size_t r = 0; r < NR; ++r) {
446 auto& current = ptr[r];
447 current -= mean_buffer[block[r]]; // centering happens here!
448 proxyvar += current * current;
449 }
450 proxyvar /= NR - 1;
451
452#ifndef SCRAN_CUSTOM_PARALLEL
453 }
454 }
455#else
456 }
457 }, nthreads);
458#endif
459
460 total_var = pca_utils::process_scale_vector(scale, scale_v);
461
462 if (scale) {
463#ifndef SCRAN_CUSTOM_PARALLEL
464 #pragma omp parallel for num_threads(nthreads)
465 for (size_t c = 0; c < NC; ++c) {
466#else
467 SCRAN_CUSTOM_PARALLEL(NC, [&](size_t first, size_t last) -> void {
468 for (size_t c = first; c < last; ++c) {
469#endif
470
471 auto ptr = emat.data() + c * NR;
472 auto sd = scale_v[c];
473 for (size_t r = 0; r < NR; ++r) {
474 ptr[r] /= sd; // process_scale_vector() should already protect against div-by-zero.
475 }
476
477#ifndef SCRAN_CUSTOM_PARALLEL
478 }
479#else
480 }
481 }, nthreads);
482#endif
483 }
484
485 return emat;
486 }
