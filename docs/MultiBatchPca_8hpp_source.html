198 template<typename Data_, typename Index_, typename Block_>
199 void run_sparse_simple(
200 const tatami::Matrix<Data_, Index_>* mat,
201 const Block_* block,
202 const pca_utils::BlockingDetails<true> block_details,
203 const irlba::Irlba& irb,
204 Eigen::MatrixXd& pcs,
205 Eigen::MatrixXd& rotation,
206 Eigen::VectorXd& variance_explained,
207 Eigen::VectorXd& center_v,
208 Eigen::VectorXd& scale_v,
209 double& total_var)
210 const {
211 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
212 pca_utils::SparseMatrix emat(mat->ncol(), mat->nrow(), std::move(extracted.values), std::move(extracted.indices), std::move(extracted.ptrs), nthreads); // CSC with genes in columns.
213
214 size_t ngenes = emat.cols();
215 center_v.resize(ngenes);
216 scale_v.resize(ngenes);
217
218 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
219 const auto& values = emat.get_values();
220 const auto& indices = emat.get_indices();
221 const auto& ptrs = emat.get_pointers();
222
223 const auto& block_size = block_details.block_size;
224 size_t nblocks = block_size.size();
225 std::vector<int> block_count(nblocks);
226 const auto& block_weight = block_details.per_element_weight;
227
228 for (size_t r = start, end = start + length; r < end; ++r) {
229 auto offset = ptrs[r];
230 size_t num_entries = ptrs[r+1] - offset;
231 auto value_ptr = values.data() + offset;
232 auto index_ptr = indices.data() + offset;
233
234 std::fill(block_count.begin(), block_count.end(), 0);
235
236 // Computing the grand mean across all blocks.
237 double& grand_mean = center_v[r];
238 grand_mean = 0;
239 for (size_t i = 0; i < num_entries; ++i) {
240 auto b = block[index_ptr[i]];
241 grand_mean += value_ptr[i] * block_weight[b];
242 ++(block_count[b]);
243 }
244 grand_mean /= block_details.total_block_weight;
245
246 // Computing pseudo-variances where each block's contribution
247 // is weighted inversely proportional to its size. This aims to
248 // match up with the variances used in the PCA but not the
249 // variances of the output components (where weightings are not used).
250 double& proxyvar = scale_v[r];
251 proxyvar = 0;
252 for (size_t b = 0; b < nblocks; ++b) {
253 double zero_sum = (block_size[b] - block_count[b]) * grand_mean * grand_mean;
254 proxyvar += zero_sum * block_weight[b];
255 }
256
257 for (size_t i = 0; i < num_entries; ++i) {
258 double diff = value_ptr[i] - grand_mean;
259 proxyvar += diff * diff * block_weight[block[index_ptr[i]]];
260 }
261
262 proxyvar /= emat.rows() - 1;
263 }
264 }, ngenes, nthreads);
265
266 total_var = pca_utils::process_scale_vector(scale, scale_v);
267
268 // Now actually performing the PCA.
269 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
270 if (scale) {
271 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
272 pca_utils::SampleScaledWrapper<decltype(scaled)> weighted(&scaled, &(block_details.expanded_weights));
273 irb.run(weighted, pcs, rotation, variance_explained);
274 } else {
275 pca_utils::SampleScaledWrapper<decltype(centered)> weighted(&centered, &(block_details.expanded_weights));
276 irb.run(weighted, pcs, rotation, variance_explained);
277 }
278
279 // This transposes 'pcs' to be a NDIM * NCELLS matrix.
280 pca_utils::project_sparse_matrix(emat, pcs, rotation, scale, scale_v, nthreads);
281
282 pca_utils::clean_up_projected<true>(pcs, variance_explained);
283 if (!transpose) {
284 pcs.adjointInPlace();
285 }
286 }

288 template<typename Data_, typename Index_, typename Block_>
289 void run_dense_simple(
290 const tatami::Matrix<Data_, Index_>* mat,
291 const Block_* block,
292 const pca_utils::BlockingDetails<true>& block_details,
293 const irlba::Irlba& irb,
294 Eigen::MatrixXd& pcs,
295 Eigen::MatrixXd& rotation,
296 Eigen::VectorXd& variance_explained,
297 Eigen::VectorXd& center_v,
298 Eigen::VectorXd& scale_v,
299 double& total_var)
300 const {
301 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads); // row-major extraction.
302
303 size_t ngenes = emat.cols();
304 center_v.resize(ngenes);
305 scale_v.resize(ngenes);
306
307 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
308 size_t nblocks = block_details.num_blocks();
309 std::vector<double> mean_buffer(nblocks);
310 const auto& block_weight = block_details.per_element_weight;
311 size_t ncells = emat.rows();
312
313 for (size_t c = start, end = start + length; c < end; ++c) {
314 auto ptr = emat.data() + c * ncells;
315
316 double& grand_mean = center_v[c];
317 grand_mean = 0;
318 for (size_t r = 0; r < ncells; ++r) {
319 grand_mean += ptr[r] * block_weight[block[r]];
320 }
321 grand_mean /= block_details.total_block_weight;
322
323 // We don't actually compute the batchwise variance, but instead
324 // the weighted sum of squared deltas, which is what PCA actually sees.
325 double& proxyvar = scale_v[c];
326 proxyvar = 0;
327 for (size_t r = 0; r < ncells; ++r) {
328 double diff = ptr[r] - grand_mean;
329 proxyvar += diff * diff * block_weight[block[r]];
330 }
331
332 proxyvar /= emat.rows() - 1;
333 }
334 }, ngenes, nthreads);
335
336 total_var = pca_utils::process_scale_vector(scale, scale_v);
337
338 // Applying the centering and scaling now so we can do the PCA with fewer wrappers.
339 pca_utils::apply_center_and_scale_to_dense_matrix(emat, center_v, scale, scale_v, nthreads);
340
341 pca_utils::SampleScaledWrapper<decltype(emat)> weighted(&emat, &(block_details.expanded_weights));
342 irb.run(weighted, pcs, rotation, variance_explained);
343
344 pcs.noalias() = emat * rotation;
345 pca_utils::clean_up_projected<false>(pcs, variance_explained);
346 if (transpose) {
347 pcs.adjointInPlace();
348 }
349 }

476 template<typename Data_, typename Index_, typename Block_>
477 bool run_internal(
478 const tatami::Matrix<Data_, Index_>* mat,
479 const Block_* block,
480 Eigen::MatrixXd& pcs,
481 Eigen::MatrixXd& rotation,
482 Eigen::VectorXd& variance_explained,
483 Eigen::MatrixXd& center_m,
484 Eigen::VectorXd& scale_v,
485 double& total_var)
486 const {
487 irlba::EigenThreadScope t(nthreads);
488 irlba::Irlba irb;
489 irb.set_number(rank);
490
491 if (block_policy == BlockPolicy::RESIDUAL_ONLY) {
492 auto bdetails = pca_utils::compute_blocking_details<false>(mat->ncol(), block);
493 if (mat->sparse()) {
494 run_sparse_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
495 } else {
496 run_dense_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
497 }
498
499 } else {
500 auto bdetails = pca_utils::compute_blocking_details<true>(mat->ncol(), block);
501
502 if (block_policy == BlockPolicy::WEIGHTED_ONLY) {
503 Eigen::VectorXd center_v;
504 if (mat->sparse()) {
505 run_sparse_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_v, scale_v, total_var);
506 } else {
507 run_dense_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_v, scale_v, total_var);
508 }
509
510 if (return_center) {
511 center_m.resize(1, center_v.size());
512 center_m.row(0) = center_v;
513 }
514
515 } else {
516 if (mat->sparse()) {
517 run_sparse_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
518 } else {
519 run_dense_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
520 }
521 }
522 }
