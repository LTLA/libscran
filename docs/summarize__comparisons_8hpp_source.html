15template<class IT>
16double median (IT start, size_t n) {
17 int halfway = n / 2;
18 std::nth_element(start, start + halfway, start + n);
19
20 if (n % 2 == 1) {
21 return start[halfway];
22 }
23
24 double med = start[halfway];
25 std::nth_element(start, start + halfway - 1, start + halfway);
26 return (med + start[halfway - 1])/ 2;
27}

29template<typename Stat>
30void summarize_comparisons(size_t ngenes, int ngroups, Stat* effects, std::vector<std::vector<Stat*> >& output) {
31 #pragma omp parallel for
32 for (size_t gene = 0; gene < ngenes; ++gene) {
33 auto base = effects + gene * ngroups * ngroups;
34 for (int l = 0; l < ngroups; ++l) {
35 auto start = base + l * ngroups;
36
37 // Ignoring the self comparison.
38 int restart = 1;
39 std::swap(start[0], start[l]);
40
41 // Pruning out NAs.
42 for (int r = restart; r < ngroups; ++r) {
43 if (std::isnan(start[r])) {
44 if (r != restart) {
45 std::swap(start[restart], start[r]);
46 }
47 ++restart;
48 }
49 }
50
51 if (restart == ngroups) {
52 for (size_t i = 0; i < 4; ++i) {
53 if (output[i][l]) {
54 output[i][l][gene] = std::numeric_limits<double>::quiet_NaN();
55 }
56 }
57 } else {
58 int ncomps = ngroups - restart;
59 if (ncomps > 1) {
60 if (output[0][l]) {
61 output[0][l][gene] = *std::min_element(start + restart, start + ngroups);
62 }
63 if (output[1][l]) {
64 output[1][l][gene] = std::accumulate(start + restart, start + ngroups, 0.0) / ncomps; // Mean
65 }
66 if (output[2][l]) {
67 output[2][l][gene] = median(start + restart, ncomps); // Median
68 }
69 if (output[3][l]) {
70 output[3][l][gene] = *std::max_element(start + restart, start + ngroups); // Maximum
71 }
72 } else {
73 for (size_t i = 0; i < 4; ++i) {
74 if (output[i][l]) {
75 output[i][l][gene] = start[restart];
76 }
77 }
78 }
79 }
80 }
81 }
82 return;
83}

85template<typename Stat, class V>
86void compute_min_rank(size_t ngenes, int ngroups, const Stat* effects, std::vector<Stat*>& output, V& buffer) {
87 auto shift = ngroups * ngroups;
88
89 for (int g = 0; g < ngroups; ++g) {
90 auto target = output[g];
91 std::fill(target, target + ngenes, ngenes + 1);
92
93 for (int g2 = 0; g2 < ngroups; ++g2) {
94 if (g == g2) {
95 continue;
96 }
97 auto copy = effects + g * ngroups + g2;
98 for (size_t i = 0; i < ngenes; ++i, copy += shift) {
99 buffer[i].first = -*copy; // negative to sort by decreasing value.
100 buffer[i].second = i;
101 }
102 std::sort(buffer.begin(), buffer.end());
103
104 double counter = 1;
105 for (const auto& x : buffer) {
106 target[x.second] = std::min(counter, target[x.second]);
107 ++counter;
108 }
109 }
110 }
111}
