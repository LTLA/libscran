101 template<typename T, typename IDX>
102 void run(const tatami::Matrix<T, IDX>* mat, Eigen::MatrixXd& pcs, Eigen::VectorXd& variance_explained, double& total_var) {
103
104 if (mat->sparse()) {
105 Eigen::VectorXd center_v(mat->nrow()), scale_v(mat->nrow());
106 auto emat = create_eigen_matrix_sparse(mat, center_v, scale_v, total_var);
107
108#ifdef SCRAN_LOGGER
109 SCRAN_LOGGER("scran::RunPCA", "Running the IRLBA algorithm");
110#endif
111
112 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
113 if (scale) {
114 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
115 auto result = irb.run(scaled);
116 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
117 } else {
118 auto result = irb.run(centered);
119 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
120 }
121 } else {
122 auto emat = create_eigen_matrix_dense(mat, total_var);
123
124#ifdef SCRAN_LOGGER
125 SCRAN_LOGGER("scran::RunPCA", "Running the IRLBA algorithm");
126#endif
127
128 auto result = irb.run(emat); // already centered and scaled, if relevant.
129 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
130 }
131
132 if (transpose) {
133 pcs.adjointInPlace();
134 }
135
136 return;
137 }

219 template<typename T, typename IDX>
220 Eigen::SparseMatrix<double> create_eigen_matrix_sparse(const tatami::Matrix<T, IDX>* mat, Eigen::VectorXd& center_v, Eigen::VectorXd& scale_v, double& total_var) {
221 size_t NR = mat->nrow(), NC = mat->ncol();
222
223 Eigen::SparseMatrix<double> A(NC, NR); // transposed; we want genes in the columns.
224 std::vector<std::vector<double> > values;
225 std::vector<std::vector<int> > indices;
226
227#ifdef SCRAN_LOGGER
228 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
229#endif
230
231 if (mat->prefer_rows()) {
232 std::vector<double> xbuffer(NC);
233 std::vector<int> ibuffer(NC);
234 std::vector<int> nonzeros(NR);
235 values.reserve(NR);
236 indices.reserve(NR);
237
238 for (size_t r = 0; r < NR; ++r) {
239 auto range = mat->sparse_row(r, xbuffer.data(), ibuffer.data());
240
241 auto stats = tatami::stats::variances::compute_direct(range, NC);
242 center_v[r] = stats.first;
243 scale_v[r] = stats.second;
244
245 nonzeros[r] = range.number;
246 values.emplace_back(range.value, range.value + range.number);
247 indices.emplace_back(range.index, range.index + range.number);
248 }
249
250 pca_utils::set_scale(scale, scale_v, total_var);
251 pca_utils::fill_sparse_matrix<true>(A, indices, values, nonzeros);
252 } else {
253 std::vector<double> xbuffer(NR);
254 std::vector<int> ibuffer(NR);
255 values.reserve(NC);
256 indices.reserve(NC);
257
258 center_v.setZero();
259 scale_v.setZero();
260 std::vector<int> nonzeros(NR);
261 int count = 0;
262
263 // First pass to compute variances and extract non-zero values.
264 for (size_t c = 0; c < NC; ++c) {
265 auto range = mat->sparse_column(c, xbuffer.data(), ibuffer.data());
266 tatami::stats::variances::compute_running(range, center_v.data(), scale_v.data(), nonzeros.data(), count);
267 values.emplace_back(range.value, range.value + range.number);
268 indices.emplace_back(range.index, range.index + range.number);
269 }
270
271 tatami::stats::variances::finish_running(NR, center_v.data(), scale_v.data(), nonzeros.data(), count);
272
273 pca_utils::set_scale(scale, scale_v, total_var);
274 pca_utils::fill_sparse_matrix<false>(A, indices, values, nonzeros);
275 }
276
277 A.makeCompressed();
278 return A;
279 }

282 template<typename T, typename IDX>
283 Eigen::MatrixXd create_eigen_matrix_dense(const tatami::Matrix<T, IDX>* mat, double& total_var) {
284 size_t NR = mat->nrow(), NC = mat->ncol();
285 total_var = 0;
286
287 Eigen::MatrixXd output(NC, NR); // transposed.
288 std::vector<double> xbuffer(NC);
289 double* outIt = output.data();
290
291#ifdef SCRAN_LOGGER
292 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
293#endif
294
295 for (size_t r = 0; r < NR; ++r, outIt += NC) {
296 auto ptr = mat->row_copy(r, outIt);
297 auto stats = tatami::stats::variances::compute_direct(outIt, NC);
298
299 auto copy = outIt;
300 for (size_t c = 0; c < NC; ++c, ++copy) {
301 *copy -= stats.first;
302 }
303
304 pca_utils::apply_scale(scale, stats.second, NC, outIt, total_var);
305 }
306
307 return output;
308 }
