178 Thresholds run(size_t n, const T* metrics, X* outliers, std::vector<double>& buffer) const {
179 Thresholds output(1);
180 std::copy(metrics, metrics + n, buffer.data());
181
182 double& lthresh = output.lower[0];
183 double& uthresh = output.upper[0];
184 compute_thresholds(n, buffer.data(), lthresh, uthresh);
185
186 auto copy = metrics;
187 for (size_t i = 0; i < n; ++i, ++copy, ++outliers) {
188 const double val = *copy;
189 *outliers = (val < lthresh || val > uthresh);
190 }
191
192 return output;
193 }

344 void compute_thresholds(size_t n, double* ptr, double& lower_threshold, double& upper_threshold) const {
345 if (!n) {
346 lower_threshold = std::numeric_limits<double>::quiet_NaN();
347 upper_threshold = std::numeric_limits<double>::quiet_NaN();
348 return;
349 }
350
351 if (log) {
352 auto copy = ptr;
353 for (size_t i = 0; i < n; ++i, ++copy) {
354 if (*copy > 0) {
355 *copy = std::log(*copy);
356 } else if (*copy == 0) {
357 *copy = -std::numeric_limits<double>::infinity();
358 } else {
359 throw std::runtime_error("cannot log-transform negative values");
360 }
361 }
362 }
363
364 // First, getting the median.
365 size_t halfway = n / 2;
366 double medtmp = compute_median(ptr, halfway, n);
367
368 if (log && std::isinf(medtmp)) {
369 if (lower) {
370 lower_threshold = 0;
371 } else {
372 lower_threshold = -std::numeric_limits<double>::infinity();
373 }
374
375 if (upper) {
376 upper_threshold = 0;
377 } else {
378 upper_threshold = std::numeric_limits<double>::infinity();
379 }
380
381 return;
382 }
383
384 // Now getting the MAD. No need to protect against -inf here;
385 // if the median is finite, the MAD must also be finite.
386 auto copy = ptr;
387 for (size_t i = 0; i < n; ++i, ++copy) {
388 *copy = std::abs(*copy - medtmp);
389 }
390 double madtmp = compute_median(ptr, halfway, n);
391
392 madtmp *= 1.4826; // for equivalence with the standard deviation under normality.
393 double delta = std::max(min_diff, nmads * madtmp);
394
395 // Computing the outputs.
396 if (lower) {
397 lower_threshold = medtmp - delta;
398 if (log) {
399 lower_threshold = std::exp(lower_threshold);
400 }
401 } else {
402 lower_threshold = -std::numeric_limits<double>::infinity();
403 }
404
405 if (upper) {
406 upper_threshold = medtmp + delta;
407 if (log) {
408 upper_threshold = std::exp(upper_threshold);
409 }
410 } else {
411 upper_threshold = std::numeric_limits<double>::infinity();
412 }
413
414 return;
415 }
416
417 static double compute_median (double* ptr, size_t halfway, size_t n) {
418 std::nth_element(ptr, ptr + halfway, ptr + n);
419 double medtmp = *(ptr + halfway);
420
421 if (n % 2 == 0) {
422 std::nth_element(ptr, ptr + halfway - 1, ptr + n);
423 medtmp = (medtmp + *(ptr + halfway - 1))/2;
424 }
425
426 return medtmp;
427 }
