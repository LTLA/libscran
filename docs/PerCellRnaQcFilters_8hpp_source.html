196 Thresholds run_internal(const Function& fun, size_t ncells, const S* sums, const D* detected, std::vector<PPTR> subset_proportions,
197 X* filter_by_sums, X* filter_by_detected, std::vector<X*> filter_by_subset_proportions, X* overall_filter)
198 {
199 Thresholds output;
200 std::vector<double> buffer(ncells);
201
202 // Filtering to remove outliers on the log-sum.
203 {
204 IsOutlier outliers;
205 outliers.set_nmads(nmads).set_upper(false).set_log(true);
206 auto res = fun(outliers, ncells, sums, filter_by_sums, buffer);
207 output.sums = res.lower;
208 std::copy(filter_by_sums, filter_by_sums + ncells, overall_filter);
209 }
210
211 // Filtering to remove outliers on the log-detected number.
212 {
213 IsOutlier outliers;
214 outliers.set_nmads(nmads).set_upper(false).set_log(true);
215 auto res = fun(outliers, ncells, detected, filter_by_detected, buffer);
216 output.detected = res.lower;
217 for (size_t i = 0; i < ncells; ++i) {
218 overall_filter[i] |= filter_by_detected[i];
219 }
220 }
221
222 // Filtering to remove outliers on the subset proportions.
223 {
224 size_t nsubsets = subset_proportions.size();
225 if (filter_by_subset_proportions.size() != nsubsets) {
226 throw std::runtime_error("mismatching number of input/outputs for subset proportion filters");
227 }
228
229 IsOutlier outliers;
230 outliers.set_nmads(nmads).set_lower(false);
231 output.subset_proportions.resize(nsubsets);
232
233 for (size_t s = 0; s < subset_proportions.size(); ++s) {
234 auto dump = filter_by_subset_proportions[s];
235 auto res = fun(outliers, ncells, subset_proportions[s], dump, buffer);
236 output.subset_proportions[s] = res.upper;
237
238 for (size_t i = 0; i < ncells; ++i) {
239 overall_filter[i] |= dump[i];
240 }
241 }
242 }

244 return output;
