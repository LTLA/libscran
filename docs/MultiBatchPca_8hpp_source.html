216 template<typename Data_, typename Index_, typename Block_>
217 void run_sparse_simple(
218 const tatami::Matrix<Data_, Index_>* mat,
219 const Block_* block,
220 const pca_utils::BlockingDetails<true> block_details,
221 const irlba::Irlba& irb,
222 Eigen::MatrixXd& pcs,
223 Eigen::MatrixXd& rotation,
224 Eigen::VectorXd& variance_explained,
225 Eigen::VectorXd& center_v,
226 Eigen::VectorXd& scale_v,
227 double& total_var)
228 const {
229 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
230 pca_utils::SparseMatrix emat(mat->ncol(), mat->nrow(), std::move(extracted.values), std::move(extracted.indices), std::move(extracted.ptrs), nthreads); // CSC with genes in columns.
231
232 size_t ngenes = emat.cols();
233 center_v.resize(ngenes);
234 scale_v.resize(ngenes);
235
236 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
237 const auto& values = emat.get_values();
238 const auto& indices = emat.get_indices();
239 const auto& ptrs = emat.get_pointers();
240
241 const auto& block_size = block_details.block_size;
242 size_t nblocks = block_size.size();
243 std::vector<int> block_count(nblocks);
244 const auto& block_weight = block_details.per_element_weight;
245
246 for (size_t r = start, end = start + length; r < end; ++r) {
247 auto offset = ptrs[r];
248 size_t num_entries = ptrs[r+1] - offset;
249 auto value_ptr = values.data() + offset;
250 auto index_ptr = indices.data() + offset;
251
252 std::fill(block_count.begin(), block_count.end(), 0);
253
254 // Computing the grand mean across all blocks.
255 double& grand_mean = center_v[r];
256 grand_mean = 0;
257 for (size_t i = 0; i < num_entries; ++i) {
258 auto b = block[index_ptr[i]];
259 grand_mean += value_ptr[i] * block_weight[b];
260 ++(block_count[b]);
261 }
262 grand_mean /= block_details.total_block_weight;
263
264 // Computing pseudo-variances where each block's contribution
265 // is weighted inversely proportional to its size. This aims to
266 // match up with the variances used in the PCA but not the
267 // variances of the output components (where weightings are not used).
268 double& proxyvar = scale_v[r];
269 proxyvar = 0;
270 for (size_t b = 0; b < nblocks; ++b) {
271 double zero_sum = (block_size[b] - block_count[b]) * grand_mean * grand_mean;
272 proxyvar += zero_sum * block_weight[b];
273 }
274
275 for (size_t i = 0; i < num_entries; ++i) {
276 double diff = value_ptr[i] - grand_mean;
277 proxyvar += diff * diff * block_weight[block[index_ptr[i]]];
278 }
279
280 proxyvar /= emat.rows() - 1;
281 }
282 }, ngenes, nthreads);
283
284 total_var = pca_utils::process_scale_vector(scale, scale_v);
285
286 // Now actually performing the PCA.
287 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
288 if (scale) {
289 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
290 pca_utils::SampleScaledWrapper<decltype(scaled)> weighted(&scaled, &(block_details.expanded_weights));
291 irb.run(weighted, pcs, rotation, variance_explained);
292 } else {
293 pca_utils::SampleScaledWrapper<decltype(centered)> weighted(&centered, &(block_details.expanded_weights));
294 irb.run(weighted, pcs, rotation, variance_explained);
295 }
296
297 // This transposes 'pcs' to be a NDIM * NCELLS matrix.
298 pca_utils::project_sparse_matrix(emat, pcs, rotation, scale, scale_v, nthreads);
299
300 pca_utils::clean_up_projected<true>(pcs, variance_explained);
301 if (!transpose) {
302 pcs.adjointInPlace();
303 }
304 }

306 template<typename Data_, typename Index_, typename Block_>
307 void run_dense_simple(
308 const tatami::Matrix<Data_, Index_>* mat,
309 const Block_* block,
310 const pca_utils::BlockingDetails<true>& block_details,
311 const irlba::Irlba& irb,
312 Eigen::MatrixXd& pcs,
313 Eigen::MatrixXd& rotation,
314 Eigen::VectorXd& variance_explained,
315 Eigen::VectorXd& center_v,
316 Eigen::VectorXd& scale_v,
317 double& total_var)
318 const {
319 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads); // row-major extraction.
320
321 size_t ngenes = emat.cols();
322 center_v.resize(ngenes);
323 scale_v.resize(ngenes);
324
325 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
326 size_t nblocks = block_details.num_blocks();
327 std::vector<double> mean_buffer(nblocks);
328 const auto& block_weight = block_details.per_element_weight;
329 size_t ncells = emat.rows();
330
331 for (size_t c = start, end = start + length; c < end; ++c) {
332 auto ptr = emat.data() + c * ncells;
333
334 double& grand_mean = center_v[c];
335 grand_mean = 0;
336 for (size_t r = 0; r < ncells; ++r) {
337 grand_mean += ptr[r] * block_weight[block[r]];
338 }
339 grand_mean /= block_details.total_block_weight;
340
341 // We don't actually compute the batchwise variance, but instead
342 // the weighted sum of squared deltas, which is what PCA actually sees.
343 double& proxyvar = scale_v[c];
344 proxyvar = 0;
345 for (size_t r = 0; r < ncells; ++r) {
346 double diff = ptr[r] - grand_mean;
347 proxyvar += diff * diff * block_weight[block[r]];
348 }
349
350 proxyvar /= emat.rows() - 1;
351 }
352 }, ngenes, nthreads);
353
354 total_var = pca_utils::process_scale_vector(scale, scale_v);
355
356 // Applying the centering and scaling now so we can do the PCA with fewer wrappers.
357 pca_utils::apply_center_and_scale_to_dense_matrix(emat, center_v, scale, scale_v, nthreads);
358
359 pca_utils::SampleScaledWrapper<decltype(emat)> weighted(&emat, &(block_details.expanded_weights));
360 irb.run(weighted, pcs, rotation, variance_explained);
361
362 pcs.noalias() = emat * rotation;
363 pca_utils::clean_up_projected<false>(pcs, variance_explained);
364 if (transpose) {
365 pcs.adjointInPlace();
366 }
367 }

371 void run_residuals_internal(
372 const Matrix_& emat,

447 template<bool weight_, typename Data_, typename Index_, typename Block_>
448 void run_dense_residuals(
449 const tatami::Matrix<Data_, Index_>* mat,
450 const Block_* block,
451 const pca_utils::BlockingDetails<weight_>& block_details,
452 const irlba::Irlba& irb,

454 Eigen::MatrixXd& rotation,
455 Eigen::VectorXd& variance_explained,
456 Eigen::MatrixXd& center_m,
457 Eigen::VectorXd& scale_v,
458 double& total_var)

469 run_residuals_internal<weight_>(

471 block,
472 block_details,
473 center_m,
474 scale_v,
475 irb,
476 pcs,
477 rotation,
478 variance_explained
479 );
480
481 if (scale) {
482 pcs.noalias() = emat * (rotation.array().colwise() / scale_v.array()).matrix();

494 template<typename Data_, typename Index_, typename Block_>
495 bool run_internal(
496 const tatami::Matrix<Data_, Index_>* mat,
497 const Block_* block,
498 Eigen::MatrixXd& pcs,
499 Eigen::MatrixXd& rotation,
500 Eigen::VectorXd& variance_explained,
501 Eigen::MatrixXd& center_m,
502 Eigen::VectorXd& scale_v,
503 double& total_var)
504 const {
505 irlba::EigenThreadScope t(nthreads);
506 irlba::Irlba irb;
507 irb.set_number(rank);
508
509 if (block_policy == BlockPolicy::RESIDUAL_ONLY) {
510 auto bdetails = pca_utils::compute_blocking_details<false>(mat->ncol(), block, weight_size_cap);
511 if (mat->sparse()) {
512 run_sparse_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
513 } else {
514 run_dense_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
515 }
516
517 } else {
518 auto bdetails = pca_utils::compute_blocking_details<true>(mat->ncol(), block, weight_size_cap);
519
520 if (block_policy == BlockPolicy::WEIGHTED_ONLY) {
521 Eigen::VectorXd center_v;
522 if (mat->sparse()) {
523 run_sparse_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_v, scale_v, total_var);
524 } else {
525 run_dense_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_v, scale_v, total_var);
526 }
527
528 if (return_center) {
529 center_m.resize(1, center_v.size());
530 center_m.row(0) = center_v;
531 }
532
533 } else {
534 if (mat->sparse()) {
535 run_sparse_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
536 } else {
537 run_dense_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
538 }
539 }
540 }

620 Results run(const tatami::Matrix<T, IDX>* mat, const Batch* batch) const {
