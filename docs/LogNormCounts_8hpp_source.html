111 std::shared_ptr<MAT> run_blocked(std::shared_ptr<MAT> mat, V size_factors, const B* block) {
112 // One might ask why we don't require a pointer for size_factors here.
113 // It's because size_factors need to be moved into the Delayed operation
114 // anyway, so we might as well ask the user to construct a vector for us.
115 if (size_factors.size() != mat->ncol()) {
116 throw std::runtime_error("number of size factors and columns are not equal");
117 }
118
119 if (center) {
120#ifdef SCRAN_LOGGER
121 SCRAN_LOGGER("scran::LogNormCounts", "Centering size factors to unity");
122#endif
123 // Falls back to centerer.run() if block=NULL.
124 centerer.run_blocked(size_factors.size(), size_factors.data(), block);
125 } else {
126 // CenterSizeFactors do their own validity checks,
127 // so we don't need to call it again in that case.
128 CenterSizeFactors::validate(size_factors.size(), size_factors.data());
129 }
130
131#ifdef SCRAN_LOGGER
132 SCRAN_LOGGER("scran::LogNormCounts", "Dividing each cell by its size factor");
133#endif
134 auto div = tatami::make_DelayedIsometricOp(mat, tatami::make_DelayedDivideVectorHelper<true, 1>(std::move(size_factors)));
135
136#ifdef SCRAN_LOGGER
137 SCRAN_LOGGER("scran::LogNormCounts", "Applying the log-transformation");
138#endif
139 if (pseudo_count == 1) {
140 return tatami::make_DelayedIsometricOp(div, tatami::DelayedLog1pHelper(2.0));
141 } else {
142 auto add = tatami::make_DelayedIsometricOp(div, tatami::DelayedAddScalarHelper<double>(pseudo_count));
143 return tatami::make_DelayedIsometricOp(add, tatami::DelayedLogHelper(2.0));
