152 template<typename T, typename IDX, typename Ref, typename Out>
153 void run(const tatami::Matrix<T, IDX>* mat, const Ref* ref, Out* output) const {
154 size_t NR = mat->nrow(), NC = mat->ncol();
155 Factory<T, Ref, Out> fact(NR, NC, ref, output);
156 tatami::apply<1>(mat, fact);
157
158 /* Mild squeezing towards library size-derived factors. Basically,
159 * we're adding a scaled version of the reference profile to each
160 * column, before normalizing against the reference profile. Given gene
161 * i and cell j:
162 *
163 * ratio_{ij} = (y_{ij} + ref_i * extra_j) / ref_i
164 * = (y_{ij} / ref_i) + extra_j
165 *
166 * which means that:
167 *
168 * median(ratio_{ij}) = median(y_{ij} / ref_i) + extra_j
169 *
170 * This allows us to avoid the actual addition, as we can compute the
171 * unshrunken size factor first and then add the j-specific scaling
172 * later. This is important as otherwise we'd need to make two passes
173 * over the matrix; once to get the mean library size to compute
174 * extra_j, and then again to compute the shrunken factors.
175 *
176 * Incidentally, extra_j is defined as:
177 *
178 * extra_j = p * t_j / T / R
179 *
180 * where p is the constant prior count, t_j is the library size for j,
181 * T is the mean library size across all j, and R is the library size
182 * for the reference profile. Basically, p * ref_i / R is how we
183 * "spread out" the prior count across all genes based on their
184 * relative abundance in the reference profile, while t_j / T
185 * represents the library size factor that we are shrinking towards.
186 *
187 * The addition of S_j means that the shrunken size factor is slightly
188 * too big to normalize against the reference. Assume that the
189 * unshrunken size factor captures the true scaling x_j for j against
190 * the reference, in which case the shrunken size factor would be:
191 *
192 * shrunk_j = median(ratio_{ij}) = x_j + extra_j
193 *
194 * To get shrunk_j = x_j, we need to divide it by:
195 *
196 * (x_j + extra_j) / x_j = 1 + (extra_j / x_j)
197 *
198 * As an approximation, we assume x_j =~ t_j / R, i.e., library size
199 * normalization against the reference. This allows us to simplify to:
200 *
201 * 1 + (p / T)
202 */
203 if (prior_count && NR && NC) {
204 const auto& sums = fact.sums;
205 double mean = std::accumulate(sums.begin(), sums.end(), static_cast<T>(0));
206 mean /= NC;
207
208 double reftotal = std::accumulate(ref, ref + NR, static_cast<double>(0));
209
210 if (mean && reftotal) {
211 double scaling = prior_count / mean;
212 for (size_t i = 0; i < NC; ++i) {
213 output[i] += sums[i] * scaling / reftotal;
214 output[i] /= 1.0 + scaling;
215 }
216 }
217 }
218
219 // Throwing in some centering.
220 if (center) {
221 CenterSizeFactors centerer;
222 centerer.run(NC, output);
223 }
224
225 return;
226 }
