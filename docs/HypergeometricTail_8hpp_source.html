96 // Using Stirling's approximation compute choose(num_black, num_drawn) / choose(num_black + num_white, num_drawn).
97 static double stirling(double x) {
98 return x * std::log(x) - x + 0.5 * std::log(2 * 3.14159265358979323846 * x);
99 }

101private:
102 /*
103 * Computing the cumulative sum after factorizing out choose(num_white + num_black, num_drawn), obviously.
104 *
105 * If num_drawn < num_black, we also factorize out choose(num_black, num_drawn), so that we can compute the series by simply increasing 'k'.
106 * This avoids the need to do two passes where the first pass computes choose(num_black, num_drawn - drawn_white) and the second pass handles the summation.
107 *
108 * Otherwise, we factorize out choose(num_white, num_drawn - num_black), which is the first non-zero probability mass in the series.
109 * This reduces the size of the summed terms to avoid overflow, and is reflected in 'start > 1'.
110 *
111 * We can check the accuracy of our calculations with:
112 * sum(choose(num_white, 0:drawn_white) * choose(num_black, num_drawn - 0:drawn_white)) / max(choose(num_white, num_drawn - num_black), choose(num_black, num_drawn)) - 1
113 */
114 static long double compute_cumulative(int drawn_white, int num_white, int num_black, int num_drawn) {
115 // We use long double for some extra precision here.
116 long double probability = 1;
117
118 // We need to add 1 for the starting term after our factorizations,
119 // but we'll do this in compute_tail_probability() to make use of the more precise log1p.
120 long double cumulative = 0;
121
122 int start = num_drawn - num_black + 1;
123 if (start <= 0) {
124 start = 1;
125 }
126
127 for (int k = start; k <= drawn_white; ++k) {
128 probability *= static_cast<double>(num_white - k + 1) * static_cast<double>(num_drawn - k + 1) / static_cast<double>(k) / static_cast<double>(num_black - num_drawn + k);
129 cumulative += probability;
130 }
131
132 return cumulative;
133 }

136 static void compute_cumulative(int drawn_white, int num_white, int num_black, int num_drawn, Cache& cache) {
137 auto& probability = cache.probability;
138 auto& cumulative = cache.cumulative;
139 auto& k = cache.drawn_white;
140
141 if (num_white == cache.num_white && num_black == cache.num_black && num_drawn == cache.num_drawn) {
142 if (k >= drawn_white) {
143 return;
144 }
145 } else {
146 cumulative.clear();
147 cumulative.push_back(0);
148 probability = 1;
149 k = 0;
150 }
151
152 cumulative.reserve(drawn_white + 1);
153
154 // Same logic as above, but we now store the probability at each step.
155 while (k < drawn_white) {
156 ++k;
157 if (num_drawn - k < num_black) {
158 probability *= static_cast<double>(num_white - k + 1) * static_cast<double>(num_drawn - k + 1) / static_cast<double>(k) / static_cast<double>(num_black - num_drawn + k);
159 cumulative.push_back(cumulative.back() + probability);
160 } else {
161 cumulative.push_back(0);
162 }
163 }

166 static double compute_log_scale(int num_white, int num_black, int num_drawn) {
167 int num_total = num_white + num_black;
168 if (num_black > num_drawn) {
169 // Approximates lchoose(num_black, num_drawn) - lchoose(num_total, num_drawn).
170 return stirling(num_black) - stirling(num_black - num_drawn)
171 - (stirling(num_total) - stirling(num_total - num_drawn));
172 } else if (num_drawn > num_black) {
173 // Approximates lchoose(num_white, num_drawn - num_black) - lchoose(num_total, num_drawn).
174 return stirling(num_white) - stirling(num_drawn - num_black)
175 - (stirling(num_total) - stirling(num_drawn));
176 } else {
177 // Approximates -lchoose(num_total, num_drawn).
178 return -(stirling(num_total) - stirling(num_total - num_drawn) - stirling(num_drawn));
179 }
180 }

183 static double invert_tail_log(double val) {
184 // Logic from https://github.com/SurajGupta/r-source/blob/master/src/nmath/dpq.h;
185 // if 'lp' is close to zero, exp(lp) will be close to 1, and thus the precision of
186 // expm1 is more important. If 'lp' is large and negative, exp(lp) will be close to
187 // zero, and thus the precision of log1p is more important.
188 if (val > -std::log(2)) {
189 auto p = -std::expm1(val);
190 return (p > 0 ? std::log(p) : -std::numeric_limits<double>::infinity());
191 } else {
192 auto p = -std::exp(val);
193 return (p > -1 ? std::log1p(p) : -std::numeric_limits<double>::infinity());
194 }
195 }

197 double compute_tail_probability(long double cumulative, double scale, bool do_lower) const {
198 if (log) {
199 double lp = std::log1p(cumulative) + scale;
200 if (do_lower) {
201 return lp;
202 } else {
203 return invert_tail_log(lp);
204 }
205 } else {
206 auto p = (1 + cumulative) * std::exp(scale);
207 return (do_lower ? p : 1 - p);
208 }
209 }

220 double core(int drawn_white, int num_white, int num_black, int num_drawn, Cache* cache) const {
221 // Subtracting 1 to include the PMF of 'drawn_white' in the upper tail calculations.
222 if (upper_tail) {
223 --drawn_white;
224 }
225
226 if (drawn_white <= 0 || drawn_white < num_drawn - num_black) {
227 return edge_handler(!upper_tail);
228 } else if (drawn_white >= num_drawn || drawn_white >= num_white) {
229 return edge_handler(upper_tail);
230 }
231
232 // Flipping the tails to avoid having to calculate large summations.
233 // While more efficient, the real reason is that it avoids having to
234 // subtract large sums from 1 when computing upper tails, which could
235 // result in catastrophic cancellation.
236 bool do_lower = !upper_tail;
237 if (drawn_white * (num_white + num_black) > num_drawn * num_white) {
238 auto tmp = num_white;
239 num_white = num_black;
240 num_black = tmp;
241 drawn_white = num_drawn - drawn_white - 1;
242 do_lower = !do_lower;
243 }
244
245 if (cache == NULL) {
246 double logscale = compute_log_scale(num_white, num_black, num_drawn);
247 auto cum = compute_cumulative(drawn_white, num_white, num_black, num_drawn);
248 return compute_tail_probability(cum, logscale, do_lower);
249 }
250
251 // Picking up from cached values if we can.
252 compute_cumulative(drawn_white, num_white, num_black, num_drawn, *cache);
253
254 double& logscale = cache->scale;
255 if (!(num_white == cache->num_white && num_black == cache->num_black && num_drawn == cache->num_drawn)) {
256 logscale = compute_log_scale(num_white, num_black, num_drawn);
257 cache->num_white = num_white;
258 cache->num_black = num_black;
259 cache->num_drawn = num_drawn;
260 }
261
262 return compute_tail_probability(cache->cumulative[drawn_white], cache->scale, do_lower);
263 }
