236 // I just can't be bothered to do anything fancier here, so this is what we've got.
237 igraph_rng_seed(igraph_rng_default(), 42);
238
239#ifdef SCRAN_LOGGER
240 SCRAN_LOGGER("scran::ClusterSNNGraph", "Performing multilevel community detection");
241#endif
242 MultiLevelResult output;
243 output.status = igraph_community_multilevel(&graph_info.graph, &graph_info.weights, resolution, &membership, &memberships, &modularity);
244
245 if (!output.status) {
246 output.max = igraph_vector_which_max(&modularity);
247
248 size_t nmods = igraph_vector_size(&modularity);
249 output.modularity.resize(nmods);
250 for (size_t i = 0; i < nmods; ++i) {
251 output.modularity[i] = VECTOR(modularity)[i];
252 }
253
254 size_t ncells = igraph_vcount(&graph_info.graph);
255 size_t nlevels = igraph_matrix_nrow(&memberships);
256 output.membership.resize(nlevels);
257
258 for (size_t i = 0; i < nlevels; ++i) {
259 auto& current = output.membership[i];
260 current.resize(ncells);
261 for (size_t j = 0; j < ncells; ++j) {
262 current[j] = MATRIX(memberships, i, j);
263 }
264 }
265 }
266
267 igraph_vector_destroy(&modularity);
268 igraph_vector_destroy(&membership);
269 igraph_matrix_destroy(&memberships);
270
271 return output;
