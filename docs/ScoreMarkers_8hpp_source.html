33// This cache tries to store as many of the reverse effects as possible before
34// it starts evicting. Evictions are based on the principle that it is better
35// to store effects that will be re-used quickly, thus freeing up the cache for
36// future stores. The 'speed' of reusability of each cache entry depends on the
37// first group in the comparison corresponding to each cached effect size; the
38// smaller the first group, the sooner it will be reached when iterating across
39// groups in the ScoreMarkers function.
40//
41// So, the policy is to evict cache entries when the identity of the first
42// group in the cached entry is larger than the identity of the first group for
43// the incoming entry. Given that, if the cache is full, we have to throw away
44// one of these effects anyway, I'd prefer to hold onto the one we're using
45// soon, because at least it'll be freed up rapidly.
46template<typename Stat>
47struct EffectsCacher {
48 EffectsCacher(size_t nge, int ngr, int cs) :
49 ngenes(nge),
50 ngroups(ngr),
51 cache_size(cs),
52 actions(ngroups),
53 staging_cache(ngroups)
54 {
55 vector_pool.reserve(cache_size);
56 }
57
58public:
59 size_t ngenes;
60 int ngroups;
61 int cache_size;
62
63 std::vector<CacheAction> actions;
64
65 // 'staging_cache' contains the set of cached effects in the other
66 // direction, i.e., all other groups compared to the current group. This is
67 // only used to avoid repeated look-ups in 'cached' while filling the
68 // effect size vectors; they will ultimately be transferred to cached after
69 // the processing for the current group is complete.
70 std::vector<std::vector<Stat> > staging_cache;
71
72 // 'vector_pool' allows us to recycle effect size vectors to avoid reallocations.
73 std::vector<std::vector<Stat> > vector_pool;
74
75 // 'cached' contains the cached effect size vectors from previous groups. Note
76 // that the use of a map is deliberate as we need the sorting.
77 std::map<std::pair<int, int>, std::vector<Stat> > cached;
78
79public:
80 void clear() {
81 cached.clear();
82 }

84public:
85 void configure(int group, std::vector<double>& full_set) {
86 // During calculation of effects, the current group (i.e., 'group') is
87 // the first group in the comparison and 'other' is the second group.
88 // However, remember that we want to cache the reverse effects, so in
89 // the cached entry, 'group' is second and 'other' is first.
90 for (int other = 0; other < ngroups; ++other) {
91 if (other == group) {
92 actions[other] = CacheAction::SKIP;
93 continue;
94 }
95
96 if (cache_size == 0) {
97 actions[other] = CacheAction::COMPUTE;
98 continue;
99 }
100
101 // If 'other' is later than 'group', it's a candidate to be cached,
102 // as it will be reused when the current group becomes 'other'.
103 if (other > group) {
104 actions[other] = CacheAction::CACHE;
105 continue;
106 }
107
108 // Need to recompute cache entries that were previously evicted. We
109 // do so if the cache is empty or the first group of the first cached
110 // entry has a higher index than the current group (and thus the
111 // desired comparison's effects cannot possibly exist in the cache).
112 if (cached.empty()) {
113 actions[other] = CacheAction::COMPUTE;
114 continue;
115 }
116
117 const auto& front = cached.begin()->first;
118 if (front.first > group || front.second > other) {
119 // Technically, the second clause should be (front.first == group && front.second > other).
120 // However, less-thans should be impossible as they should have been used up during processing
121 // of previous 'group' values. Thus, equality is already implied if the first condition fails.
122 actions[other] = CacheAction::COMPUTE;
123 continue;
124 }
125
126 // If we got past the previous clause, this implies that the first cache entry
127 // contains the effect sizes for the desired comparison (i.e., 'group - other').
128 // We thus transfer the cached vector to the full_set.
129 actions[other] = CacheAction::SKIP;
130 auto& x = cached.begin()->second;
131 for (size_t i = 0, end = x.size(); i < end; ++i) {
132 full_set[other + i * ngroups] = x[i];
133 }
134
135 vector_pool.emplace_back(std::move(x)); // recycle memory to avoid heap reallocations.
136 cached.erase(cached.begin());
137 }
138
139 // Refining our choice of cacheable entries by doing a dummy run and
140 // seeing whether eviction actually happens. If it doesn't, we won't
141 // bother caching, because that would be a waste of memory accesses.
142 for (int other = 0; other < ngroups; ++other) {
143 if (actions[other] != CacheAction::CACHE) {
144 continue;
145 }
146
147 std::pair<int, int> key(other, group);
148 if (cached.size() < static_cast<size_t>(cache_size)) {
149 cached[key] = std::vector<Stat>();
150 prepare_staging_cache(other);
151 continue;
152 }
153
154 // Looking at the last cache entry. If the first group of this
155 // entry is larger than the first group of the incoming entry, we
156 // evict it, as the incoming entry has faster reusability.
157 auto it = cached.end();
158 --it;
159 if ((it->first).first > other) {
160 auto& evicted = it->second;
161 if (evicted.size()) {
162 vector_pool.emplace_back(std::move(evicted));
163 }
164 cached.erase(it);
165 prepare_staging_cache(other);
166 cached[key] = std::vector<Stat>();
167 } else {
168 // Otherwise, if we're not going to do any evictions, we
169 // indicate that we shouldn't even bother computing the
170 // reverse, because we're won't cache the incoming entry.
171 actions[other] = CacheAction::COMPUTE;
172 }
173 }
174 }
175
176 void transfer(int group) {
177 for (int other = 0; other < ngroups; ++other) {
178 if (actions[other] != CacheAction::CACHE) {
179 continue;
180 }
181
182 // All the to-be-staged logic is already implemented in 'configure()'.
183 auto& staged = staging_cache[other];
184 std::pair<int, int> key(other, group);
185 cached[key] = std::move(staged);
186 }
187 }
188
189private:
190 void prepare_staging_cache(int other) {
191 if (vector_pool.empty()) {
192 staging_cache[other].resize(ngenes);
193 } else {
194 // Recycling existing heap allocations.
195 staging_cache[other] = std::move(vector_pool.back());
196 vector_pool.pop_back();
197 }
198 }
199};

713 void process_simple_effects(
714 size_t ngenes, // using size_t consistently here, to eliminate integer overflow bugs when computing products.
715 size_t ngroups,
716 size_t nblocks,
717 const differential_analysis::MatrixCalculator::State& state,
718 std::vector<Stat_*>& means,
719 std::vector<Stat_*>& detected,
720 std::vector<std::vector<Stat_*> >& cohen,
721 std::vector<std::vector<Stat_*> >& lfc,
722 std::vector<std::vector<Stat_*> >& delta_detected)
723 const {
724 const auto& level_size = state.level_size;
725 auto nlevels = level_size.size();
726 const auto* tmp_means = state.means.data();
727 const auto* tmp_variances = state.variances.data();
728 const auto* tmp_detected = state.detected.data();
729
730 // Transferring the block-wise statistics over.
731 {
732 auto my_means = tmp_means;
733 auto my_detected = tmp_detected;
734 for (size_t gene = 0; gene < ngenes; ++gene) {
735 for (size_t l = 0; l < nlevels; ++l, ++my_means, ++my_detected) {
736 means[l][gene] = *my_means;
737 detected[l][gene] = *my_detected;
738 }
739 }
740 }
741
742 // Looping over each group and computing the various summaries. We do this on
743 // a per-group basis to avoid having to store the full group-by-group matrix of
744 // effect sizes that we would otherwise need as input to SummarizeEffects.
745 differential_analysis::EffectsCacher<Stat_> cache(ngenes, ngroups, cache_size);
746 std::vector<double> full_set(ngroups * ngenes);
747
748 if (cohen.size()) {
749 cache.clear();
750 for (int group = 0; group < ngroups; ++group) {
751 cache.configure(group, full_set);
752
753 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
754 size_t in_offset = nlevels * start;
755 auto my_means = tmp_means + in_offset;
756 auto my_variances = tmp_variances + in_offset;
757
758 const auto& actions = cache.actions;
759 auto& staging_cache = cache.staging_cache;
760
761 auto cohen_ptr = full_set.data() + start * ngroups;
762 std::vector<double> effect_buffer(ngroups);
763
764 for (size_t gene = start, end = start + length; gene < end; ++gene, my_means += nlevels, my_variances += nlevels, cohen_ptr += ngroups) {
765 for (int other = 0; other < ngroups; ++other) {
766 if (actions[other] == differential_analysis::CacheAction::SKIP) {
767 continue;
768 }
769
770 if (actions[other] == differential_analysis::CacheAction::COMPUTE) {
771 cohen_ptr[other] = differential_analysis::compute_pairwise_cohens_d<false>(group, other, my_means, my_variances, level_size, ngroups, nblocks, threshold);
772 continue;
773 }
774
775 auto tmp = differential_analysis::compute_pairwise_cohens_d<true>(group, other, my_means, my_variances, level_size, ngroups, nblocks, threshold);
776 cohen_ptr[other] = tmp.first;
777 staging_cache[other][gene] = tmp.second;
778 }
779
780 differential_analysis::summarize_comparisons(ngroups, cohen_ptr, group, gene, cohen, effect_buffer);
781 }
782 }, ngenes, nthreads);
783
784 if (cohen[differential_analysis::summary::MIN_RANK].size()) {
785 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_set.data(), cohen[differential_analysis::summary::MIN_RANK][group], nthreads);
786 }
787
788 cache.transfer(group);
789 }
790 }
791
792 if (lfc.size()) {
793 cache.clear();
794 for (int group = 0; group < ngroups; ++group) {
795 cache.configure(group, full_set);
796
797 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
798 auto my_means = tmp_means + nlevels * start;
799
800 const auto& actions = cache.actions;
801 auto& staging_cache = cache.staging_cache;
802
803 auto lfc_ptr = full_set.data() + start * ngroups;
804 std::vector<double> effect_buffer(ngroups);
805
806 for (size_t gene = start, end = start + length; gene < end; ++gene, my_means += nlevels, lfc_ptr += ngroups) {
807 for (int other = 0; other < ngroups; ++other) {
808 if (actions[other] == differential_analysis::CacheAction::SKIP) {
809 continue;
810 }
811
812 auto val = differential_analysis::compute_pairwise_simple_diff(group, other, my_means, level_size, ngroups, nblocks);
813 lfc_ptr[other] = val;
814 if (actions[other] == differential_analysis::CacheAction::CACHE) {
815 staging_cache[other][gene] = -val;
816 }
817 }
818
819 differential_analysis::summarize_comparisons(ngroups, lfc_ptr, group, gene, lfc, effect_buffer);
820 }
821 }, ngenes, nthreads);
822
823 if (lfc[differential_analysis::summary::MIN_RANK].size()) {
824 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_set.data(), lfc[differential_analysis::summary::MIN_RANK][group], nthreads);
825 }
826
827 cache.transfer(group);
828 }
829 }
830
831 if (delta_detected.size()) {
832 cache.clear();
833 for (int group = 0; group < ngroups; ++group) {
834 cache.configure(group, full_set);
835
836 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
837 auto my_detected = tmp_detected + nlevels * start;
838
839 const auto& actions = cache.actions;
840 auto& staging_cache = cache.staging_cache;
841
842 auto delta_detected_ptr = full_set.data() + start * ngroups;
843 std::vector<double> effect_buffer(ngroups);
844
845 for (size_t gene = start, end = start + length; gene < end; ++gene, my_detected += nlevels, delta_detected_ptr += ngroups) {
846 for (int other = 0; other < ngroups; ++other) {
847 if (actions[other] == differential_analysis::CacheAction::SKIP) {
848 continue;
849 }
850
851 auto val = differential_analysis::compute_pairwise_simple_diff(group, other, my_detected, level_size, ngroups, nblocks);
852 delta_detected_ptr[other] = val;
853 if (actions[other] == differential_analysis::CacheAction::CACHE) {
854 staging_cache[other][gene] = -val;
855 }
856 }
857
858 differential_analysis::summarize_comparisons(ngroups, delta_detected_ptr, group, gene, delta_detected, effect_buffer);
859 }
860 }, ngenes, nthreads);
861
862 if (delta_detected[differential_analysis::summary::MIN_RANK].size()) {
863 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_set.data(), delta_detected[differential_analysis::summary::MIN_RANK][group], nthreads);
864 }
865
866 cache.transfer(group);
867 }
868 }

905 Results(
906 size_t ngenes,
907 int ngroups,
908 int nblocks,
909 const ComputeSummaries& do_cohen,
910 const ComputeSummaries& do_auc,
911 const ComputeSummaries& do_lfc,
912 const ComputeSummaries& do_delta_detected)
913 {
914 auto fill_inner = [&](int N, auto& type) {
915 type.reserve(N);
916 for (int n = 0; n < N; ++n) {
917 type.emplace_back(ngenes);
918 }
919 };
920
921 means.resize(ngroups);
922 detected.resize(ngroups);
923 for (int g = 0; g < ngroups; ++g) {
924 fill_inner(nblocks, means[g]);
925 fill_inner(nblocks, detected[g]);
926 }
927
928 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
929 bool has_any = false;
930 for (size_t i = 0; i < do_this.size(); ++i) {
931 if (do_this[i]) {
932 has_any = true;
933 break;
934 }
935 }
936
937 if (has_any) {
938 effect.resize(differential_analysis::n_summaries);
939 if (do_this[differential_analysis::MIN]) {
940 fill_inner(ngroups, effect[differential_analysis::MIN]);
941 }
942 if (do_this[differential_analysis::MEAN]) {
943 fill_inner(ngroups, effect[differential_analysis::MEAN]);
944 }
945 if (do_this[differential_analysis::MEDIAN]) {
946 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
947 }
948 if (do_this[differential_analysis::MAX]) {
949 fill_inner(ngroups, effect[differential_analysis::MAX]);
950 }
951 if (do_this[differential_analysis::MIN_RANK]) {
952 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
953 }
954 }
955 return;
956 };

959 fill_effect(do_auc, auc);
960 fill_effect(do_lfc, lfc);
