149 static std::vector<double> compute_scale(const std::vector<std::pair<double, double> >& distances) {
150 std::vector<double> output(distances.size());
151
152 // Use the first entry with a non-zero RMSD as the reference.
153 bool found_ref = false;
154 size_t ref = 0;
155 for (size_t e = 0; e < distances.size(); ++e) {
156 if (distances[e].second) {
157 found_ref = true;
158 ref = e;
159 break;
160 }
161 }
162
163 // If all of them have a zero RMSD, then all scalings are zero, because it doesn't matter.
164 if (found_ref) {
165 const auto& dref = distances[ref];
166 for (size_t e = 0; e < distances.size(); ++e) {
167 output[e] = (e == ref ? 1 : compute_scale(dref, distances[e]));
168 }
169 }
170
171 return output;
172 }

190 static double compute_scale(const std::pair<double, double>& ref, const std::pair<double, double>& target) {
191 if (target.first == 0 || ref.first == 0) {
192 if (target.second == 0) {
193 return std::numeric_limits<double>::infinity();
194 } else if (ref.second == 0) {
195 return 0;
196 } else {
197 return ref.second / target.second;
198 }
199 } else {
200 return ref.first / target.first;

231 size_t offset = 0;
232
233 for (size_t e = 0; e < nembed; ++e) {
234 size_t curdim = ndims[e];
235 auto inptr = embeddings[e];
236 auto outptr = output + offset;

238
239 if (std::isinf(s)) {
240 // If the scaling factor is infinite, it implies that the current
241 // embedding is all-zero, so we just fill with zeros, and move on.
242 for (size_t c = 0; c < ncells; ++c, inptr += curdim, outptr += ntotal) {
243 std::fill(outptr, outptr + curdim, 0);
244 }
245 } else {
246 for (size_t c = 0; c < ncells; ++c, inptr += curdim, outptr += ntotal) {
247 for (size_t d = 0; d < curdim; ++d) {
248 outptr[d] = inptr[d] * s;
