409 void compute_thresholds(size_t n, double* ptr, double& lower_threshold, double& upper_threshold) const {
410 if (!n) {
411 lower_threshold = std::numeric_limits<double>::quiet_NaN();
412 upper_threshold = std::numeric_limits<double>::quiet_NaN();
413 return;
414 }
415
416 if (log) {
417 auto copy = ptr;
418 for (size_t i = 0; i < n; ++i, ++copy) {
419 if (*copy > 0) {
420 *copy = std::log(*copy);
421 } else if (*copy == 0) {
422 *copy = -std::numeric_limits<double>::infinity();
423 } else {
424 throw std::runtime_error("cannot log-transform negative values");
425 }
426 }
427 }
428
429 // First, getting the median.
430 size_t halfway = n / 2;
431 double medtmp = compute_median(ptr, halfway, n);
432
433 if (log && std::isinf(medtmp)) {
434 if (lower) {
435 lower_threshold = 0;
436 } else {
437 lower_threshold = -std::numeric_limits<double>::infinity();
438 }
439
440 if (upper) {
441 upper_threshold = 0;
442 } else {
443 upper_threshold = std::numeric_limits<double>::infinity();
444 }
445
446 return;
447 }
448
449 // Now getting the MAD. No need to protect against -inf here;
450 // if the median is finite, the MAD must also be finite.
451 auto copy = ptr;
452 for (size_t i = 0; i < n; ++i, ++copy) {
453 *copy = std::abs(*copy - medtmp);
454 }
455 double madtmp = compute_median(ptr, halfway, n);
456
457 madtmp *= 1.4826; // for equivalence with the standard deviation under normality.
458 double delta = std::max(min_diff, nmads * madtmp);
459
460 // Computing the outputs.
461 if (lower) {
462 lower_threshold = medtmp - delta;
463 if (log) {
464 lower_threshold = std::exp(lower_threshold);
465 }
466 } else {
467 lower_threshold = -std::numeric_limits<double>::infinity();
468 }
469
470 if (upper) {
471 upper_threshold = medtmp + delta;
472 if (log) {
473 upper_threshold = std::exp(upper_threshold);
474 }
475 } else {
476 upper_threshold = std::numeric_limits<double>::infinity();
477 }
478
479 return;
480 }
481
482 static double compute_median (double* ptr, size_t halfway, size_t n) {
483 std::nth_element(ptr, ptr + halfway, ptr + n);
484 double medtmp = *(ptr + halfway);
485
486 if (n % 2 == 0) {
487 std::nth_element(ptr, ptr + halfway - 1, ptr + n);
488 medtmp = (medtmp + *(ptr + halfway - 1))/2;
489 }
490
491 return medtmp;
492 }
