297 void run_blocked(const MAT* mat, const B* block, std::vector<Stat*> means, std::vector<Stat*> variances, std::vector<Stat*> fitted, std::vector<Stat*> residuals) const {
298 size_t NR = mat->nrow(), NC = mat->ncol();
299 std::vector<typename MAT::index_type> block_size(means.size());
300
301 if (block) {
302 auto copy = block;
303 for (size_t j = 0; j < NC; ++j, ++copy) {
304 ++block_size[*copy];
305 }
306 compute<true>(mat, means, variances, block, block_size);
307 } else {
308 block_size[0] = NC;
309 compute<false>(mat, means, variances, block, block_size);
310 }
311
312 // Applying the trend fit to each block.
313 FitTrendVar fit;
314 fit.set_span(span);
315 fit.set_minimum_mean(min_mean);
316 fit.set_use_fixed_width(use_fixed_width);
317 fit.set_fixed_width(fixed_width);
318 fit.set_minimum_window_count(minimum_window_count);
319
320 for (size_t b = 0; b < block_size.size(); ++b) {
321 if (block_size[b] >= 2) {
322 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
323 } else {
324 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
325 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
326 }
327 }
328
329 return;
330 }
