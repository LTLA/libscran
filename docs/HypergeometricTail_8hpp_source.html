68 /*
69 * Computing the cumulative sum after factorizing out the probability mass at drawn_inside.
70 * This allows us to do one pass from k to 0 to compute the probability.
71 *
72 * We can check the accuracy of our calculations with:
73 * sum(choose(num_inside, 0:drawn_inside) * choose(num_outside, num_drawn - 0:drawn_inside)) / max(choose(num_inside, num_drawn - num_outside), choose(num_outside, num_drawn)) - 1
74 */
75 static long double compute_cumulative(int drawn_inside, int num_inside, int num_outside, int num_drawn) {
76 // Improved precision for this step, possibly involving small probabilities.
77 long double probability = 1;
78
79 // We need to add 1 for the probability mass at drawn_inside,
80 // but we'll do this in compute_tail_probability() to make use of the more precise log1p.
81 long double cumulative = 0;
82
83 for (int k = drawn_inside; k > 0 && probability > 0; --k) {
84 probability *= static_cast<double>(k) * static_cast<double>(num_outside - num_drawn + k) / static_cast<double>(num_inside - k + 1) / static_cast<double>(num_drawn - k + 1);
85 cumulative += probability;
86 }
87
88 return cumulative;
89 }

92 // Check out https://www.johndcook.com/blog/2012/09/25/ramanujans-factorial-approximation/.
93 static double lfactorial(double x) {
94 if (x == 0) {
95 return 0;
96 } else {
97 return 1.0/6.0 * std::log(x * (1 + 4 * x * (1 + 2 * x)) + 1.0/30.0) + x * std::log(x) - x + 0.5 * std::log(3.14159265358979323846);
98 }

102 int num_total = num_inside + num_outside;
103 return lfactorial(num_inside) - lfactorial(drawn_inside) - lfactorial(num_inside - drawn_inside) // lchoose(num_inside, drawn_inside)
104 + lfactorial(num_outside) - lfactorial(num_drawn - drawn_inside) - lfactorial(num_outside - num_drawn + drawn_inside) // lchoose(num_outside, num_drawn - drawn_inside)
105 - lfactorial(num_total) + lfactorial(num_drawn) + lfactorial(num_total - num_drawn); // -lchoose(num_total, num_drawn)

109 static double invert_tail_log(double val) {
110 // Logic from https://github.com/SurajGupta/r-source/blob/master/src/nmath/dpq.h;
111 // if 'lp' is close to zero, exp(lp) will be close to 1, and thus the precision of
112 // expm1 is more important. If 'lp' is large and negative, exp(lp) will be close to
113 // zero, and thus the precision of log1p is more important.
114 if (val > -std::log(2)) {
115 auto p = -std::expm1(val);
116 return (p > 0 ? std::log(p) : -std::numeric_limits<double>::infinity());
117 } else {
118 auto p = -std::exp(val);
119 return (p > -1 ? std::log1p(p) : -std::numeric_limits<double>::infinity());
120 }
121 }

123 double compute_tail_probability(long double cumulative, double scale, bool do_lower) const {
124 if (log) {
125 double lp = std::log1p(cumulative) + scale;
126 if (do_lower) {
127 return lp;
128 } else {
129 return invert_tail_log(lp);
130 }
131 } else {
132 auto p = (1 + cumulative) * std::exp(scale);
133 return (do_lower ? p : 1 - p);
134 }
135 }

146 double core(int drawn_inside, int num_inside, int num_outside, int num_drawn) const {
147 // Subtracting 1 to include the PMF of 'drawn_inside' in the upper tail calculations.
148 if (upper_tail) {
149 --drawn_inside;
150 }
151
152 if (drawn_inside <= 0 || drawn_inside < num_drawn - num_outside) {
153 return edge_handler(!upper_tail);
154 } else if (drawn_inside >= num_drawn || drawn_inside >= num_inside) {
155 return edge_handler(upper_tail);
156 }
157
158 // Flipping the tails to avoid having to calculate large summations.
159 // This it avoids having to subtract large sums from 1 when computing
160 // upper tails, which could result in catastrophic cancellation. It can
161 // also sometimes improve efficiency by summing over the smaller tail.
162 bool do_lower = !upper_tail;
163 if (drawn_inside * (num_inside + num_outside) > num_drawn * num_inside) {
164 auto tmp = num_inside;
165 num_inside = num_outside;
166 num_outside = tmp;
167 drawn_inside = num_drawn - drawn_inside - 1;
168 do_lower = !do_lower;
169 }
170
171 double logscale = compute_probability_mass(drawn_inside, num_inside, num_outside, num_drawn);
172 auto cum = compute_cumulative(drawn_inside, num_inside, num_outside, num_drawn);
173 return compute_tail_probability(cum, logscale, do_lower);
174 }
