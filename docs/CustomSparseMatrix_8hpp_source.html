39 void fill_columns(const std::vector<std::vector<double> >& values, const std::vector<std::vector<int> >& indices) {
40#ifdef TEST_SCRAN_CUSTOM_SPARSE_MATRIX
41 if (eigen) {
42 std::vector<int> nnzeros;
43 nnzeros.reserve(values.size());
44 for (const auto& v : values) {
45 nnzeros.push_back(v.size());
46 }
47
48 spmat.reserve(nnzeros);
49 for (size_t z = 0; z < values.size(); ++z) {
50 const auto& curi = indices[z];
51 const auto& curv = values[z];
52 for (size_t i = 0; i < curi.size(); ++i) {
53 spmat.insert(curi[i], z) = curv[i];
54 }
55 }
56 spmat.makeCompressed();
57 return;
58 }
59#endif
60
61 size_t nnzeros = 0;
62 auto pIt = p.begin() + 1;
63 for (const auto& v : values) {
64 nnzeros += v.size();
65 *pIt = nnzeros;
66 ++pIt;
67 }
68

70 for (const auto& v : values) {
71 x.insert(x.end(), v.begin(), v.end());
72 }
73
74 i.reserve(nnzeros);
75 for (const auto& v : indices) {
76 i.insert(i.end(), v.begin(), v.end());
77 }

120 }
121 }

132 void fragment_threads() {
133 double total_nzeros = p.back();
134 size_t per_thread = std::ceil(total_nzeros / nthreads);
135
136 // Splitting columns across threads so each thread processes the same number of nonzero elements.
137 col_starts.resize(nthreads);
138 col_ends.resize(nthreads);
139 {
140 size_t col_counter = 0;
141 size_t sofar = per_thread;
142 for (int t = 0; t < nthreads; ++t) {
143 col_starts[t] = col_counter;
144 while (col_counter < ncol && p[col_counter + 1] <= sofar) {
145 ++col_counter;
146 }
147 col_ends[t] = col_counter;
148 sofar += per_thread;
149 }
150 }
151
152 // Splitting rows across threads so each thread processes the same number of nonzero elements.
153 row_nonzero_starts.resize(nthreads + 1, std::vector<size_t>(ncol));
154 {
155 std::vector<size_t> row_nonzeros(nrow);
156 for (auto i_ : i) {
157 ++(row_nonzeros[i_]);
158 }
159
160 std::vector<size_t> row_starts(nthreads), row_ends(nthreads);
161 size_t row_counter = 0;
162 size_t sofar = per_thread;
163 size_t cum_rows = 0;
164
165 for (int t = 0; t < nthreads; ++t) {
166 row_starts[t] = row_counter;
167 while (row_counter < nrow && cum_rows <= sofar) {
168 cum_rows += row_nonzeros[row_counter];
169 ++row_counter;
170 }
171 row_ends[t] = row_counter;
172 sofar += per_thread;
173 }
174
175 for (size_t c = 0; c < ncol; ++c) {
176 size_t col_start = p[c], col_end = p[c + 1];
177 row_nonzero_starts[0][c] = col_start;
178
179 size_t s = col_start;
180 for (int thread = 0; thread < nthreads; ++thread) {
181 while (s < col_end && i[s] < row_ends[thread]) {
182 ++s;
183 }
184 row_nonzero_starts[thread + 1][c] = s;
185 }
186 }
187 }

276 double column_dot_product(size_t c, const Right& rhs) const {
277 size_t col_start = p[c], col_end = p[c + 1];
278 double dot = 0;
279 for (size_t s = col_start; s < col_end; ++s) {
