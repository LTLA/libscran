238 int nblocks = 1;
239 if (block) {
240 if (mat->ncol()) {
241 nblocks = *std::max_element(block, block + mat->ncol()) + 1;
242 }

244

246 std::vector<double*> mean_ptr, var_ptr, fit_ptr, resid_ptr;
247 for (int b = 0; b < nblocks; ++b) {
248 mean_ptr.push_back(output.means[b].data());
249 var_ptr.push_back(output.variances[b].data());
