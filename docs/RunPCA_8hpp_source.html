120 void run(const tatami::Matrix<T, IDX>* mat, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
121 irlba::EigenThreadScope t(nthreads);
122 irlba::Irlba irb;
123 irb.set_number(rank);
124
125 if (mat->sparse()) {
126 Eigen::VectorXd center_v(mat->nrow()), scale_v(mat->nrow());
127 auto emat = create_custom_sparse_matrix(mat, center_v, scale_v, total_var);
128
129 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
130 if (scale) {
131 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
132 irb.run(scaled, pcs, rotation, variance_explained);
133 } else {
134 irb.run(centered, pcs, rotation, variance_explained);
135 }
136 } else {
137 auto emat = create_eigen_matrix_dense(mat, total_var);
138 irb.run(emat, pcs, rotation, variance_explained); // already centered and scaled, if relevant.
139 }
140
141 pca_utils::clean_up(mat->ncol(), pcs, variance_explained);
142 if (transpose) {
143 pcs.adjointInPlace();
144 }
145
146 return;
147 }
