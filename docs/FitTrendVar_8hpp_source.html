134 void run(size_t n, const double* mean, const double* variance, double* fitted, double* residuals) const {
135 std::vector<double> xbuffer(n), ybuffer(n);
136
137 size_t counter = 0;
138 for (size_t i = 0; i < n; ++i) {
139 if (!filter || mean[i] >= min_mean) {
140 xbuffer[counter] = mean[i];
141 if (transform) {
142 ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
143 } else {
144 ybuffer[counter] = variance[i];
145 }
146 ++counter;
147 }
148 }
149
150 if (counter < 2) {
151 throw std::runtime_error("not enough observations above the minimum mean");
152 }
153
154 // Determining the left edge. This needs to be done before
155 // run_in_place, which mutates the input xbuffer.
156 size_t left_index = std::min_element(xbuffer.begin(), xbuffer.begin() + counter) - xbuffer.begin();
157 double left_x = xbuffer[left_index];
158
159 WeightedLowess::WeightedLowess<> smoother;
160 smoother.set_span(span);
161
162 std::vector<double> fbuffer(counter), rbuffer(counter);
163 smoother.run(counter, xbuffer.data(), ybuffer.data(), NULL, fbuffer.data(), rbuffer.data());
164
165 // Identifying the left-most fitted value.
166 double left_fitted = (transform ? quad(fbuffer[left_index]) : fbuffer[left_index]);
167
168 counter = 0;
169 for (size_t i = 0; i < n; ++i) {
170 if (!filter || mean[i] >= min_mean) {
171 fitted[i] = (transform ? quad(fbuffer[counter]) : fbuffer[counter]);
172 ++counter;
173 } else {
174 fitted[i] = mean[i] / left_x * left_fitted; // draw a y = x line to the origin from the left of the fitted trend.
175 }
176 residuals[i] = variance[i] - fitted[i];
177 }
178 return;
179 }
