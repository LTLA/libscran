226 template<class MAT, class V, typename B>
227 std::shared_ptr<MAT> run_blocked(std::shared_ptr<MAT> mat, V size_factors, const B* block) const {
228 // One might ask why we don't require a pointer for size_factors here.
229 // It's because size_factors need to be moved into the Delayed operation
230 // anyway, so we might as well ask the user to construct a vector for us.
231 if (size_factors.size() != mat->ncol()) {
232 throw std::runtime_error("number of size factors and columns are not equal");
233 }
234
235 bool has_zero = false;
236 if (center) {
237 // Falls back to centerer.run() if block=NULL.
238 has_zero = centerer.run_blocked(size_factors.size(), size_factors.data(), block);
239 } else {
240 // CenterSizeFactors do their own validity checks,
241 // so we don't need to call it again in that case.
242 has_zero = CenterSizeFactors::validate(size_factors.size(), size_factors.data());
243 }
244
245 if (has_zero) {
246 if (!handle_zeros) {
247 throw std::runtime_error("all size factors should be positive");
248 } else if (size_factors.size()) {
249 // Replacing them with the smallest non-zero size factor, or 1.
250 auto smallest = std::numeric_limits<double>::infinity();
251 for (auto s : size_factors) {
252 if (s && smallest > s) {
253 smallest = s;
254 }
255 }
256
257 if (std::isinf(smallest)) {
258 smallest = 1;
259 }
260
261 for (auto& s : size_factors) {
262 if (s == 0) {
263 s = smallest;
264 }
265 }
266 }
267 }
268
269 double current_pseudo = pseudo_count;
270 if (choose_pseudo_count) {
271 current_pseudo = pseudo_chooser.run(size_factors.size(), size_factors.data());
272 }
273
274 if (sparse_addition && current_pseudo != 1) {
275 for (auto& d : size_factors) {
276 d *= current_pseudo;
277 }
278 current_pseudo = 1; // effectively 1 now.
279 }
280
281 auto div = tatami::make_DelayedIsometricOp(mat, tatami::make_DelayedDivideVectorHelper<true, 1>(std::move(size_factors)));
282
283 if (current_pseudo == 1) {
284 return tatami::make_DelayedIsometricOp(div, tatami::DelayedLog1pHelper(2.0));
285 } else {
286 auto add = tatami::make_DelayedIsometricOp(div, tatami::DelayedAddScalarHelper<double>(current_pseudo));
287 return tatami::make_DelayedIsometricOp(add, tatami::DelayedLogHelper(2.0));
288 }
289 }
