27template<class Matrix, typename Block>
28struct BlockedEigenMatrix {
29 BlockedEigenMatrix(const Matrix* m, const Block* b, const Eigen::MatrixXd* mx) : mat(m), block(b), means(mx) {}
30
31 auto rows() const { return mat->rows(); }
32 auto cols() const { return mat->cols(); }
33
34 template<class Right>
35 void multiply(const Right& rhs, Eigen::VectorXd& output) const {
36 output.noalias() = (*mat) * rhs;
37 Eigen::MatrixXd sub = (*means) * rhs;
38 for (Eigen::Index i = 0; i < output.size(); ++i) {
39 output[i] -= sub(block[i], 0);
40 }
41 return;
42 }
43
44 template<class Right>
45 void adjoint_multiply(const Right& rhs, Eigen::VectorXd& output) const {
46 output.noalias() = mat->adjoint() * rhs;
47
48 Eigen::VectorXd aggr(means->rows());
49 aggr.setZero();
50 for (Eigen::Index i = 0; i < rhs.size(); ++i) {
51 aggr[block[i]] += rhs[i];
52 }
53
54 output.noalias() -= means->adjoint() * aggr;
55 return;
56 }
57
58 Eigen::MatrixXd realize() const {
59 Eigen::MatrixXd output(*mat);
60 for (Eigen::Index c = 0; c < output.cols(); ++c) {
61 for (Eigen::Index r = 0; r < output.rows(); ++r) {
62 output(r, c) -= (*means)(block[r], c);
63 }
64 }
65 return output;
66 }
67private:
68 const Matrix* mat;
69 const Block* block;
70 const Eigen::MatrixXd* means;
71};

136 template<typename T, typename IDX, typename Block>
137 void run(const tatami::Matrix<T, IDX>* mat, const Block* block, Eigen::MatrixXd& pcs, Eigen::VectorXd& variance_explained, double& total_var) {
138 const size_t NC = mat->ncol();
139 const int nblocks = (NC ? *std::max_element(block, block + NC) + 1 : 1);
140 std::vector<int> block_size(nblocks);
141 for (size_t j = 0; j < NC; ++j) {
142 ++block_size[block[j]];
143 }
144
145 if (mat->sparse()) {
146 Eigen::MatrixXd center_m(nblocks, mat->nrow());
147 Eigen::VectorXd scale_v(mat->nrow());
148 auto emat = create_eigen_matrix_sparse(mat, center_m, scale_v, block, block_size, total_var);
149
150 BlockedEigenMatrix<decltype(emat), Block> centered(&emat, block, &center_m);
151 if (scale) {
152 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
153 auto result = irb.run(scaled);
154 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
155 } else {
156 auto result = irb.run(centered);
157 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
158 }
159 } else {
160 auto emat = create_eigen_matrix_dense(mat, block, block_size, total_var);
161 auto result = irb.run(emat); // already centered and scaled, if relevant.
162 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
163 }
164
165 return;
166 }

250 template<typename T, typename IDX, typename Block>
251 Eigen::SparseMatrix<double> create_eigen_matrix_sparse(const tatami::Matrix<T, IDX>* mat,
252 Eigen::MatrixXd& center_m,
253 Eigen::VectorXd& scale_v,
254 const Block* block,
255 const std::vector<int>& block_size,
256 double& total_var)
257 {
258 size_t NR = mat->nrow(), NC = mat->ncol();
259 total_var = 0;
260
261 Eigen::SparseMatrix<double> A(NC, NR); // transposed; we want genes in the columns.
262 std::vector<std::vector<double> > values;
263 std::vector<std::vector<int> > indices;
264
265 const size_t nblocks = block_size.size();
266 double * mbuffer = center_m.data();
267
268#ifdef SCRAN_LOGGER
269 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
270#endif
271
272 if (mat->prefer_rows()) {
273 std::vector<double> xbuffer(NC);
274 std::vector<int> ibuffer(NC);
275 std::vector<int> nnzeros(NR);
276 values.reserve(NR);
277 indices.reserve(NR);
278
279 for (size_t r = 0; r < NR; ++r, mbuffer += nblocks) {
280 auto range = mat->sparse_row(r, xbuffer.data(), ibuffer.data());
281
282 // Computing the block-wise means.
283 std::fill(mbuffer, mbuffer + nblocks, 0);
284 for (size_t i = 0; i < range.number; ++i) {
285 mbuffer[block[range.index[i]]] += range.value[i];
286 }
287 for (size_t b = 0; b < nblocks; ++b) {
288 mbuffer[b] /= block_size[b];
289 }
290
291 // Computing the variance from the sum of squared differences.
292 // This is technically not the correct variance estimate if
293 // we were to consider the blocks, but it's what the PCA sees.
294 double& proxyvar = scale_v[r];
295 proxyvar = 0;
296 {
297 auto block_copy = block_size;
298 for (size_t i = 0; i < range.number; ++i) {
299 Block curb = block[range.index[i]];
300 double diff = range.value[i] - mbuffer[curb];
301 proxyvar += diff * diff;
302 --block_copy[curb];
303 }
304
305 for (size_t b = 0; b < nblocks; ++b) {
306 proxyvar += mbuffer[b] * mbuffer[b] * block_copy[b];
307 }
308
309 proxyvar /= NC - 1;
310 }
311
312 nnzeros[r] = range.number;
313 values.emplace_back(range.value, range.value + range.number);
314 indices.emplace_back(range.index, range.index + range.number);
315 }
316
317 pca_utils::set_scale(scale, scale_v, total_var);
318 pca_utils::fill_sparse_matrix<true>(A, indices, values, nnzeros);
319 } else {
320 std::vector<double> xbuffer(NR);
321 std::vector<int> ibuffer(NR);
322 values.reserve(NC);
323 indices.reserve(NC);
324 std::vector<int> nnzeros(NR);
325
326 std::vector<std::vector<double> > tmp_means(nblocks, std::vector<double>(NR));
327 std::vector<std::vector<int> > tmp_nonzeros(nblocks, std::vector<int>(NR));
328
329 for (size_t c = 0; c < NC; ++c) {
330 auto range = mat->sparse_column(c, xbuffer.data(), ibuffer.data());
331 values.emplace_back(range.value, range.value + range.number);
332 indices.emplace_back(range.index, range.index + range.number);
333
334 // Collecting values for the means.
335 Block curb = block[c];
336 auto& cur_means = tmp_means[curb];
337 auto& cur_nonzeros = tmp_nonzeros[curb];
338 for (size_t i = 0; i < range.number; ++i) {
339 auto r = range.index[i];
340 cur_means[r] += range.value[i];
341 ++cur_nonzeros[r];
342 ++nnzeros[r];
343 }
344 }
345
346 // Finalizing the means.
347 for (size_t b = 0; b < nblocks; ++b) {
348 auto& cur_means = tmp_means[b];
349 for (size_t r = 0; r < cur_means.size(); ++r) {
350 cur_means[r] /= block_size[b];
351 center_m(b, r) = cur_means[r];
352 }
353 }
354
355 // Computing the pseudo-variances for each gene.
356 {
357 scale_v.setZero();
358
359 for (size_t c = 0; c < NC; ++c) {
360 const auto& cur_means = tmp_means[block[c]];
361 const auto& cur_vals = values[c];
362 const auto& cur_idx = indices[c];
363
364 for (size_t i = 0; i < cur_idx.size(); ++i) {
365 auto r = cur_idx[i];
366 double diff = cur_vals[i] - cur_means[r];
367 scale_v[r] += diff * diff;
368 }
369 }
370
371 for (size_t b = 0; b < nblocks; ++b) {
372 const auto& cur_means = tmp_means[b];
373 const auto& cur_nonzeros = tmp_nonzeros[b];
374 for (size_t r = 0; r < NR; ++r) {
375 scale_v[r] += cur_means[r] * cur_means[r] * (block_size[b] - cur_nonzeros[r]);
376 }
377 }
378
379 for (size_t r = 0; r < NR; ++r) {
380 scale_v[r] /= NC - 1;
381 }
382 }
383
384 pca_utils::set_scale(scale, scale_v, total_var);
385 pca_utils::fill_sparse_matrix<false>(A, indices, values, nnzeros);
386 }
387
388 A.makeCompressed();
389 return A;
390 }

393 template<typename T, typename IDX, typename Block>
394 Eigen::MatrixXd create_eigen_matrix_dense(
395 const tatami::Matrix<T, IDX>* mat,
396 const Block* block,
397 const std::vector<int>& block_size,
398 double& total_var)
399 {
400 size_t NR = mat->nrow(), NC = mat->ncol();
401 total_var = 0;
402
403 Eigen::MatrixXd output(NC, NR); // transposed.
404 std::vector<double> xbuffer(NC);
405 double* outIt = output.data();
406
407 int nblocks = block_size.size();
408 std::vector<double> mean_buffer(nblocks);
409
410#ifdef SCRAN_LOGGER
411 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
412#endif
413
414 for (size_t r = 0; r < NR; ++r, outIt += NC) {
415 auto ptr = mat->row_copy(r, outIt);
416
417 // Compute the means of each block.
418 std::fill(mean_buffer.begin(), mean_buffer.end(), 0);
419 {
420 auto copy = outIt;
421 auto bcopy = block;
422 for (size_t c = 0; c < NC; ++c, ++copy, ++bcopy) {
423 mean_buffer[*bcopy] += *copy;
424 }
425 }
426 for (int b = 0; b < nblocks; ++b) {
427 if (block_size[b]) {
428 mean_buffer[b] /= block_size[b];
429 }
430 }
431
432 // We don't actually compute the blockwise variance, but instead
433 // the squared sum of deltas, which is what PCA actually sees.
434 double proxyvar = 0;
435 {
436 auto copy = outIt;
437 auto bcopy = block;
438 for (size_t c = 0; c < NC; ++c, ++copy, ++bcopy) {
439 *copy -= mean_buffer[*bcopy];
440 proxyvar += *copy * *copy;
441 }
442 proxyvar /= NC - 1;
443 }
444
445 pca_utils::apply_scale(scale, proxyvar, NC, outIt, total_var);
446 }
447
448 return output;
449 }
