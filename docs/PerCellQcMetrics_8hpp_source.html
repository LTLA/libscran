356 void compute(size_t c, const tatami::SparseRange<T, IDX>& range) {
357 if (output.total) {
358 auto& current = output.total[c];
359 for (size_t i = 0; i < range.number; ++i) {
360 current += range.value[i];
361 }
362 }
363
364 if (output.detected) {
365 auto& current = output.detected[c];
366 for (size_t i = 0; i < range.number; ++i) {
367 current += (range.value[i] != 0);
368 }
369 }
370
371 if (output.max_index || output.max_count) {
372 auto max_count = PerCellQcMetrics::pick_fill_value<Float>();
373 Integer max_index = 0;
374 for (size_t i = 0; i < range.number; ++i) {
375 if (max_count < range.value[i]) {
376 max_count = range.value[i];
377 max_index = range.index[i];
378 }
379 }
380
381 if (max_count <= 0 && range.number < NR) {
382 // Zero is the max.
383 max_count = 0;
384
385 // Finding the index of the first zero by tracking all
386 // indices with non-zero values. This isn't the fastest
387 // approach but it's simple and avoids assuming that
388 // indices are sorted. Hopefully we don't have to hit
389 // this section often.
390 if (output.max_index) {
391 for (size_t i = 0; i < range.number; ++i) {
392 if (range.value[i]) {
393 internal_is_nonzero[range.index[i]] = 1;
394 }
395 }
396 for (size_t r = 0; r < NR; ++r) {
397 if (internal_is_nonzero[r] == 0) {
398 max_index = r;
399 break;
400 }
401 }
402 for (size_t i = 0; i < range.number; ++i) { // setting back to zero.
403 internal_is_nonzero[range.index[i]] = 0;
404 }
405 }
406 }
407
408 if (output.max_index) {
409 output.max_index[c] = max_index;
410 }
411 if (output.max_count) {
412 output.max_count[c] = max_count;
413 }
414 }
415
416 size_t nsubsets = subsets_ptr->size();
417 for (size_t s = 0; s < nsubsets; ++s) {
418 const auto& sub = (*subsets_ptr)[s];
419
420 if (output.subset_total[s]) {
421 auto& current = output.subset_total[s][c];
422 for (size_t i = 0; i < range.number; ++i) {
423 current += (sub[range.index[i]] != 0) * range.value[i];
424 }
425 }
426
427 if (output.subset_detected[s]) {
428 auto& current = output.subset_detected[s][c];
429 for (size_t i = 0; i < range.number; ++i) {
430 current += (sub[range.index[i]] != 0) * (range.value[i] != 0);

433 }

663 template<class Matrix, typename Subset = const uint8_t*, typename Float, typename Integer>
664 void run(const Matrix* mat, const std::vector<Subset>& subsets, Buffers<Float, Integer> output) const {
665 if (!output.already_zeroed) {
666 size_t n = mat->ncol();
667 auto check_and_fill = [&](auto* ptr, auto value) -> void {
668 if (ptr) {
669 std::fill(ptr, ptr + n, value);
670 }
671 };
672
673 check_and_fill(output.total, static_cast<Float>(0));
674 check_and_fill(output.detected, static_cast<Integer>(0));
675 check_and_fill(output.max_count, pick_fill_value<Float>());
676 check_and_fill(output.max_index, static_cast<Integer>(0));
677
678 for (size_t s = 0; s < subsets.size(); ++s) {
679 check_and_fill(output.subset_total[s], static_cast<Float>(0));
680 check_and_fill(output.subset_detected[s], static_cast<Integer>(0));
681 }
682 }
683
684 size_t nr = mat->nrow(), nc = mat->ncol();
685 Factory fact(nr, nc, &subsets, std::move(output));
686 tatami::apply<1>(mat, fact, num_threads);
687 return;
688 }
