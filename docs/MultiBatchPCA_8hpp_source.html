27template<class Matrix>
28struct MultiBatchEigenMatrix {
29 MultiBatchEigenMatrix(const Matrix* m, const Eigen::VectorXd* w, const Eigen::VectorXd* mx) : mat(m), weights(w), means(mx) {}
30
31 auto rows() const { return mat->rows(); }
32 auto cols() const { return mat->cols(); }
33
34 template<class Right>
35 void multiply(const Right& rhs, Eigen::VectorXd& output) const {
36 if constexpr(irlba::has_multiply_method<Matrix>::value) {
37 output.noalias() = *mat * rhs;
38 } else {
39 mat->multiply(rhs, output);
40 }
41
42 double sub = means->dot(rhs);
43 for (Eigen::Index i = 0; i < output.size(); ++i) {
44 auto& val = output.coeffRef(i);
45 val -= sub;
46 val *= weights->coeff(i);
47 }
48 }
49
50 template<class Right>
51 void adjoint_multiply(const Right& rhs, Eigen::VectorXd& output) const {
52 Eigen::VectorXd combined = weights->cwiseProduct(rhs);
53
54 if constexpr(irlba::has_adjoint_multiply_method<Matrix>::value) {
55 output.noalias() = mat->adjoint() * combined;
56 } else {
57 mat->adjoint_multiply(combined, output);
58 }
59
60 double sum = combined.sum();
61 for (Eigen::Index i = 0; i < output.size(); ++i) {
62 output.coeffRef(i) -= means->coeff(i) * sum;
63 }
64 return;
65 }
66
67 Eigen::MatrixXd realize() const {
68 Eigen::MatrixXd output;
69 if constexpr(irlba::has_realize_method<Matrix>::value) {
70 output = mat->realize();
71 } else {
72 output = Eigen::MatrixXd(*mat);
73 }
74
75 for (Eigen::Index c = 0; c < output.cols(); ++c) {
76 for (Eigen::Index r = 0; r < output.rows(); ++r) {
77 auto& val = output.coeffRef(r, c);
78 val -= means->coeff(c);
79 val *= weights->coeff(r);
80 }
81 }
82 return output;
83 }
84private:
85 const Matrix* mat;
86 const Eigen::VectorXd* weights;
87 const Eigen::VectorXd* means;
88};

183 static Eigen::VectorXd compute_weights(size_t NC, const Batch* batch, const std::vector<int>& batch_size) {
184 Eigen::VectorXd weights(NC);
185 for (size_t i = 0; i < NC; ++i) {
186 weights[i] = 1/std::sqrt(static_cast<double>(batch_size[batch[i]]));

224 template<class Matrix, class Rotation>
225 void reapply(const Matrix& emat,
226 const Eigen::VectorXd& center_v,
227 Eigen::MatrixXd& pcs,
228 const Rotation& rotation,
229 Eigen::VectorXd& variance_explained)
230 const {
231 if constexpr(!std::is_same<Matrix, pca_utils::SparseMatrix>::value) {
232 pcs.noalias() = emat * rotation;
233 } else {
234 size_t nvec = pcs.cols();
235 size_t nrow = emat.rows();
236 size_t ncol = emat.cols();
237
238 Eigen::MatrixXd transposed(nvec, nrow); // using a transposed version for more cache efficiency.
239 transposed.setZero();
240
241 const auto& x = emat.get_values();
242 const auto& i = emat.get_indices();
243 const auto& p = emat.get_pointers();
244
245 if (nthreads == 1) {
246 Eigen::VectorXd multipliers(nvec);
247 for (size_t c = 0; c < ncol; ++c) {
248 multipliers.noalias() = rotation.row(c);
249 auto start = p[c], end = p[c + 1];
250 for (size_t s = start; s < end; ++s) {
251 transposed.col(i[s]).noalias() += x[s] * multipliers;
252 }
253 }
254 } else {
255 const auto& row_nonzero_starts = emat.get_secondary_nonzero_starts();
256
257#ifndef SCRAN_CUSTOM_PARALLEL
258 #pragma omp parallel for num_threads(nthreads)
259 for (int t = 0; t < nthreads; ++t) {
260#else
261 SCRAN_CUSTOM_PARALLEL(nthreads, [&](int first, int last) -> void {
262 for (int t = first; t < last; ++t) {
263#endif
264
265 const auto& starts = row_nonzero_starts[t];
266 const auto& ends = row_nonzero_starts[t + 1];
267 Eigen::VectorXd multipliers(nvec);
268 for (size_t c = 0; c < ncol; ++c) {
269 multipliers.noalias() = rotation.row(c);
270 auto start = starts[c], end = ends[c];
271 for (size_t s = start; s < end; ++s) {
272 transposed.col(i[s]).noalias() += x[s] * multipliers;
273 }
274 }
275
276#ifndef SCRAN_CUSTOM_PARALLEL
277 }
278#else
279 }
280 }, nthreads);
281#endif
282 }
283
284 pcs.noalias() = transposed.adjoint();
285 }
286
287 // Effective centering because I don't want to modify 'emat'.
288 auto pIt = pcs.data();
289 for (size_t i = 0, iend = pcs.cols(); i < iend; ++i) {
290 double meanval = center_v.dot(rotation.col(i));
291 for (size_t j = 0, jend = pcs.rows(); j < jend; ++j, ++pIt) {
292 *pIt -= meanval;
293 }
294 }
295
296 // Variance is a somewhat murky concept with weights, so we just square
297 // it and assume that only the relative value matters.
298 for (auto& d : variance_explained) {
299 d = d * d;
300 }
301 return;
302 }

305 template<typename T, typename IDX, typename Batch>
306 void run(const tatami::Matrix<T, IDX>* mat, const Batch* batch, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
307 const size_t NC = mat->ncol();
308 auto batch_size = block_sizes(NC, batch);
309 auto weights = compute_weights(NC, batch, batch_size);
310
311 Eigen::VectorXd center_v(mat->nrow());
312 Eigen::VectorXd scale_v(mat->nrow());
313
314 // Remember, we want to run the PCA on the modified matrix,
315 // but we want to apply the rotation vectors to the original matrix
316 // (after any centering/scaling but without the batch weights).
317 auto executor = [&](const auto& emat) -> void {
318 irlba::EigenThreadScope t(nthreads);
319 irlba::Irlba irb;
320 irb.set_number(rank);
321
322 MultiBatchEigenMatrix<typename std::remove_reference<decltype(emat)>::type> thing(&emat, &weights, &center_v);
323 if (scale) {
324 irb.run(irlba::Scaled<decltype(thing)>(&thing, &scale_v), pcs, rotation, variance_explained);
325
326 // Dividing the rotation vectors by the scaling factor to mimic
327 // the division of 'emat' by the scaling factor (after centering).
328 Eigen::MatrixXd temp = rotation.array().colwise() / scale_v.array();
329 reapply(emat, center_v, pcs, temp, variance_explained);
330 } else {
331 irb.run(thing, pcs, rotation, variance_explained);
332 reapply(emat, center_v, pcs, rotation, variance_explained);
333 }
334 };
335
336 if (mat->sparse()) {
337 auto emat = create_custom_sparse_matrix(mat, center_v, scale_v, batch, batch_size, total_var);
338 executor(emat);
339 } else {
340 auto emat = create_eigen_matrix_dense(mat, center_v, scale_v, batch, batch_size, total_var);
341 executor(emat);
342 }
343
344 if (transpose) {
345 pcs.adjointInPlace();
346 }
347
348 return;
349 }

445 template<typename T, typename IDX, typename Batch>
446 pca_utils::SparseMatrix create_custom_sparse_matrix(const tatami::Matrix<T, IDX>* mat,
447 Eigen::VectorXd& center_v,
448 Eigen::VectorXd& scale_v,
449 const Batch* batch,
450 const std::vector<int>& batch_size,
451 double& total_var)
452 const {
453
454 size_t NR = mat->nrow(), NC = mat->ncol();
455 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
456 auto& ptrs = extracted.ptrs;
457 auto& values = extracted.values;
458 auto& indices = extracted.indices;
459
460 // Computing grand means and variances with weights.
461 {
462 size_t nbatchs = batch_size.size();
463
464#ifndef SCRAN_CUSTOM_PARALLEL
465 #pragma omp parallel num_threads(nthreads)
466 {
467#else
468 SCRAN_CUSTOM_PARALLEL(NR, [&](size_t first, size_t last) -> void {
469#endif
470
471 std::vector<double> batch_means(nbatchs);
472 std::vector<int> batch_count(nbatchs);
473
474#ifndef SCRAN_CUSTOM_PARALLEL
475 #pragma omp for
476 for (size_t r = 0; r < NR; ++r) {
477#else
478 for (size_t r = first; r < last; ++r) {
479#endif
480
481 auto offset = ptrs[r];
482 size_t num_entries = ptrs[r+1] - offset;
483 auto value_ptr = values.data() + offset;
484 auto index_ptr = indices.data() + offset;
485
486 // Computing the grand mean across all batchs.
487 std::fill(batch_means.begin(), batch_means.end(), 0);
488 std::fill(batch_count.begin(), batch_count.end(), 0);
489 for (size_t i = 0; i < num_entries; ++i) {
490 auto b = batch[index_ptr[i]];
491 batch_means[b] += value_ptr[i];
492 ++batch_count[b];
493 }
494
495 double& grand_mean = center_v[r];
496 grand_mean = 0;
497 for (size_t b = 0; b < nbatchs; ++b) {
498 auto bsize = batch_size[b];
499 if (bsize) {
500 grand_mean += batch_means[b] / bsize;
501 }
502 }
503 grand_mean /= nbatchs;
504
505 // Computing pseudo-variances where each batch's contribution
506 // is weighted inversely proportional to its size. This aims to
507 // match up with the variances used in the PCA but not the
508 // variances of the output components (where weightings are not used).
509 double& proxyvar = scale_v[r];
510 proxyvar = 0;
511 for (size_t b = 0; b < nbatchs; ++b) {
512 double zero_sum = (batch_size[b] - batch_count[b]) * grand_mean * grand_mean;
513 proxyvar += zero_sum / batch_size[b];
514 }
515
516 for (size_t i = 0; i < num_entries; ++i) {
517 double diff = value_ptr[i] - grand_mean;
518 auto bsize = batch_size[batch[index_ptr[i]]];
519 if (bsize) {
520 proxyvar += diff * diff / bsize;
521 }
522 }
523
524#ifndef SCRAN_CUSTOM_PARALLEL
525 }
526 }
527#else
528 }
529 }, nthreads);
530#endif
531
532 total_var = pca_utils::process_scale_vector(scale, scale_v);
533 }
534
535 // Actually filling the sparse matrix. Note that this is transposed
536 // because we want genes in the columns.
537 return pca_utils::SparseMatrix(NC, NR, std::move(values), std::move(indices), std::move(ptrs), nthreads);
538 }

541 template<typename T, typename IDX, typename Batch>
542 Eigen::MatrixXd create_eigen_matrix_dense(
543 const tatami::Matrix<T, IDX>* mat,
544 Eigen::VectorXd& center_v,
545 Eigen::VectorXd& scale_v,
546 const Batch* batch,
547 const std::vector<int>& batch_size,
548 double& total_var)
549 const {
550
551 // Extract a column-major matrix with genes in columns.
552 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads);
553
554 // Looping across genes (i.e., columns) to fill center_v and scale_v.
555 size_t NC = emat.cols(), NR = emat.rows();
556 size_t nbatchs = batch_size.size();
557
558#ifndef SCRAN_CUSTOM_PARALLEL
559 #pragma omp parallel num_threads(nthreads)
560 {
561 std::vector<double> mean_buffer(nbatchs);
562 #pragma omp for
563 for (size_t c = 0; c < NC; ++c) {
564#else
565 SCRAN_CUSTOM_PARALLEL(NC, [&](size_t first, size_t last) -> void {
566 std::vector<double> mean_buffer(nbatchs);
567 for (size_t c = first; c < last; ++c) {
568#endif
569
570 auto ptr = emat.data() + c * NR;
571 std::fill(mean_buffer.begin(), mean_buffer.end(), 0.0);
572 for (size_t r = 0; r < NR; ++r) {
573 mean_buffer[batch[r]] += ptr[r];
574 }
575
576 double& grand_mean = center_v[c];
577 grand_mean = 0;
578 int non_empty_batches = 0;
579 for (size_t b = 0; b < nbatchs; ++b) {
580 const auto& bsize = batch_size[b];
581 if (bsize) {
582 grand_mean += mean_buffer[b] / bsize;
583 ++non_empty_batches;
584 }
585 }
586 grand_mean /= non_empty_batches;
587
588 // We don't actually compute the batchwise variance, but instead
589 // the weighted sum of squared deltas, which is what PCA actually sees.
590 double& proxyvar = scale_v[c];
591 proxyvar = 0;
592 for (size_t r = 0; r < NR; ++r) {
593 const auto& bsize = batch_size[batch[r]];
594 if (bsize) {
595 double diff = ptr[r] - grand_mean;
596 proxyvar += diff * diff / bsize;
597 }
598 }
599
600#ifndef SCRAN_CUSTOM_PARALLEL
601 }
602 }
603#else
604 }
605 }, nthreads);
606#endif
607
608 total_var = pca_utils::process_scale_vector(scale, scale_v);
609 return emat;
610 }
