277 void run_blocked(
278 const Matrix* p,
279 const G* group,
280 const B* block,
281 std::vector<std::vector<Stat*> > means,
282 std::vector<std::vector<Stat*> > detected,
283 Stat* cohen,
284 Stat* auc,
285 Stat* lfc,
286 Stat* delta_detected)
287 const {
288 if (block == NULL) {
289 run(p, group, fetch_first(means), fetch_first(detected), cohen, auc, lfc, delta_detected);
290 return;
291 }
292
293 int ngroups = means.size();
294 int nblocks = (ngroups ? means[0].size() : 0); // no blocks = no groups.
295
296 int ncombos = ngroups * nblocks;
297 std::vector<Stat*> means2(ncombos), detected2(ncombos);
298 {
299 auto mIt = means2.begin(), dIt = detected2.begin();
300 for (int g = 0; g < ngroups; ++g) {
301 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
302 *mIt = means[g][b];
303 *dIt = detected[g][b];
304 }
305 }
306 }
307
308 differential_analysis::EffectsCalculator runner(nthreads, threshold);
309 Overlord overlord(auc);
310 auto state = runner.run_blocked(p, group, ngroups, block, nblocks, overlord);
311 process_simple_effects(state, means2, detected2, cohen, lfc, delta_detected);
312 }

370 size_t in_offset = nlevels * gene;
