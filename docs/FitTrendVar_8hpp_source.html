115  size_t counter = 0;
116  for (size_t i = 0; i < n; ++i) {
117  if (!filter || mean[i] >= min_mean) {
118  xbuffer[counter] = mean[i];
119  if (transform) {
120  ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
121  } else {

125  }
126  }

140 
141  // Identifying the left-most fitted value.

143 
144  counter = 0;
145  for (size_t i = 0; i < n; ++i) {

148  ++counter;
