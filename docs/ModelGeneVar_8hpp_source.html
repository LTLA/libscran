111  std::vector<double*> fitted,
112  std::vector<double*> residuals)
113  {
114  size_t NR = p->nrow(), NC = p->ncol();
115  std::vector<int> block_size(means.size());
116  if (block) {
117  auto copy = block;
118  for (size_t j = 0; j < NC; ++j, ++copy) {
119  ++block_size[*copy];
120  }
121  feature_selection::BlockedVarianceFactory<true, decltype(means), B, decltype(block_size)> fact(NR, NC, means, variances, block, block_size);
122  tatami::apply<0>(p, fact);
123  } else {
124  block_size[0] = NC;
125  feature_selection::BlockedVarianceFactory<false, decltype(means), B, decltype(block_size)> fact(NR, NC, means, variances, block, block_size);
126  tatami::apply<0>(p, fact);
127  }
128 
129  // Applying the trend fit to each block.
130  for (size_t b = 0; b < block_size.size(); ++b) {
131  if (block_size[b] >= 2) {
132  fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
133  } else {
134  std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
135  std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
136  }
137  }
