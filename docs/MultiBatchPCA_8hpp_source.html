29struct MultiBatchEigenMatrix {
30 MultiBatchEigenMatrix(const Matrix* m, const Eigen::VectorXd* w, const Eigen::VectorXd* mx) : mat(m), weights(w), means(mx) {}
31
32 auto rows() const { return mat->rows(); }
33 auto cols() const { return mat->cols(); }
34
35 template<class Right>
36 void multiply(const Right& rhs, Eigen::VectorXd& output) const {
37 output.noalias() = *mat * rhs;
38 double sub = means->dot(rhs);
39 for (Eigen::Index i = 0; i < output.size(); ++i) {
40 output[i] -= sub;
41 output[i] *= (*weights)[i];
42 }
43 }
44
45 template<class Right>
46 void adjoint_multiply(const Right& rhs, Eigen::VectorXd& output) const {
47 auto combined = weights->cwiseProduct(rhs);
48 output.noalias() = mat->adjoint() * combined;
49 double sum = combined.sum();
50 for (Eigen::Index i = 0; i < output.size(); ++i) {
51 output[i] -= (*means)[i] * sum;
52 }
53 return;
54 }
55
56 Eigen::MatrixXd realize() const {
57 Eigen::MatrixXd output(*mat);
58 for (Eigen::Index c = 0; c < output.cols(); ++c) {
59 for (Eigen::Index r = 0; r < output.rows(); ++r) {
60 auto& val = output(r, c);
61 val -= (*means)[c];
62 val *= (*weights)[r];
63 }
64 }
65 return output;
66 }
67private:
68 const Matrix* mat;
69 const Eigen::VectorXd* weights;
70 const Eigen::VectorXd* means;

159 void reapply(const Matrix& emat,
160 const Eigen::VectorXd& center_v,
161 const Eigen::VectorXd& scale_v,
162 Eigen::MatrixXd& pcs,
163 Eigen::MatrixXd& rotation,
164 Eigen::VectorXd& variance_explained)
165 {
166 // Dividing by the scaling factor to mimic the division of 'emat'
167 // by the scaling factor (after centering).
168 auto rIt = rotation.data();
169 for (size_t i = 0, iend = rotation.cols(); i < iend; ++i) {
170 auto sIt = scale_v.data();
171 for (size_t j = 0, jend = rotation.rows(); j < jend; ++j, ++rIt, ++sIt) {
172 (*rIt) /= *sIt;
173 }

179 template<class Matrix>
180 void reapply(const Matrix& emat,
181 const Eigen::VectorXd& center_v,
182 Eigen::MatrixXd& pcs,
183 const Eigen::MatrixXd& rotation,
184 Eigen::VectorXd& variance_explained)
185 {
186 pcs = emat * rotation;
187
188 // Effective centering because I don't want to modify 'emat'.
189 auto pIt = pcs.data();
190 for (size_t i = 0, iend = pcs.cols(); i < iend; ++i) {
191 double meanval = center_v.dot(rotation.col(i));
192 for (size_t j = 0, jend = pcs.rows(); j < jend; ++j, ++pIt) {
193 *pIt -= meanval;
194 }
195 }
196
197 // Variance is a somewhat murky concept with weights, so we just square
198 // it and assume that only the relative value matters.
199 for (auto& d : variance_explained) {
200 d = d * d;
201 }
202 return;
203 }

206 template<typename T, typename IDX, typename Batch>
207 void run(const tatami::Matrix<T, IDX>* mat, const Batch* batch, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) {
208 const size_t NC = mat->ncol();
209 const auto& batch_size = block_sizes(NC, batch);
210 const size_t nbatchs = batch_size.size();
211
212 // Computing weights.
213 Eigen::VectorXd weights(NC);
214 for (size_t i = 0; i < NC; ++i) {
215 weights[i] = 1/std::sqrt(static_cast<double>(batch_size[batch[i]]));
216 }
217
218 Eigen::VectorXd center_v(mat->nrow());
219 Eigen::VectorXd scale_v(mat->nrow());
220
221 // Remember, we want to run the PCA on the modified matrix,
222 // but we want to apply the rotation vectors to the original matrix
223 // (after any centering/scaling but without the batch weights).
224 auto executor = [&](const auto& emat) -> void {
225 MultiBatchEigenMatrix<typename std::remove_reference<decltype(emat)>::type> thing(&emat, &weights, &center_v);
226 if (scale) {
227 irb.run(irlba::Scaled<decltype(thing)>(&thing, &scale_v), pcs, rotation, variance_explained);
228 reapply(emat, center_v, scale_v, pcs, rotation, variance_explained);
229 } else {
230 irb.run(thing, pcs, rotation, variance_explained);
231 reapply(emat, center_v, pcs, rotation, variance_explained);
232 }
233 };
234
235 if (mat->sparse()) {
236 auto emat = create_eigen_matrix_sparse(mat, center_v, scale_v, batch, batch_size, total_var);
237 executor(emat);
238 } else {
239 auto emat = create_eigen_matrix_dense(mat, center_v, scale_v, batch, batch_size, total_var);
240 executor(emat);
241 }
242
243 if (transpose) {
244 pcs.adjointInPlace();
245 }
246
247 return;
248 }

344 template<typename T, typename IDX, typename Batch>
345 Eigen::SparseMatrix<double> create_eigen_matrix_sparse(const tatami::Matrix<T, IDX>* mat,
346 Eigen::VectorXd& mean_v,
347 Eigen::VectorXd& scale_v,
348 const Batch* batch,
349 const std::vector<int>& batch_size,
350 double& total_var)
351 {
352 size_t NR = mat->nrow(), NC = mat->ncol();
353 size_t nbatchs = batch_size.size();
354 total_var = 0;
355
356 Eigen::SparseMatrix<double> A(NC, NR); // transposed; we want genes in the columns.
357 std::vector<std::vector<double> > values;
358 std::vector<std::vector<int> > indices;
359
360 if (mat->prefer_rows()) {
361 std::vector<double> xbuffer(NC);
362 std::vector<int> ibuffer(NC);
363 std::vector<int> nnzeros(NR);
364 values.reserve(NR);
365 indices.reserve(NR);
366 std::vector<double> batch_means(nbatchs);
367 std::vector<int> batch_count(nbatchs);
368
369 for (size_t r = 0; r < NR; ++r) {
370 auto range = mat->sparse_row(r, xbuffer.data(), ibuffer.data());
371
372 // Computing the grand mean across all batchs.
373 std::fill(batch_means.begin(), batch_means.end(), 0);
374 std::fill(batch_count.begin(), batch_count.end(), 0);
375 for (size_t i = 0; i < range.number; ++i) {
376 auto b = batch[range.index[i]];
377 batch_means[b] += range.value[i];
378 ++batch_count[b];
379 }
380
381 double& grand_mean = mean_v[r];
382 grand_mean = 0;
383 for (size_t b = 0; b < nbatchs; ++b) {
384 grand_mean += batch_means[b] / batch_size[b];
385 }
386 grand_mean /= nbatchs;
387
388 // Computing pseudo-variances where each batch's contribution
389 // is weighted inversely proportional to its size. This aims to
390 // match up with the variances used in the PCA but not the
391 // variances of the output components (where weightings are not used).
392 double& proxyvar = scale_v[r];
393 proxyvar = 0;
394 for (size_t b = 0; b < nbatchs; ++b) {
395 double zero_sum = (batch_size[b] - batch_count[b]) * grand_mean * grand_mean;
396 proxyvar += zero_sum / batch_size[b];
397 }
398
399 for (size_t i = 0; i < range.number; ++i) {
400 double diff = range.value[i] - grand_mean;
401 auto b = batch[range.index[i]];
402 proxyvar += diff * diff / batch_size[b];
403 }
404
405 nnzeros[r] = range.number;
406 values.emplace_back(range.value, range.value + range.number);
407 indices.emplace_back(range.index, range.index + range.number);
408 }
409
410 pca_utils::set_scale(scale, scale_v, total_var);
411 pca_utils::fill_sparse_matrix<true>(A, indices, values, nnzeros);
412 } else {
413 std::vector<double> xbuffer(NR);
414 std::vector<int> ibuffer(NR);
415 values.reserve(NC);
416 indices.reserve(NC);
417 std::vector<int> nnzeros(NR);
418
419 std::vector<std::vector<double> > tmp_means(nbatchs, std::vector<double>(NR));
420 std::vector<std::vector<int> > tmp_nonzero(nbatchs, std::vector<int>(NR));
421
422 for (size_t c = 0; c < NC; ++c) {
423 auto range = mat->sparse_column(c, xbuffer.data(), ibuffer.data());
424 values.emplace_back(range.value, range.value + range.number);
425 indices.emplace_back(range.index, range.index + range.number);
426
427 // Collecting values for the means.
428 Batch curb = batch[c];
429 auto& cur_means = tmp_means[curb];
430 auto& cur_nonzero= tmp_nonzero[curb];
431 for (size_t i = 0; i < range.number; ++i) {
432 auto r = range.index[i];
433 cur_means[r] += range.value[i];
434 ++cur_nonzero[r];
435 ++nnzeros[r];
436 }
437 }
438
439 // Finalizing the means.
440 mean_v.setZero();
441 for (size_t b = 0; b < nbatchs; ++b) {
442 const auto& cur_means = tmp_means[b];
443 for (size_t r = 0; r < NR; ++r) {
444 mean_v[r] += cur_means[r] / batch_size[b];
445 }
446 }
447 for (size_t r = 0; r < NR; ++r) {
448 mean_v[r] /= nbatchs;
449 }
450
451 // Computing the pseudo-variances for each gene.
452 scale_v.setZero();
453 for (size_t c = 0; c < NC; ++c) {
454 const auto& cur_vals = values[c];
455 const auto& cur_idx = indices[c];
456 auto bs = batch_size[batch[c]];
457
458 for (size_t i = 0; i < cur_idx.size(); ++i) {
459 auto r = cur_idx[i];
460 double diff = cur_vals[i] - mean_v[r];
461 scale_v[r] += diff * diff / bs;
462 }
463 }
464
465 for (size_t b = 0; b < nbatchs; ++b) {
466 const auto& used = tmp_nonzero[b];
467 for (size_t r = 0; r < NR; ++r) {
468 double zero_sum = mean_v[r] * mean_v[r] * (batch_size[b] - used[r]);
469 scale_v[r] += zero_sum / batch_size[b];
470 }
471 }
472
473 pca_utils::set_scale(scale, scale_v, total_var);
474 pca_utils::fill_sparse_matrix<false>(A, indices, values, nnzeros);
475 }
476
477 A.makeCompressed();
478 return A;
479 }

482 template<typename T, typename IDX, typename Batch>
483 Eigen::MatrixXd create_eigen_matrix_dense(
484 const tatami::Matrix<T, IDX>* mat,
485 Eigen::VectorXd& mean_v,
486 Eigen::VectorXd& scale_v,
487 const Batch* batch,
488 const std::vector<int>& batch_size,
489 double& total_var)
490 {
491 size_t NR = mat->nrow(), NC = mat->ncol();
492 total_var = 0;
493
494 Eigen::MatrixXd output(NC, NR); // transposed.
495 std::vector<double> xbuffer(NC);
496 double* outIt = output.data();
497
498 size_t nbatchs = batch_size.size();
499 std::vector<double> mean_buffer(nbatchs);
500
501 for (size_t r = 0; r < NR; ++r, outIt += NC) {
502 auto ptr = mat->row_copy(r, outIt);
503
504 std::fill(mean_buffer.begin(), mean_buffer.end(), 0);
505 for (size_t c = 0; c < NC; ++c) {
506 mean_buffer[batch[c]] += ptr[c];
507 }
508 double& grand_mean = mean_v[r];
509 grand_mean = 0;
510 for (size_t b = 0; b < nbatchs; ++b) {
511 grand_mean += mean_buffer[b] / batch_size[b];
512 }
513 grand_mean /= nbatchs;
514
515 // We don't actually compute the batchwise variance, but instead
516 // the weighted sum of squared deltas, which is what PCA actually sees.
517 double& proxyvar = scale_v[r];
518 proxyvar = 0;
519 for (size_t c = 0; c < NC; ++c) {
520 double diff = outIt[c] - grand_mean;
521 proxyvar += diff * diff / batch_size[batch[c]];
522 }
523 }
524
525 pca_utils::set_scale(scale, scale_v, total_var);
526 return output;
527 }
