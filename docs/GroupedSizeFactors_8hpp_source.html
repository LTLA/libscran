128 template<typename T, typename IDX, typename Group, typename Ref, typename Out>
129 void run_internal(const tatami::Matrix<T, IDX>* mat, const Group* group, Ref reference, Out* output) {
130 size_t NR = mat->nrow(), NC = mat->ncol();
131 if (!NC) {
132 return;
133 }
134
135 // Aggregating each group to get a pseudo-bulk sample.
136 auto ngroups = *std::max_element(group, group + NC) + 1;
137 std::vector<double> combined(ngroups * NR);
138 {
139 std::vector<double*> sums(ngroups);
140 for (size_t i = 0; i < ngroups; ++i) {
141 sums[i] = combined.data() + i * NR;
142 }
143 AggregateAcrossCells aggregator;
144 aggregator.run(mat, group, std::move(sums), std::vector<int*>());
145 }
146
147 size_t ref = 0;
148 if constexpr(std::is_same<Ref, size_t>::value) {
149 ref = reference;
150 } else {
151 // Choosing one of them to be the reference. Here, we borrow some logic
152 // from edgeR and use the one with the largest sum of square roots,
153 // which provides a compromise between transcriptome coverage and
154 // complexity. The root ensures that we don't pick a sample that just
155 // has very high expression in a small subset of genes, while still
156 // remaining responsive to the overall coverage level.
157 double best = 0;
158
159 for (size_t i = 0; i < ngroups; ++i) {
160 auto start = combined.data() + i * NR;
161 double current = 0;
162 for (size_t j = 0; j < NR; ++j) {
163 current += std::sqrt(start[j]);
164 }
165 if (current > best) {
166 ref = i;
167 best = current;
168 }
169 }
170 }
171
172 // Computing median-based size factors. No need to center
173 // here as we'll be recentering afterwards anyway.
174 tatami::ArrayView view(combined.data(), combined.size());
175 tatami::DenseColumnMatrix<T, IDX, decltype(view)> aggmat(NR, ngroups, std::move(view));
176 MedianSizeFactors med;
177 med.set_center(false).set_prior_count(prior_count);
178 auto mres = med.run(&aggmat, combined.data() + ref * NR);
179
180 // Propagating to each cell via library size-based normalization.
181 auto aggcolsums = tatami::column_sums(&aggmat);
182 auto colsums = tatami::column_sums(mat);
183 for (size_t i = 0; i < NC; ++i) {
184 auto curgroup = group[i];
185 auto scale = static_cast<double>(colsums[i])/static_cast<double>(aggcolsums[curgroup]);
186 output[i] = scale * mres.factors[curgroup];
187 }
188
189 // Throwing in some centering.
190 if (center) {
191 CenterSizeFactors centerer;
192 centerer.run(NC, output);
193 }
194 }
