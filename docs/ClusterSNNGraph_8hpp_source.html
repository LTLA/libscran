330 Graph build(size_t ncells, const std::deque<BuildSNNGraph::WeightedEdge>& store) const {
331 IgraphVector edge_holder(store.size() * 2);
332 auto& edges = edge_holder.vector;
333
334 IgraphVector weight_holder(store.size());
335 auto& weights = weight_holder.vector;
336
337 size_t counter = 0;
338 for (size_t i = 0, end = store.size(); i < end; ++i, counter += 2) {
339 const auto& edge = store[i];
340 VECTOR(edges)[counter] = std::get<0>(edge);
341 VECTOR(edges)[counter + 1] = std::get<1>(edge);
342 VECTOR(weights)[i] = std::get<2>(edge);
343 }
344
345 IgraphGraph graph(edge_holder, ncells, /* directed = */ false);
346 return Graph(std::move(graph), std::move(weight_holder));

951 Results run(const Graph& graph_info) {
952 IgraphVector membership_holder;
953 auto& membership = membership_holder.vector;
954 igraph_integer_t nb_clusters;
955 igraph_real_t quality;
956
957 IgraphRNGScope rngs(seed);
958 Results output;
959
960 if (!modularity) {
961 for (int i = 0; i < iterations; ++i) {
962 output.status = igraph_community_leiden(graph_info.get_graph(), graph_info.get_weights(), NULL, resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
963 if (output.status) {
964 break;
965 }
966 }
967 } else {
968 // Based on https://igraph.org/c/doc/igraph-Community.html#igraph_community_leiden.
969 IgraphVector degree_holder(igraph_vcount(graph_info.get_graph()));
970 auto& degree = degree_holder.vector;
971 igraph_degree(graph_info.get_graph(), &degree, igraph_vss_all(), IGRAPH_ALL, 1);
972
973 // This assumes that resolution = 1 in the example in the C documentation.
974 // igraph::cluster_leiden in the R package does the same thing.
975 double mod_resolution = resolution / (2 * igraph_ecount(graph_info.get_graph()));
976
977 for (int i = 0; i < iterations; ++i) {
978 output.status = igraph_community_leiden(graph_info.get_graph(), graph_info.get_weights(), &degree, mod_resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
979 if (output.status) {
980 break;
981 }
982 }
983 }
984
985 if (!output.status) {
986 size_t ncells = igraph_vcount(graph_info.get_graph());
987 output.membership.resize(ncells);
988 for (size_t i = 0; i < ncells; ++i) {
989 output.membership[i] = VECTOR(membership)[i];
990 }
991 output.quality = quality;
992 }
993
994 return output;
995 }
