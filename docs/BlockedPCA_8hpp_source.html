29template<class Matrix, typename Block>
30struct BlockedEigenMatrix {
31 BlockedEigenMatrix(const Matrix* m, const Block* b, const Eigen::MatrixXd* mx) : mat(m), block(b), means(mx) {}
32
33 auto rows() const { return mat->rows(); }
34 auto cols() const { return mat->cols(); }
35
36 template<class Right>
37 void multiply(const Right& rhs, Eigen::VectorXd& output) const {
38 if constexpr(irlba::has_multiply_method<Matrix>::value) {
39 output.noalias() = (*mat) * rhs;
40 } else {
41 mat->multiply(rhs, output);
42 }
43
44 Eigen::VectorXd sub = (*means) * rhs;
45 for (Eigen::Index i = 0; i < output.size(); ++i) {
46 output.coeffRef(i) -= sub.coeff(block[i]);
47 }
48 return;
49 }
50
51 template<class Right>
52 void adjoint_multiply(const Right& rhs, Eigen::VectorXd& output) const {
53 if constexpr(irlba::has_adjoint_multiply_method<Matrix>::value) {
54 output.noalias() = mat->adjoint() * rhs;
55 } else {
56 mat->adjoint_multiply(rhs, output);
57 }
58
59 Eigen::VectorXd aggr(means->rows());
60 aggr.setZero();
61 for (Eigen::Index i = 0; i < rhs.size(); ++i) {
62 aggr.coeffRef(block[i]) += rhs.coeff(i);
63 }
64
65 output.noalias() -= means->adjoint() * aggr;
66 return;
67 }
68
69 Eigen::MatrixXd realize() const {
70 Eigen::MatrixXd output;
71 if constexpr(irlba::has_realize_method<Matrix>::value) {
72 output = mat->realize();
73 } else {
74 output = Eigen::MatrixXd(*mat);
75 }
76
77 for (Eigen::Index c = 0; c < output.cols(); ++c) {
78 for (Eigen::Index r = 0; r < output.rows(); ++r) {
79 output.coeffRef(r, c) -= means->coeff(block[r], c);
80 }
81 }
82 return output;
83 }
84private:
85 const Matrix* mat;
86 const Block* block;
87 const Eigen::MatrixXd* means;

188 template<typename T, typename IDX, typename Block>
189 void run(const tatami::Matrix<T, IDX>* mat, const Block* block, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
190 const size_t NC = mat->ncol();
191 const int nblocks = (NC ? *std::max_element(block, block + NC) + 1 : 1);
192 std::vector<int> block_size(nblocks);
193 for (size_t j = 0; j < NC; ++j) {
194 ++block_size[block[j]];
195 }
196
197 irlba::EigenThreadScope t(nthreads);
198 irlba::Irlba irb;
199 irb.set_number(rank);
200
201 if (mat->sparse()) {
202 Eigen::MatrixXd center_m(nblocks, mat->nrow());
203 Eigen::VectorXd scale_v(mat->nrow());
204 auto emat = create_custom_sparse_matrix(mat, center_m, scale_v, block, block_size, total_var);
205
206 BlockedEigenMatrix<decltype(emat), Block> centered(&emat, block, &center_m);
207 if (scale) {
208 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
209 irb.run(scaled, pcs, rotation, variance_explained);
210 } else {
211 irb.run(centered, pcs, rotation, variance_explained);
212 }
213 } else {
214 auto emat = create_eigen_matrix_dense(mat, block, block_size, total_var);
215 irb.run(emat, pcs, rotation, variance_explained); // already centered and scaled, if relevant.
216 }
217
218 pca_utils::clean_up(mat->ncol(), pcs, variance_explained);
219 if (transpose) {
220 pcs.adjointInPlace();
221 }
222
223 return;
224 }

317 template<typename T, typename IDX, typename Block>
318 pca_utils::CustomSparseMatrix create_custom_sparse_matrix(const tatami::Matrix<T, IDX>* mat,
319 Eigen::MatrixXd& center_m,
320 Eigen::VectorXd& scale_v,
321 const Block* block,
322 const std::vector<int>& block_size,
323 double& total_var)
324 const {
325 size_t NR = mat->nrow(), NC = mat->ncol();
326 pca_utils::CustomSparseMatrix A(NC, NR, nthreads); // transposed; we want genes in the columns.
327
328#ifdef TEST_SCRAN_CUSTOM_SPARSE_MATRIX
329 if (use_eigen) {
330 A.use_eigen();
331 }
332#endif
333
334 std::vector<double> values;
335 std::vector<int> indices;
336 std::vector<size_t> ptrs = pca_utils::fill_transposed_compressed_sparse_vectors(mat, values, indices, nthreads);
337
338 // Computing block-specific means and variances.
339 {
340 const size_t nblocks = block_size.size();
341
342#ifndef SCRAN_CUSTOM_PARALLEL
343 #pragma omp parallel for num_threads(nthreads)
344 for (size_t r = 0; r < NR; ++r) {
345#else
346 SCRAN_CUSTOM_PARALLEL(NR, [&](size_t first, size_t last) -> void {
347 for (size_t r = first; r < last; ++r) {
348#endif
349
350 auto offset = ptrs[r];
351 size_t num_entries = ptrs[r+1] - offset;
352 auto value_ptr = values.data() + offset;
353 auto index_ptr = indices.data() + offset;
354
355 // Computing the block-wise means.
356 auto mbuffer = center_m.data() + nblocks * r;
357 std::fill(mbuffer, mbuffer + nblocks, 0);
358 for (size_t i = 0; i < num_entries; ++i) {
359 mbuffer[block[index_ptr[i]]] += value_ptr[i];
360 }
361 for (size_t b = 0; b < nblocks; ++b) {
362 mbuffer[b] /= block_size[b];
363 }
364
365 // Computing the variance from the sum of squared differences.
366 // This is technically not the correct variance estimate if we
367 // were to consider the loss of residual d.f. from estimating
368 // the block means, but it's what the PCA sees, so whatever.
369 double& proxyvar = scale_v[r];
370 proxyvar = 0;
371 {
372 auto block_copy = block_size;
373 for (size_t i = 0; i < num_entries; ++i) {
374 Block curb = block[index_ptr[i]];
375 double diff = value_ptr[i] - mbuffer[curb];
376 proxyvar += diff * diff;
377 --block_copy[curb];
378 }
379
380 for (size_t b = 0; b < nblocks; ++b) {
381 proxyvar += mbuffer[b] * mbuffer[b] * block_copy[b];
382 }
383
384 proxyvar /= NC - 1;
385 }
386
387#ifndef SCRAN_CUSTOM_PARALLEL
388 }
389#else
390 }
391 }, nthreads);
392#endif
393
394 pca_utils::set_scale(scale, scale_v, total_var);
395 }
396
397 A.fill_direct(std::move(values), std::move(indices), std::move(ptrs));
398 return A;
399 }

402 template<typename T, typename IDX, typename Block>
403 Eigen::MatrixXd create_eigen_matrix_dense(
404 const tatami::Matrix<T, IDX>* mat,
405 const Block* block,
406 const std::vector<int>& block_size,
407 double& total_var)
408 const {
409 size_t NR = mat->nrow(), NC = mat->ncol();
410 total_var = 0;
411
412 Eigen::MatrixXd output(NC, NR); // transposed.
413 std::vector<double> xbuffer(NC);
414 double* outIt = output.data();
415
416 int nblocks = block_size.size();
417 std::vector<double> mean_buffer(nblocks);
418
419#ifdef SCRAN_LOGGER
420 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
421#endif
422
423 for (size_t r = 0; r < NR; ++r, outIt += NC) {
424 auto ptr = mat->row_copy(r, outIt);
425
426 // Compute the means of each block.
427 std::fill(mean_buffer.begin(), mean_buffer.end(), 0);
428 {
429 auto copy = outIt;
430 auto bcopy = block;
431 for (size_t c = 0; c < NC; ++c, ++copy, ++bcopy) {
432 mean_buffer[*bcopy] += *copy;
433 }
434 }
435 for (int b = 0; b < nblocks; ++b) {
436 if (block_size[b]) {
437 mean_buffer[b] /= block_size[b];
438 }
439 }
440
441 // We don't actually compute the blockwise variance, but instead
442 // the squared sum of deltas, which is what PCA actually sees.
443 double proxyvar = 0;
444 {
445 auto copy = outIt;
446 auto bcopy = block;
447 for (size_t c = 0; c < NC; ++c, ++copy, ++bcopy) {
448 *copy -= mean_buffer[*bcopy];
449 proxyvar += *copy * *copy;
450 }
451 proxyvar /= NC - 1;
452 }
453
454 pca_utils::apply_scale(scale, proxyvar, NC, outIt, total_var);
455 }
456
457 return output;
458 }
