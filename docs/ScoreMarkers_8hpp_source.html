558 template<class MAT, typename G, typename Stat>
559 void run_internal(const MAT* p, const G* group, int ngroups,
560 std::vector<Stat*>& means,
561 std::vector<Stat*>& detected,
562 std::vector<std::vector<Stat*> >& cohen,
563 std::vector<std::vector<Stat*> >& auc,
564 std::vector<std::vector<Stat*> >& lfc,
565 std::vector<std::vector<Stat*> >& delta_detected)
566 {
567 std::vector<int> group_size(ngroups);
568 for (size_t i = 0; i < p->ncol(); ++i) {
569 ++(group_size[group[i]]);

574 template<class MAT, typename G, typename B, typename Stat>
575 void run_blocked_internal(const MAT* p, const G* group, const B* block, int ngroups, int nblocks,
576 std::vector<std::vector<Stat*> >& means,
577 std::vector<std::vector<Stat*> >& detected,
578 std::vector<std::vector<Stat*> >& cohen,
579 std::vector<std::vector<Stat*> >& auc,
580 std::vector<std::vector<Stat*> >& lfc,
581 std::vector<std::vector<Stat*> >& delta_detected)
582 {
583 int ncombos = ngroups * nblocks;
584 std::vector<int> combos(p->ncol());
585 std::vector<int> combo_size(ncombos);
586
587 for (size_t i = 0; i < combos.size(); ++i) {
588 combos[i] = group[i] * nblocks + block[i];
589 ++(combo_size[combos[i]]);
590 }
591
592 std::vector<Stat*> means2(ncombos), detected2(ncombos);
593 auto mIt = means2.begin(), dIt = detected2.begin();
594 for (int g = 0; g < ngroups; ++g) {
595 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
596 *mIt = means[g][b];
597 *dIt = detected[g][b];
598 }
599 }
600
601 core(p, combos.data(), combo_size, group, ngroups, block, nblocks, means2, detected2, cohen, auc, lfc, delta_detected);
602 }

604 template<class MAT, typename L, class Ls, typename G, typename B, typename Stat>
605 void core(const MAT* p,
606 const L* level, const Ls& level_size,
607 const G* group, int ngroups,
608 const B* block, int nblocks,
609 std::vector<Stat*>& means,
610 std::vector<Stat*>& detected,
611 std::vector<std::vector<Stat*> >& cohen,
612 std::vector<std::vector<Stat*> >& auc,
613 std::vector<std::vector<Stat*> >& lfc,
614 std::vector<std::vector<Stat*> >& delta_detected)
615 {
616 size_t buffer_size = p->nrow() * ngroups * ngroups;
617
618 const bool do_cohen = !cohen.empty();
619 std::vector<Stat> cohen_buffer(do_cohen ? buffer_size : 0);
620 Stat* cohen_ptr = do_cohen ? cohen_buffer.data() : NULL;
621
622 const bool do_auc = !auc.empty();
623 std::vector<Stat> auc_buffer(do_auc ? buffer_size : 0);
624 Stat* auc_ptr = do_auc ? auc_buffer.data() : NULL;
625
626 const bool do_lfc = !lfc.empty();
627 std::vector<Stat> lfc_buffer(do_lfc ? buffer_size : 0);
628 Stat* lfc_ptr = do_lfc ? lfc_buffer.data() : NULL;
629
630 const bool do_delta = !delta_detected.empty();
631 std::vector<Stat> delta_buffer(do_delta ? buffer_size : 0);
632 Stat* delta_ptr = do_delta ? delta_buffer.data() : NULL;
633
634 std::vector<Stat*> effects { cohen_ptr, auc_ptr, lfc_ptr, delta_ptr };
635
636#ifdef SCRAN_LOGGER
637 SCRAN_LOGGER("scran::ScoreMarkers", "Performing pairwise comparisons between groups of cells");
638#endif
639 if (!do_auc) {
640 differential_analysis::BidimensionalFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, ngroups, nblocks, threshold);
641 tatami::apply<0>(p, fact, num_threads);
642
643 } else {
644 // Need to remake this, as there's no guarantee that 'blocks' exists.
645 std::vector<B> tmp_blocks;
646 if (!block) {
647 tmp_blocks.resize(p->ncol());
648 block = tmp_blocks.data();
649 }
650
651 differential_analysis::PerRowFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, group, ngroups, block, nblocks, threshold);
652 tatami::apply<0>(p, fact, num_threads);
653 }
654
655 auto summarize = [&](Stat* ptr, std::vector<std::vector<Stat*> >& output) -> void {
656 auto& min_rank = output[scran::differential_analysis::MIN_RANK];
657 if (min_rank.size()) {
658 differential_analysis::compute_min_rank(p->nrow(), ngroups, ptr, min_rank, num_threads);
659 }
660 differential_analysis::summarize_comparisons(p->nrow(), ngroups, ptr, output, num_threads);
661 };
662
663 if (do_cohen) {
664 summarize(cohen_ptr, cohen);
665 }
666 if (do_auc) {
667 summarize(auc_ptr, auc);
668 }
669 if (do_lfc) {
670 summarize(lfc_ptr, lfc);
671 }
672 if (do_delta) {
673 summarize(delta_ptr, delta_detected);
674 }
675
676 return;
677 }

689 template<typename Stat>
690 struct Results {
694 Results() {}
695
696 Results(
697 size_t ngenes,
698 int ngroups,
699 int nblocks,
700 const ComputeSummaries& do_cohen,
701 const ComputeSummaries& do_auc,
702 const ComputeSummaries& do_lfc,
703 const ComputeSummaries& do_delta_detected)
704 {
705 auto fill_inner = [&](int N, auto& type) {
706 type.reserve(N);
707 for (int n = 0; n < N; ++n) {
708 type.emplace_back(ngenes);
709 }
710 };
711
712 means.resize(ngroups);
713 detected.resize(ngroups);
714 for (int g = 0; g < ngroups; ++g) {
715 fill_inner(nblocks, means[g]);
716 fill_inner(nblocks, detected[g]);
717 }
718
719 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
720 bool has_any = false;
721 for (size_t i = 0; i < do_this.size(); ++i) {
722 if (do_this[i]) {
723 has_any = true;
724 break;
725 }
726 }
727
728 if (has_any) {
729 effect.resize(differential_analysis::n_summaries);
730 if (do_this[differential_analysis::MIN]) {
731 fill_inner(ngroups, effect[differential_analysis::MIN]);
732 }
733 if (do_this[differential_analysis::MEAN]) {
734 fill_inner(ngroups, effect[differential_analysis::MEAN]);
735 }
736 if (do_this[differential_analysis::MEDIAN]) {
737 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
738 }
739 if (do_this[differential_analysis::MAX]) {
740 fill_inner(ngroups, effect[differential_analysis::MAX]);
741 }
742 if (do_this[differential_analysis::MIN_RANK]) {
743 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
744 }
745 }
746 return;
747 };
748
749 fill_effect(do_cohen, cohen);
750 fill_effect(do_auc, auc);
751 fill_effect(do_lfc, lfc);
752 fill_effect(do_delta_detected, delta_detected);
753 return;
754 }
