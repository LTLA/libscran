229 template<class Function, typename D, typename PPTR, typename X>
230 Thresholds run_internal(const Function& fun, size_t ncells, const D* detected, std::vector<PPTR> subset_totals,
231 X* filter_by_detected, std::vector<X*> filter_by_subset_totals, X* overall_filter)
232 const {
233 Thresholds output;
234 std::vector<double> buffer(ncells);
235
236 // Filtering to remove outliers on the log-detected number.
237 {
238 IsOutlier outliers;
239 outliers.set_nmads(detected_nmads).set_lower(true).set_upper(false).set_log(true);
240 if (min_detected_drop > 0) {
241 if (min_detected_drop >= 1) {
242 throw std::runtime_error("minimum drop in the detected features should lie in [0, 1)");
243 }
244 outliers.set_min_diff(-std::log(1 - min_detected_drop));
245 }
246
247 auto res = fun(outliers, ncells, detected, filter_by_detected, buffer);
248 output.detected = res.lower;
249 for (size_t i = 0; i < ncells; ++i) {
250 overall_filter[i] |= filter_by_detected[i];
251 }
252 }
253
254 // Filtering to remove outliers on the subset totals.
255 {
256 size_t nsubsets = subset_totals.size();
257 if (filter_by_subset_totals.size() != nsubsets) {
258 throw std::runtime_error("mismatching number of input/outputs for subset filters");
259 }
260
261 IsOutlier outliers;
262 outliers.set_nmads(subset_nmads).set_upper(true).set_lower(false).set_log(true);
263 output.subset_totals.resize(nsubsets);
264
265 for (size_t s = 0; s < subset_totals.size(); ++s) {
266 auto dump = filter_by_subset_totals[s];
267 auto res = fun(outliers, ncells, subset_totals[s], dump, buffer);
268 output.subset_totals[s] = res.upper;
269
270 for (size_t i = 0; i < ncells; ++i) {
271 overall_filter[i] |= dump[i];
272 }
273 }
274 }
275
276 return output;
277 }
