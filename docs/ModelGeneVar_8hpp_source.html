100  void run_blocked(const MAT* p, const B* block, std::vector<Stat*> means, std::vector<Stat*> variances, std::vector<Stat*> fitted, std::vector<Stat*> residuals) {
101  size_t NR = p->nrow(), NC = p->ncol();
102  std::vector<int> block_size(means.size());
103  if (block) {
104  auto copy = block;
105  for (size_t j = 0; j < NC; ++j, ++copy) {
106  ++block_size[*copy];
107  }
108  feature_selection::BlockedVarianceFactory<true, decltype(means), B, decltype(block_size)> fact(NR, NC, means, variances, block, block_size);
109  tatami::apply<0>(p, fact);
110  } else {
111  block_size[0] = NC;
112  feature_selection::BlockedVarianceFactory<false, decltype(means), B, decltype(block_size)> fact(NR, NC, means, variances, block, block_size);
113  tatami::apply<0>(p, fact);
114  }
115 
116  // Applying the trend fit to each block.
117  for (size_t b = 0; b < block_size.size(); ++b) {
118  if (block_size[b] >= 2) {
119  fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
120  } else {
121  std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
122  std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
123  }
124  }

126  return;
