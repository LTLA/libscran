223 void filter_(size_t n, const PerCellRnaQcMetrics::Buffers<Float, Integer>& buffers, Output* output, Function indexer) const {
224 size_t nsubsets = subset_proportions.size();
225
226 for (size_t i = 0; i < n; ++i) {
227 auto b = indexer(i);
228 if (quality_control::is_less_than(buffers.sums[i], sums[b])) {
229 output[i] = true;
230 continue;
231 }
232
233 if (quality_control::is_less_than<double>(buffers.detected[i], detected[b])) {
234 output[i] = true;
235 continue;
236 }
237
238 bool fail = false;
239 for (size_t s = 0; s < nsubsets; ++s) {
240 if (quality_control::is_greater_than(buffers.subset_proportions[s][i], subset_proportions[s][b])) {
241 fail = true;
242 break;
243 }
244 }
245 if (fail) {
246 output[i] = true;

248 }

299 Thresholds output;
300 std::vector<double> workspace(n);
301 std::vector<int> starts;

304 }
305
306 // Filtering on the total counts.
307 {
308 ComputeMedianMad meddler;
309 meddler.set_log(true);
310 auto sums_res = meddler.run_blocked(n, block, starts, buffers.sums, workspace.data());
311
312 ChooseOutlierFilters filter;
313 filter.set_num_mads(sums_num_mads);
314 filter.set_upper(false);
315 auto sums_filt = filter.run(std::move(sums_res));
316
317 output.sums = std::move(sums_filt.lower);
318 }
319
320 // Filtering on the detected features.
321 {
322 ComputeMedianMad meddler;
323 meddler.set_log(true);
324 auto detected_res = meddler.run_blocked(n, block, starts, buffers.detected, workspace.data());
325
326 ChooseOutlierFilters filter;
327 filter.set_num_mads(sums_num_mads);
328 filter.set_upper(false);
329 auto detected_filt = filter.run(std::move(detected_res));
330
331 output.detected = std::move(detected_filt.lower);
332 }
333
334 // Filtering on the subset proportions (no log).
335 {
336 ComputeMedianMad meddler;
337
338 ChooseOutlierFilters filter;
339 filter.set_num_mads(subset_num_mads);
340 filter.set_lower(false);
341

343 for (size_t s = 0; s < nsubsets; ++s) {
