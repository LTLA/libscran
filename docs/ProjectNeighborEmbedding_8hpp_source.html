117 template<typename Index, typename Float>
118 void compute_weighted_average(const std::vector<std::pair<Index, Float> >& neighbors, Float* buffer, int ndim, const Float* embedding, Float* output) const {
119 size_t nneighbors = neighbors.size();
120
121 Float bandwidth = 0;
122 if (nneighbors) {
123 for (size_t n = 0; n < nneighbors; ++n) {
124 buffer[n] = neighbors[n].second;
125 }
126 Float median = tatami::stats::compute_median<Float>(buffer, nneighbors);
127
128 for (size_t n = 0; n < nneighbors; ++n) {
129 buffer[n] = std::abs(neighbors[n].second - median);
130 }
131
132 // Scaling equivalence with a normal distribution's SD. Not sure
133 // how much difference it makes, but we do it for IsOutlier so we
134 // might as well do it here.
135 Float mad = tatami::stats::compute_median<Float>(buffer, nneighbors) * 1.4826;
136
137 bandwidth = median + mad * nmads;
138 }
139
140 Float total = 0;
141 if (bandwidth) {
142 for (size_t i = 0; i < neighbors.size(); ++i) {
143 Float diff = std::min(1.0, neighbors[i].second / bandwidth);
144 Float sub = 1 - diff * diff * diff;
145 buffer[i] = sub * sub * sub;
146 total += buffer[i];
147 }
148 }
149
150 if (total == 0) {
151 if (nneighbors) { // everyone's equal distance.
152 std::fill(output, output + ndim, 0);
153 for (size_t i = 0; i < neighbors.size(); ++i) {
154 auto nptr = embedding + neighbors[i].first * ndim;
155 for (int d = 0; d < ndim; ++d) {
156 output[d] += nptr[d];
157 }
158 }
159 for (int d = 0; d < ndim; ++d) {
160 output[d] /= neighbors.size();
161 }
162 } else { // no neighbors at all.
163 std::fill(output, output + ndim, std::numeric_limits<Float>::quiet_NaN());
164 }
165 } else {
166 std::fill(output, output + ndim, 0);
167 for (size_t i = 0; i < neighbors.size(); ++i) {
168 auto nptr = embedding + neighbors[i].first * ndim;
169 Float weight = buffer[i] / total;
170 for (int d = 0; d < ndim; ++d) {
171 output[d] += nptr[d] * weight;
172 }
173 }
174 }
175 }
