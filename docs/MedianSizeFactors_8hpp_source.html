149 template<typename T, typename IDX, typename Ref, typename Out>
150 void run(const tatami::Matrix<T, IDX>* mat, const Ref* ref, Out* output) const {
151 size_t NR = mat->nrow(), NC = mat->ncol();
152 Factory<T, Ref, Out> fact(NR, NC, ref, output);
153 tatami::apply<1>(mat, fact);
154
155 // Mild squeezing towards library size-derived factors.
156 if (prior_count && NR && NC) {
157 const auto& sums = fact.sums;
158 double mean = std::accumulate(sums.begin(), sums.end(), static_cast<T>(0));
159 mean /= NC;
160
161 double reftotal = std::accumulate(ref, ref + NR, static_cast<double>(0));
162
163 if (mean && reftotal) {
164 double scaling = prior_count / mean;
165 for (size_t i = 0; i < NC; ++i) {
166 output[i] += sums[i] * scaling / reftotal;
167 output[i] /= 1.0 + scaling; // adjusting for the addition, as if we were dealing with a library size.
168 }
169 }
170 }
171
172 return;
173 }
