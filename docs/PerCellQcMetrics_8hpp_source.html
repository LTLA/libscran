475 template<typename T, typename IDX>
476 void compute(size_t c, const tatami::SparseRange<T, IDX>& range) {
477 if (output.total) {
478 auto& current = output.total[c];
479 for (size_t i = 0; i < range.number; ++i) {
480 current += range.value[i];
481 }
482 }
483
484 if (output.detected) {
485 auto& current = output.detected[c];
486 for (size_t i = 0; i < range.number; ++i) {
487 current += (range.value[i] != 0);
488 }
489 }
490
491 if (output.max_index || output.max_count) {
492 auto max_count = PerCellQcMetrics::pick_fill_value<Float>();
493 Integer max_index = 0;
494 for (size_t i = 0; i < range.number; ++i) {
495 if (max_count < range.value[i]) {
496 max_count = range.value[i];
497 max_index = range.index[i];
498 }
499 }
500
501 if (max_count <= 0 && range.number < NR) {
502 // Zero is the max.
503 max_count = 0;
504
505 // Finding the index of the first zero by tracking all
506 // indices with non-zero values. This isn't the fastest
507 // approach but it's simple and avoids assuming that
508 // indices are sorted. Hopefully we don't have to hit
509 // this section often.
510 if (output.max_index) {
511 for (size_t i = 0; i < range.number; ++i) {
512 if (range.value[i]) {
513 internal_is_nonzero[range.index[i]] = 1;
514 }
515 }
516 for (size_t r = 0; r < NR; ++r) {
517 if (internal_is_nonzero[r] == 0) {
518 max_index = r;
519 break;
520 }
521 }
522 for (size_t i = 0; i < range.number; ++i) { // setting back to zero.
523 internal_is_nonzero[range.index[i]] = 0;
524 }
525 }
526 }
527
528 if (output.max_index) {
529 output.max_index[c] = max_index;
530 }
531 if (output.max_count) {
532 output.max_count[c] = max_count;
533 }
534 }
535
536 size_t nsubsets = subsets.size();
537 for (size_t s = 0; s < nsubsets; ++s) {
538 const auto& sub = subsets[s];
539
540 if (!output.subset_total.empty() && output.subset_total[s]) {
541 auto& current = output.subset_total[s][c];
542 for (size_t i = 0; i < range.number; ++i) {
543 current += (sub[range.index[i]] != 0) * range.value[i];
544 }
545 }
546
547 if (!output.subset_detected.empty() && output.subset_detected[s]) {
548 auto& current = output.subset_detected[s][c];
549 for (size_t i = 0; i < range.number; ++i) {
550 current += (sub[range.index[i]] != 0) * (range.value[i] != 0);
551 }
552 }
553 }
554 }
555
556 size_t NR;
557 const std::vector<Subset>& subsets;
558 Buffers<Float, Integer>& output;
559 std::vector<uint8_t> internal_is_nonzero;
560 };

669 output.total[range.index[i]] += range.value[i];
670 }
671 }
672

676 }

678

681 for (size_t i = 0; i < range.number; ++i) {
682 auto offset = range.index[i] - start;
683
684 auto& curmax = tracker[offset];
685 if (curmax < range.value[i]) {
686 curmax = range.value[i];

689 }

693 if (static_cast<size_t>(last) == counter) {

696 }

698 }
699 }

704 if (sub[counter] == 0) {

726 void finish() {
727 if (output.max_count || output.max_index) {
728 auto* tracker = (output.max_count ? output.max_count + start : internal_max_count.data());

781 template<class Matrix, typename Subset = const uint8_t*, typename Float, typename Integer>
782 void run(const Matrix* mat, const std::vector<Subset>& subsets, Buffers<Float, Integer>& output) const {
783 if (!output.already_zeroed) {
784 size_t n = mat->ncol();
785 auto check_and_fill = [&](auto* ptr, auto value) -> void {
786 if (ptr) {
787 std::fill(ptr, ptr + n, value);
788 }
789 };
790
791 check_and_fill(output.total, static_cast<Float>(0));
792 check_and_fill(output.detected, static_cast<Integer>(0));
793 check_and_fill(output.max_count, pick_fill_value<Float>());
794 check_and_fill(output.max_index, static_cast<Integer>(0));
795
796 for (size_t s = 0; s < subsets.size(); ++s) {
797 if (!output.subset_total.empty()) {
798 check_and_fill(output.subset_total[s], static_cast<Float>(0));
799 }
800 if (!output.subset_detected.empty()) {
801 check_and_fill(output.subset_detected[s], static_cast<Integer>(0));
802 }
803 }
804 }
805
806 size_t nr = mat->nrow(), nc = mat->ncol();
807 Factory fact(nr, nc, subsets, output);
808 tatami::apply<1>(mat, fact, num_threads);
809 return;
810 }
