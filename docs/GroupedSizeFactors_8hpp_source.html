183 template<typename T, typename IDX, typename Group, typename Ref, typename Out>
184 void run_internal(const tatami::Matrix<T, IDX>* mat, const Group* group, Ref reference, Out* output) const {
185 size_t NR = mat->nrow(), NC = mat->ncol();
186 if (!NC) {
187 return;
188 }
189
190 // Aggregating each group to get a pseudo-bulk sample.
191 auto ngroups = count_ids(NC, group);
192 std::vector<double> combined(ngroups * NR);
193 {
194 std::vector<double*> sums(ngroups);
195 for (size_t i = 0; i < ngroups; ++i) {
196 sums[i] = combined.data() + i * NR;
197 }
198 AggregateAcrossCells aggregator;
199 aggregator.set_num_threads(num_threads).run(mat, group, std::move(sums), std::vector<int*>());
200 }
201
202 size_t ref = 0;
203 if constexpr(std::is_same<Ref, size_t>::value) {
204 ref = reference;
205 } else {
206 // Choosing one of them to be the reference. Here, we borrow some logic
207 // from edgeR and use the one with the largest sum of square roots,
208 // which provides a compromise between transcriptome coverage and
209 // complexity. The root ensures that we don't pick a sample that just
210 // has very high expression in a small subset of genes, while still
211 // remaining responsive to the overall coverage level.
212 double best = 0;
213
214 for (size_t i = 0; i < ngroups; ++i) {
215 auto start = combined.data() + i * NR;
216 double current = 0;
217 for (size_t j = 0; j < NR; ++j) {
218 current += std::sqrt(start[j]);
219 }
220 if (current > best) {
221 ref = i;
222 best = current;
223 }
224 }
225 }
226
227 // Computing median-based size factors. No need to center
228 // here as we'll be recentering afterwards anyway.
229 tatami::ArrayView view(combined.data(), combined.size());
230 tatami::DenseColumnMatrix<T, IDX, decltype(view)> aggmat(NR, ngroups, std::move(view));
231
232 MedianSizeFactors med;
233 med.set_num_threads(num_threads).set_center(false).set_prior_count(prior_count);
234 auto mres = med.run(&aggmat, combined.data() + ref * NR);
235
236 // We need to rescue odd size factors here; if they're zero or infinite,
237 // we lose information about the relative scaling of cells within each group.
238 auto cres = CenterSizeFactors::validate(mres.factors.size(), mres.factors.data());
239 if (cres.has_zero) {
240 if (!handle_zeros) {
241 throw std::runtime_error("all pseudo-cell size factors should be positive");
242 } else {
243 sanitize_zeros(mres.factors);
244 }
245 }
246
247 if (cres.has_non_finite) {
248 if (!handle_non_finite) {
249 throw std::runtime_error("all pseudo-cell size factors should be finite");
250 } else {
251 sanitize_non_finite(mres.factors);
252 }
253 }
254
255 // Propagating to each cell via library size-based normalization.
256 auto aggcolsums = tatami::column_sums(&aggmat, num_threads);
257 auto colsums = tatami::column_sums(mat, num_threads);
258 for (size_t i = 0; i < NC; ++i) {
259 auto curgroup = group[i];
260 if (aggcolsums[curgroup]) {
261 auto scale = static_cast<double>(colsums[i])/static_cast<double>(aggcolsums[curgroup]);
262 output[i] = scale * mres.factors[curgroup];
263 } else {
264 // implies colsums[i] == 0, so instead of getting NaN size factors,
265 // we just set the factors to zero, given that everything in the group is also zero.
266 output[i] = 0;
267 }
268 }
269
270 if (center) {
271 CenterSizeFactors centerer;
272 centerer.run(NC, output);
273 }
274 }
