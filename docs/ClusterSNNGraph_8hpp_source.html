440 igraph::IntegerVector membership_holder;
441 auto& membership = membership_holder.vector;
442 igraph_integer_t nb_clusters;
443 igraph_real_t quality;
444
445 igraph::RNGScope rngs(seed);
446 Results output;
447
448 // No need to free this, as it's just a view.
449 igraph_vector_t weight_view;
450 size_t nedges = igraph_ecount(graph.get_graph());
451 igraph_vector_view(&weight_view, weights, nedges);
452
453 if (!modularity) {
454 output.status = igraph_community_leiden(graph.get_graph(), &weight_view, NULL, resolution, beta, false, iterations, &membership, &nb_clusters, &quality);
455 } else {
456 // More-or-less translated from igraph::cluster_leiden in the R package,
457 // but with the iterations moved into igraph_community_leiden itself.
458 igraph::RealVector strength_holder(igraph_vcount(graph.get_graph()));
459 auto& strength = strength_holder.vector;
460 igraph_strength(graph.get_graph(), &strength, igraph_vss_all(), IGRAPH_ALL, 1, &weight_view);
461
462 double total_weights = std::accumulate(weights, weights + nedges, 0.0);
463 double mod_resolution = resolution / total_weights;
464
465 output.status = igraph_community_leiden(graph.get_graph(), &weight_view, &strength, mod_resolution, beta, false, iterations, &membership, &nb_clusters, &quality);
466 }
467
468 if (!output.status) {
469 size_t ncells = igraph_vcount(graph.get_graph());
470 output.membership.resize(ncells);
471 for (size_t i = 0; i < ncells; ++i) {
472 output.membership[i] = VECTOR(membership)[i];
473 }
474 output.quality = quality;
475 }
