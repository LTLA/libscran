246 template<class MAT, class V, typename B>
247 std::shared_ptr<MAT> run_blocked(std::shared_ptr<MAT> mat, V size_factors, const B* block) const {
248 // One might ask why we don't require a pointer for size_factors here.
249 // It's because size_factors need to be moved into the Delayed operation
250 // anyway, so we might as well ask the user to construct a vector for us.
251 if (size_factors.size() != mat->ncol()) {
252 throw std::runtime_error("number of size factors and columns are not equal");
253 }
254
255 bool has_zero = false;
256 if (center) {
257 // Falls back to centerer.run() if block=NULL.
258 has_zero = centerer.run_blocked(size_factors.size(), size_factors.data(), block);
259 } else {
260 // CenterSizeFactors do their own validity checks,
261 // so we don't need to call it again in that case.
262 has_zero = CenterSizeFactors::validate(size_factors.size(), size_factors.data());
263 }
264
265 if (has_zero) {
266 if (!handle_zeros) {
267 throw std::runtime_error("all size factors should be positive");
268 } else if (size_factors.size()) {
269 // Replacing them with the smallest non-zero size factor, or 1.
270 auto smallest = std::numeric_limits<double>::infinity();
271 for (auto s : size_factors) {
272 if (s && smallest > s) {
273 smallest = s;
274 }
275 }
276
277 if (std::isinf(smallest)) {
278 smallest = 1;
279 }
280
281 for (auto& s : size_factors) {
282 if (s == 0) {
283 s = smallest;
284 }
285 }
286 }
287 }
288
289 double current_pseudo = pseudo_count;
290 if (choose_pseudo_count) {
291 current_pseudo = pseudo_chooser.run(size_factors.size(), size_factors.data());
292 }
293
294 if (sparse_addition && current_pseudo != 1) {
295 for (auto& d : size_factors) {
296 d *= current_pseudo;
297 }
298 current_pseudo = 1; // effectively 1 now.
299 }
300
301 typedef typename MAT::value_type Value_;
302 auto div = tatami::make_DelayedUnaryIsometricOp(std::move(mat), tatami::make_DelayedDivideVectorHelper<true, 1, Value_>(std::move(size_factors)));
303
304 if (current_pseudo == 1) {
305 return tatami::make_DelayedUnaryIsometricOp(std::move(div), tatami::DelayedLog1pHelper<Value_>(2.0));
306 } else {
307 auto add = tatami::make_DelayedUnaryIsometricOp(std::move(div), tatami::make_DelayedAddScalarHelper<Value_>(current_pseudo));
308 return tatami::make_DelayedUnaryIsometricOp(std::move(add), tatami::DelayedLogHelper<Value_>(2.0));
309 }
310 }
