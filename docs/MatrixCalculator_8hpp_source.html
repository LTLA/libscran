20/****************************************************************************
21 * The MatrixCalculator class handles the Factory construction and the apply()
22 * call, given an Overlord object that describes how to process the statistics
23 * in a 'with-AUC' and 'without-AUC' scenario. The overlord should implement:
24 *
25 * - `needs_auc()`, indicating whether AUCs should be computed.
26 * - `obtain_auc_buffer()`, returning a pointer to an array to store the AUCs.
27 *
28 * Check out PairwiseMatrix and ScoreMarkers for concrete examples.
29 ****************************************************************************/

47 template<class Matrix, typename G, class Overlord>
48 State run(const Matrix* p, const G* group, int ngroups, Overlord& overlord) const {
49 std::vector<int> group_size(ngroups);
50 for (size_t i = 0, end = p->ncol(); i < end; ++i) {
51 ++(group_size[group[i]]);
52 }
53 return core(p, group, std::move(group_size), group, ngroups, static_cast<const int*>(NULL), 1, overlord);
54 }

56 template<class Matrix, typename G, typename B, class Overlord>
57 State run_blocked(const Matrix* p, const G* group, int ngroups, const B* block, int nblocks, Overlord& overlord) const {
58 if (block == NULL) {
59 return run(p, group, ngroups, overlord);
60 }
61
62 size_t ncombos = ngroups * nblocks;
63 std::vector<int> combos(p->ncol());
64 std::vector<int> combo_size(ncombos);
65 for (size_t i = 0, end = combos.size(); i < end; ++i) {
66 combos[i] = group[i] * nblocks + block[i];
67 ++(combo_size[combos[i]]);
68 }
69
70 return core(p, combos.data(), std::move(combo_size), group, ngroups, block, nblocks, overlord);
71 }

74 struct AucBundle {
75 AucBundle(size_t ngroups, size_t nblocks, const std::vector<int>& level_size) :
76 paired(nblocks),
77 num_zeros(nblocks, std::vector<int>(ngroups)),
78 totals(nblocks, std::vector<int>(ngroups)),
79 auc_buffer(ngroups * ngroups),
80 denominator(ngroups, std::vector<double>(ngroups))
81 {
82 auto lsIt = level_size.begin();
83 for (size_t g = 0; g < ngroups; ++g) {
84 for (size_t b = 0; b < nblocks; ++b, ++lsIt) {
85 totals[b][g] = *lsIt;
86 }
87 }
88
89 for (size_t b = 0; b < nblocks; ++b) {
90 for (int g1 = 0; g1 < ngroups; ++g1) {
91 for (int g2 = 0; g2 < ngroups; ++g2) {
92 denominator[g1][g2] += totals[b][g1] * totals[b][g2];
93 }
94 }
95 }
96 }

109 template<bool sparse_, bool auc_, typename Data_, typename Index_, typename Level_, typename Group_, typename Block_, class State_, class Overlord_>
110 void by_row(
111 const tatami::Matrix<Data_, Index_>* p,
112 const Level_* level,
113 const std::vector<int>& level_size,
114 const Group_* group,
115 size_t ngroups,
116 const Block_* block,
117 size_t nblocks,
118 State_& state,
119 Overlord_& overlord)
120 const {
121 // Using size_t consistently in this file, so as to avoid bugs with
122 // integer overflow for 'Index_' when computing matrix products.
123 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
124 size_t NC = p->ncol();
125 std::vector<Data_> vbuffer(NC);
126 typename std::conditional<sparse_, std::vector<Index_>, size_t>::type ibuffer(NC);
127 auto ext = tatami::consecutive_extractor<true, sparse_, Data_, Index_>(p, start, length);
128
129 // AUC-only object.
130 typename std::conditional<auc_, AucBundle, DummyBundle>::type auc_info(ngroups, nblocks, level_size);
131
132 size_t nlevels = level_size.size();
133 size_t offset = nlevels * start;
134 for (size_t r = start, end = start + length; r < end; ++r, offset += nlevels) {
135 auto mptr = state.means.data() + offset;
136 auto vptr = state.variances.data() + offset;
137 auto dptr = state.detected.data() + offset;
138
139 if constexpr(!sparse_) {
140 auto ptr = ext->fetch(r, vbuffer.data());
141 feature_selection::blocked_variance_with_mean<true>(ptr, NC, level, nlevels, level_size.data(), mptr, vptr);
142
143 std::fill(dptr, dptr + nlevels, 0);
144 for (size_t j = 0; j < NC; ++j) {
145 dptr[level[j]] += (ptr[j] != 0);
146 }
147
148 if constexpr(auc_) {
149 for (auto& z : auc_info.num_zeros) {
150 std::fill(z.begin(), z.end(), 0);
151 }
152 for (auto& p : auc_info.paired) {
153 p.clear();
154 }
155
156 for (size_t c = 0; c < NC; ++c) {
157 auto b = block[c];
158 auto g = group[c];
159 if (ptr[c]) {
160 auc_info.paired[b].emplace_back(ptr[c], g);
161 } else {
162 ++(auc_info.num_zeros[b][g]);
163 }
164 }
165
166 auto store = overlord.prepare_auc_buffer(r, ngroups);
167 process_auc_for_rows(ngroups, nblocks, threshold, auc_info, store);
168 }
169
170 } else {
171 auto range = ext->fetch(r, vbuffer.data(), ibuffer.data());
172 feature_selection::blocked_variance_with_mean<true>(range, level, nlevels, level_size.data(), mptr, vptr, dptr);
173
174 if constexpr(auc_) {
175 auto nzIt = auc_info.num_zeros.begin();
176 for (const auto& t : auc_info.totals) {
177 std::copy(t.begin(), t.end(), nzIt->begin());
178 ++nzIt;
179 }
180 for (auto& p : auc_info.paired) {
181 p.clear();
182 }
183
184 for (size_t j = 0; j < range.number; ++j) {
185 if (range.value[j]) {
186 size_t c = range.index[j];
187 auto b = block[c];
188 auto g = group[c];
189 auc_info.paired[b].emplace_back(range.value[j], g);
190 --(auc_info.num_zeros[b][g]);
191 }
192 }
193
194 auto store = overlord.prepare_auc_buffer(r, ngroups);
195 process_auc_for_rows(ngroups, nblocks, threshold, auc_info, store);
196 }
197 }
198 }
199 }, static_cast<size_t>(p->nrow()), num_threads);
200 }

202 static void process_auc_for_rows(size_t ngroups, size_t nblocks, double threshold, AucBundle& bundle, double* output) {
203 std::fill(output, output + ngroups * ngroups, 0);
204 auto& auc_buffer = bundle.auc_buffer;
205
206 for (size_t b = 0; b < nblocks; ++b) {
207 auto& pr = bundle.paired[b];
208 auto& nz = bundle.num_zeros[b];
209 const auto& tt = bundle.totals[b];
210
211 std::fill(auc_buffer.begin(), auc_buffer.end(), 0);
212 if (threshold) {
213 compute_pairwise_auc(pr, nz, tt, auc_buffer.data(), threshold, false);
214 } else {
215 compute_pairwise_auc(pr, nz, tt, auc_buffer.data(), false);
216 }
217
218 // Adding to the blocks.
219 for (size_t g = 0, end = auc_buffer.size(); g < end; ++g) {
220 output[g] += auc_buffer[g];
221 }
222 }
223
224 for (size_t g1 = 0; g1 < ngroups; ++g1) {
225 for (size_t g2 = 0; g2 < ngroups; ++g2) {
226 auto& current = output[g1 * ngroups + g2];
227 if (bundle.denominator[g1][g2]) {
228 current /= bundle.denominator[g1][g2];
229 } else {
230 current = std::numeric_limits<double>::quiet_NaN();
231 }
232 }
233 }
234 }

237 template<bool sparse_, typename Data_, typename Index_, typename Level_, class State_>
238 void by_column(const tatami::Matrix<Data_, Index_>* p, const Level_* level, const std::vector<int>& level_size, State_& state) const {
239 size_t nlevels = level_size.size();
240
241 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
242 size_t NC = p->ncol();
243 std::vector<Data_> vbuffer(length);
244 typename std::conditional<sparse_, std::vector<Index_>, size_t>::type ibuffer(length);
245 auto ext = tatami::consecutive_extractor<false, sparse_, Data_, Index_>(p, 0, NC, start, length);
246
247 std::vector<std::vector<double> > tmp_means(nlevels), tmp_vars(nlevels), tmp_detected(nlevels);
248 for (size_t l = 0; l < nlevels; ++ l) {
249 tmp_means[l].resize(length);
250 tmp_vars[l].resize(length);
251 tmp_detected[l].resize(length);
252 }
253 std::vector<int> tmp_counts(nlevels);
254
255 for (size_t c = 0; c < NC; ++c) {
256 auto b = level[c];
257 auto mptr = tmp_means[b].data();
258 auto vptr = tmp_vars[b].data();
259 auto dptr = tmp_detected[b].data();
260
261 if constexpr(!sparse_) {
262 auto ptr = ext->fetch(c, vbuffer.data());
263 tatami::stats::variances::compute_running(ptr, length, mptr, vptr, tmp_counts[b]);
264 for (size_t j = 0; j < length; ++j, ++dptr) {
265 *dptr += (ptr[j] != 0);
266 }
267 } else {
268 auto range = ext->fetch(c, vbuffer.data(), ibuffer.data());
269 tatami::stats::variances::compute_running(range, mptr, vptr, dptr, tmp_counts[b], /* skip_zeros = */ true, /* subtract = */ static_cast<Index_>(start));
270 }
271 }
272
273 for (size_t l = 0; l < nlevels; ++l) {
274 if constexpr(!sparse_) {
275 tatami::stats::variances::finish_running(length, tmp_means[l].data(), tmp_vars[l].data(), tmp_counts[l]);
276 } else {
277 tatami::stats::variances::finish_running(length, tmp_means[l].data(), tmp_vars[l].data(), tmp_detected[l].data(), tmp_counts[l]);
278 }
279 }
280
281 transpose_for_column(tmp_means, state.means.data(), start, length);
282 transpose_for_column(tmp_vars, state.variances.data(), start, length);
283 transpose_for_column(tmp_detected, state.detected.data(), start, length);
284 }, static_cast<size_t>(p->nrow()), num_threads);
285 }

287 static void transpose_for_column(const std::vector<std::vector<double> >& source, double* sink, size_t start, size_t length) {
288 size_t nlevels = source.size();
289 for (size_t r = 0; r < length; ++r) {
290 auto output = sink + (r + start) * nlevels;
291 for (size_t l = 0; l < nlevels; ++l, ++output) {
292 *output = source[l][r];
293 }
294 }
295 }

298 template<typename Data_, typename Index_, typename Level_, typename Group_, typename Block_, class Overlord_>
299 State core(
300 const tatami::Matrix<Data_, Index_>* p,
301 const Level_* level,
302 std::vector<int> level_size,
303 const Group_* group,
304 size_t ngroups,
305 const Block_* block,
306 size_t nblocks,
307 Overlord_& overlord)
308 const {
309 auto ngenes = p->nrow();
310 size_t nlevels = level_size.size();
311 State state(static_cast<size_t>(ngenes) * nlevels);
312
313 if (!overlord.needs_auc()) {
314 if (p->prefer_rows()) {
315 if (p->sparse()) {
316 by_row<true, false>(p, level, level_size, group, ngroups, block, nblocks, state, overlord);
317 } else {
318 by_row<false, false>(p, level, level_size, group, ngroups, block, nblocks, state, overlord);
319 }
320 } else {
321 if (p->sparse()) {
322 by_column<true>(p, level, level_size, state);
323 } else {
324 by_column<false>(p, level, level_size, state);
325 }
326 }
327 } else {
328 // Need to remake this, as there's no guarantee that 'blocks' exists.
329 std::vector<Block_> tmp_blocks;
330 if (!block) {
331 tmp_blocks.resize(p->ncol());
332 block = tmp_blocks.data();
333 }
334
335 if (p->sparse()) {
336 by_row<true, true>(p, level, level_size, group, ngroups, block, nblocks, state, overlord);
337 } else {
338 by_row<false, true>(p, level, level_size, group, ngroups, block, nblocks, state, overlord);
339 }
340 }
341
342 // Dividing through to get the actual detected proportions.
343 // Don't bother parallelizing this, given how simple it is.
344 auto dptr = state.detected.data();
345 for (size_t gene = 0; gene < ngenes; ++gene) {
346 size_t in_offset = gene * nlevels;
347 for (size_t l = 0; l < nlevels; ++l, ++dptr) {
348 if (level_size[l]) {
349 *dptr /= level_size[l];
350 } else {
351 *dptr = std::numeric_limits<double>::quiet_NaN();
352 }
353 }
354 }
355
356 // Moving it in for output's sake.
357 state.level_size = std::move(level_size);
358 return state;
359 }
