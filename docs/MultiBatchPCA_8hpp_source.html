27struct MultiBatchEigenMatrix {
28 MultiBatchEigenMatrix(const Matrix* m, const Eigen::VectorXd* w, const Eigen::VectorXd* mx) : mat(m), weights(w), means(mx) {}
29
30 auto rows() const { return mat->rows(); }
31 auto cols() const { return mat->cols(); }
32
33public:
34 typedef irlba::WrappedWorkspace<Matrix> Workspace;
35
36 Workspace workspace() const {
37 return irlba::wrapped_workspace(mat);
38 }

40 template<class Right>
41 void multiply(const Right& rhs, Workspace& work, Eigen::VectorXd& output) const {
42 irlba::wrapped_multiply(mat, rhs, work, output);
43
44 double sub = means->dot(rhs);
45 for (Eigen::Index i = 0; i < output.size(); ++i) {
46 auto& val = output.coeffRef(i);
47 val -= sub;
48 val *= weights->coeff(i);
49 }

53 struct AdjointWorkspace {
54 AdjointWorkspace(size_t n, irlba::WrappedAdjointWorkspace<Matrix> c) : combined(n), child(std::move(c)) {}
55 Eigen::VectorXd combined;
56 irlba::WrappedAdjointWorkspace<Matrix> child;
57 };

59 AdjointWorkspace adjoint_workspace() const {
60 return AdjointWorkspace(weights->size(), irlba::wrapped_adjoint_workspace(mat));
61 }
62
63 template<class Right>
64 void adjoint_multiply(const Right& rhs, AdjointWorkspace& work, Eigen::VectorXd& output) const {
65 work.combined.noalias() = weights->cwiseProduct(rhs);
66
67 irlba::wrapped_adjoint_multiply(mat, work.combined, work.child, output);
68
69 double sum = work.combined.sum();
70 for (Eigen::Index i = 0; i < output.size(); ++i) {
71 output.coeffRef(i) -= means->coeff(i) * sum;
72 }
73 return;
74 }
75
76public:
77 Eigen::MatrixXd realize() const {
78 Eigen::MatrixXd output = irlba::wrapped_realize(mat);
79
80 for (Eigen::Index c = 0; c < output.cols(); ++c) {
81 for (Eigen::Index r = 0; r < output.rows(); ++r) {
82 auto& val = output.coeffRef(r, c);
83 val -= means->coeff(c);
84 val *= weights->coeff(r);
85 }
86 }
87 return output;
88 }

92 const Eigen::VectorXd* means;

229 template<class Matrix, class Rotation>
230 void reapply(const Matrix& emat,
231 const Eigen::VectorXd& center_v,
232 Eigen::MatrixXd& pcs,
233 const Rotation& rotation,
234 Eigen::VectorXd& variance_explained)
235 const {
236 if constexpr(!std::is_same<Matrix, pca_utils::SparseMatrix>::value) {
237 pcs.noalias() = emat * rotation;
238 } else {
239 size_t nvec = pcs.cols();
240 size_t nrow = emat.rows();
241 size_t ncol = emat.cols();
242
243 Eigen::MatrixXd transposed(nvec, nrow); // using a transposed version for more cache efficiency.
244 transposed.setZero();
245
246 const auto& x = emat.get_values();
247 const auto& i = emat.get_indices();
248 const auto& p = emat.get_pointers();
249
250 if (nthreads == 1) {
251 Eigen::VectorXd multipliers(nvec);
252 for (size_t c = 0; c < ncol; ++c) {
253 multipliers.noalias() = rotation.row(c);
254 auto start = p[c], end = p[c + 1];
255 for (size_t s = start; s < end; ++s) {
256 transposed.col(i[s]).noalias() += x[s] * multipliers;
257 }
258 }
259 } else {
260 const auto& row_nonzero_starts = emat.get_secondary_nonzero_starts();
261
262#ifndef SCRAN_CUSTOM_PARALLEL
263 #pragma omp parallel for num_threads(nthreads)
264 for (int t = 0; t < nthreads; ++t) {
265#else
266 SCRAN_CUSTOM_PARALLEL([&](size_t, int start, int length) -> void {
267 for (int t = start, end = start + length; t < end; ++t) {
268#endif
269
270 const auto& starts = row_nonzero_starts[t];
271 const auto& ends = row_nonzero_starts[t + 1];
272 Eigen::VectorXd multipliers(nvec);
273 for (size_t c = 0; c < ncol; ++c) {
274 multipliers.noalias() = rotation.row(c);
275 auto start = starts[c], end = ends[c];
276 for (size_t s = start; s < end; ++s) {
277 transposed.col(i[s]).noalias() += x[s] * multipliers;
278 }
279 }
280
281#ifndef SCRAN_CUSTOM_PARALLEL
282 }
283#else
284 }
285 }, nthreads, nthreads);
286#endif
287 }
288
289 pcs.noalias() = transposed.adjoint();
290 }
291
292 // Effective centering because I don't want to modify 'emat'.
293 auto pIt = pcs.data();
294 for (size_t i = 0, iend = pcs.cols(); i < iend; ++i) {
295 double meanval = center_v.dot(rotation.col(i));
296 for (size_t j = 0, jend = pcs.rows(); j < jend; ++j, ++pIt) {
297 *pIt -= meanval;
298 }
299 }
300
301 // Variance is a somewhat murky concept with weights, so we just square
302 // it and assume that only the relative value matters.
303 for (auto& d : variance_explained) {
304 d = d * d;
305 }
306 return;
307 }

310 template<typename T, typename IDX, typename Batch>
311 void run(const tatami::Matrix<T, IDX>* mat, const Batch* batch, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
312 const size_t NC = mat->ncol();
313 auto batch_size = block_sizes(NC, batch);
314 auto weights = compute_weights(NC, batch, batch_size);
315
316 Eigen::VectorXd center_v(mat->nrow());
317 Eigen::VectorXd scale_v(mat->nrow());
318
319 // Remember, we want to run the PCA on the modified matrix,
320 // but we want to apply the rotation vectors to the original matrix
321 // (after any centering/scaling but without the batch weights).
322 auto executor = [&](const auto& emat) -> void {
323 irlba::EigenThreadScope t(nthreads);
324 irlba::Irlba irb;
325 irb.set_number(rank);
326
327 MultiBatchEigenMatrix<typename std::remove_reference<decltype(emat)>::type> thing(&emat, &weights, &center_v);
328 if (scale) {
329 irb.run(irlba::Scaled<decltype(thing)>(&thing, &scale_v), pcs, rotation, variance_explained);
330
331 // Dividing the rotation vectors by the scaling factor to mimic
332 // the division of 'emat' by the scaling factor (after centering).
333 Eigen::MatrixXd temp = rotation.array().colwise() / scale_v.array();
334 reapply(emat, center_v, pcs, temp, variance_explained);
335 } else {
336 irb.run(thing, pcs, rotation, variance_explained);
337 reapply(emat, center_v, pcs, rotation, variance_explained);
338 }
339 };
340
341 if (mat->sparse()) {
342 auto emat = create_custom_sparse_matrix(mat, center_v, scale_v, batch, batch_size, total_var);
343 executor(emat);
344 } else {
345 auto emat = create_eigen_matrix_dense(mat, center_v, scale_v, batch, batch_size, total_var);
346 executor(emat);
347 }
348
349 if (transpose) {
350 pcs.adjointInPlace();
351 }
352
353 return;
354 }

450 template<typename T, typename IDX, typename Batch>
451 pca_utils::SparseMatrix create_custom_sparse_matrix(const tatami::Matrix<T, IDX>* mat,
452 Eigen::VectorXd& center_v,
453 Eigen::VectorXd& scale_v,
454 const Batch* batch,
455 const std::vector<int>& batch_size,
456 double& total_var)
457 const {
458
459 auto NR = mat->nrow(), NC = mat->ncol();
460 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
461 auto& ptrs = extracted.ptrs;
462 auto& values = extracted.values;
463 auto& indices = extracted.indices;
464
465 tatami::parallelize([&](size_t, IDX start, IDX length) -> void {
466 size_t nbatchs = batch_size.size();
467 std::vector<double> batch_means(nbatchs);
468 std::vector<int> batch_count(nbatchs);
469
470 for (IDX r = start, end = start + length; r < end; ++r) {
471 auto offset = ptrs[r];
472 size_t num_entries = ptrs[r+1] - offset;
473 auto value_ptr = values.data() + offset;
474 auto index_ptr = indices.data() + offset;
475
476 // Computing the grand mean across all batchs.
477 std::fill(batch_means.begin(), batch_means.end(), 0);
478 std::fill(batch_count.begin(), batch_count.end(), 0);
479 for (size_t i = 0; i < num_entries; ++i) {
480 auto b = batch[index_ptr[i]];
481 batch_means[b] += value_ptr[i];
482 ++batch_count[b];
483 }
484
485 double& grand_mean = center_v[r];
486 grand_mean = 0;
487 for (size_t b = 0; b < nbatchs; ++b) {
488 auto bsize = batch_size[b];
489 if (bsize) {
490 grand_mean += batch_means[b] / bsize;
491 }
492 }
493 grand_mean /= nbatchs;
494
495 // Computing pseudo-variances where each batch's contribution
496 // is weighted inversely proportional to its size. This aims to
497 // match up with the variances used in the PCA but not the
498 // variances of the output components (where weightings are not used).
499 double& proxyvar = scale_v[r];
500 proxyvar = 0;
501 for (size_t b = 0; b < nbatchs; ++b) {
502 double zero_sum = (batch_size[b] - batch_count[b]) * grand_mean * grand_mean;
503 proxyvar += zero_sum / batch_size[b];
504 }
505
506 for (size_t i = 0; i < num_entries; ++i) {
507 double diff = value_ptr[i] - grand_mean;
508 auto bsize = batch_size[batch[index_ptr[i]]];
509 if (bsize) {
510 proxyvar += diff * diff / bsize;
511 }
512 }
513 }
514 }, NR, nthreads);
515
516 total_var = pca_utils::process_scale_vector(scale, scale_v);
517
518 // Actually filling the sparse matrix. Note that this is transposed
519 // because we want genes in the columns.
520 return pca_utils::SparseMatrix(NC, NR, std::move(values), std::move(indices), std::move(ptrs), nthreads);
521 }

524 template<typename T, typename IDX, typename Batch>
525 Eigen::MatrixXd create_eigen_matrix_dense(
526 const tatami::Matrix<T, IDX>* mat,
527 Eigen::VectorXd& center_v,
528 Eigen::VectorXd& scale_v,
529 const Batch* batch,
530 const std::vector<int>& batch_size,
531 double& total_var)
532 const {
533
534 // Extract a column-major matrix with genes in columns.
535 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads);
536
537 // Looping across genes (i.e., columns) to fill center_v and scale_v.
538 size_t NC = emat.cols(), NR = emat.rows();
539 size_t nbatchs = batch_size.size();
540
541 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
542 std::vector<double> mean_buffer(nbatchs);
543 for (size_t c = start, end = start + length; c < end; ++c) {
544 auto ptr = emat.data() + c * NR;
545 std::fill(mean_buffer.begin(), mean_buffer.end(), 0.0);
546 for (size_t r = 0; r < NR; ++r) {
547 mean_buffer[batch[r]] += ptr[r];
548 }
549
550 double& grand_mean = center_v[c];
551 grand_mean = 0;
552 int non_empty_batches = 0;
553 for (size_t b = 0; b < nbatchs; ++b) {
554 const auto& bsize = batch_size[b];
555 if (bsize) {
556 grand_mean += mean_buffer[b] / bsize;
557 ++non_empty_batches;
558 }
559 }
560 grand_mean /= non_empty_batches;
561
562 // We don't actually compute the batchwise variance, but instead
563 // the weighted sum of squared deltas, which is what PCA actually sees.
564 double& proxyvar = scale_v[c];
565 proxyvar = 0;
566 for (size_t r = 0; r < NR; ++r) {
567 const auto& bsize = batch_size[batch[r]];
568 if (bsize) {
569 double diff = ptr[r] - grand_mean;
570 proxyvar += diff * diff / bsize;
571 }
572 }
573 }
574 }, NC, nthreads);
575
576 total_var = pca_utils::process_scale_vector(scale, scale_v);
577 return emat;
578 }
