193 std::deque<WeightedEdge> run(const std::vector<std::vector<int> >& indices) const {
194 size_t ncells = indices.size();
195
196 // Not parallel-frendly, so we don't construct this with the neighbor search
197 std::vector<std::vector<std::pair<int, size_t> > > hosts(ncells);
198 for (size_t i = 0; i < ncells; ++i) {
199 hosts[i].push_back(std::make_pair(0, i)); // each point is its own 0-th nearest neighbor
200 const auto& current = indices[i];
201 int counter = 1;
202 for (auto x : current) {
203 hosts[x].push_back(std::make_pair(counter, i));
204 ++counter;
205 }
206 }
207
208 // Constructing the shared neighbor graph.
209#ifdef SCRAN_LOGGER
210 SCRAN_LOGGER("scran::BuildSNNGraph", "Computing shared neighbor weights");
211#endif
212 std::vector<size_t> current_added;
213 current_added.reserve(ncells);
214 std::vector<int> current_score(ncells);
215 std::deque<WeightedEdge> store;
216
217 for (size_t j = 0; j < ncells; ++j) {
218 const auto& current_neighbors = indices[j];
219
220 for (int i = 0; i <= current_neighbors.size(); ++i) {
221 // First iteration treats 'j' as the zero-th neighbor.
222 // Remaining iterations go through the neighbors of 'j'.
223 const int cur_neighbor = (i==0 ? j : current_neighbors[i-1]);
224
225 // Going through all observations 'h' for which 'cur_neighbor'
226 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
227 // neighbor of both 'h' and 'j'.
228 for (auto& h : hosts[cur_neighbor]) {
229 const auto& othernode = h.second;
230
231 if (othernode < j) { // avoid duplicates from symmetry in the SNN calculations.
232 int& existing_other = current_score[othernode];
233 if (weight_scheme == RANKED) {
234 // Recording the lowest combined rank per neighbor.
235 int currank = h.first + i;
236 if (existing_other == 0) {
237 existing_other = currank;
238 current_added.push_back(othernode);
239 } else if (existing_other > currank) {
240 existing_other = currank;
241 }
242 } else {
243 // Recording the number of shared neighbors.
244 if (existing_other==0) {
245 current_added.push_back(othernode);
246 }
247 ++existing_other;
248 }
249 }
250 }
251 }
252
253 // Converting to edges.
254 for (auto othernode : current_added) {
255 int& otherscore=current_score[othernode];
256 double finalscore;
257 if (weight_scheme == RANKED) {
258 finalscore = static_cast<double>(current_neighbors.size()) - 0.5 * static_cast<double>(otherscore);
259 } else {
260 finalscore = otherscore;
261 if (weight_scheme == JACCARD) {
262 finalscore = finalscore / (2 * (current_neighbors.size() + 1) - finalscore);
263 }
264 }
265
266 // Ensuring that an edge with a positive weight is always reported.
267 store.push_back(WeightedEdge(j, othernode, std::max(finalscore, 1e-6)));
268
269 // Resetting all those added to zero.
270 otherscore=0;
271 }
272 current_added.clear();
273 }
274
275 return store;
276 }
