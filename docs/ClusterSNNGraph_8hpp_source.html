325 Graph build(size_t ncells, const std::deque<BuildSNNGraph::WeightedEdge>& store) const {
326 IgraphVector edge_holder(store.size() * 2);
327 auto& edges = edge_holder.vector;
328
329 IgraphVector weight_holder(store.size());
330 auto& weights = weight_holder.vector;
331
332 size_t counter = 0;
333 for (size_t i = 0, end = store.size(); i < end; ++i, counter += 2) {
334 const auto& edge = store[i];
335 VECTOR(edges)[counter] = std::get<0>(edge);
336 VECTOR(edges)[counter + 1] = std::get<1>(edge);
337 VECTOR(weights)[i] = std::get<2>(edge);
338 }
339
340 IgraphGraph graph(edge_holder, ncells, /* directed = */ false);
341 return Graph(std::move(graph), std::move(weight_holder));

946 Results run(const Graph& graph_info) {
947 IgraphVector membership_holder;
948 auto& membership = membership_holder.vector;
949 igraph_integer_t nb_clusters;
950 igraph_real_t quality;
951
952 IgraphRNGScope rngs(seed);
953 Results output;
954
955 if (!modularity) {
956 for (int i = 0; i < iterations; ++i) {
957 output.status = igraph_community_leiden(graph_info.get_graph(), graph_info.get_weights(), NULL, resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
958 if (output.status) {
959 break;
960 }
961 }
962 } else {
963 // Based on https://igraph.org/c/doc/igraph-Community.html#igraph_community_leiden.
964 IgraphVector degree_holder(igraph_vcount(graph_info.get_graph()));
965 auto& degree = degree_holder.vector;
966 igraph_degree(graph_info.get_graph(), &degree, igraph_vss_all(), IGRAPH_ALL, 1);
967
968 // This assumes that resolution = 1 in the example in the C documentation.
969 // igraph::cluster_leiden in the R package does the same thing.
970 double mod_resolution = resolution / (2 * igraph_ecount(graph_info.get_graph()));
971
972 for (int i = 0; i < iterations; ++i) {
973 output.status = igraph_community_leiden(graph_info.get_graph(), graph_info.get_weights(), &degree, mod_resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
974 if (output.status) {
975 break;
976 }
977 }
978 }
979
980 if (!output.status) {
981 size_t ncells = igraph_vcount(graph_info.get_graph());
982 output.membership.resize(ncells);
983 for (size_t i = 0; i < ncells; ++i) {
984 output.membership[i] = VECTOR(membership)[i];
985 }
986 output.quality = quality;
987 }
988
989 return output;
990 }
