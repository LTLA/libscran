101 void run(size_t n, const double* mean, const double* variance, double* fitted, double* residuals) {
102 xbuffer.resize(n);
103 ybuffer.resize(n);
104
105 size_t counter = 0;
106 for (size_t i = 0; i < n; ++i) {
107 if (!filter || mean[i] >= min_mean) {
108 xbuffer[counter] = mean[i];
109 if (transform) {
110 ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
111 } else {
112 ybuffer[counter] = variance[i];
113 }
114 ++counter;
115 }
116 }
117
118 if (counter < 2) {
119 throw std::runtime_error("not enough observations above the minimum mean");
120 }
121
122 // Determining the left edge. This needs to be done before
123 // run_in_place, which mutates the input xbuffer.
124 size_t left_index = std::min_element(xbuffer.begin(), xbuffer.begin() + counter) - xbuffer.begin();
125 double left_x = xbuffer[left_index];
126
127 fbuffer.resize(counter);
128 rbuffer.resize(counter);
129 smoother.run(counter, xbuffer.data(), ybuffer.data(), NULL, fbuffer.data(), rbuffer.data());
130
131 // Identifying the left-most fitted value.
132 double left_fitted = (transform ? quad(fbuffer[left_index]) : fbuffer[left_index]);
133
134 counter = 0;
135 for (size_t i = 0; i < n; ++i) {
136 if (!filter || mean[i] >= min_mean) {
137 fitted[i] = (transform ? quad(fbuffer[counter]) : fbuffer[counter]);
138 ++counter;
139 } else {
140 fitted[i] = mean[i] / left_x * left_fitted; // draw a y = x line to the origin from the left of the fitted trend.
141 }
142 residuals[i] = variance[i] - fitted[i];
143 }
144 return;
145 }
