195 Results run(const Algorithm* search) const {
196 // Collecting neighbors.
197 size_t ncells = search->nobs();
198 std::vector<std::vector<int> > indices(ncells);
199
200#ifndef SCRAN_CUSTOM_PARALLEL
201 #pragma omp parallel for num_threads(nthreads)
202 for (size_t i = 0; i < ncells; ++i) {
203#else
204 SCRAN_CUSTOM_PARALLEL(ncells, [&](size_t start, size_t end) -> void {
205 for (size_t i = start; i < end; ++i) {
206#endif
207 auto neighbors = search->find_nearest_neighbors(i, num_neighbors);
208 auto& current = indices[i];
209 for (auto x : neighbors) {
210 current.push_back(x.first);
211 }
212 }
213#ifdef SCRAN_CUSTOM_PARALLEL
214 }, nthreads);
215#endif

227 template<class Indices>
228 Results run(const std::vector<Indices>& indices) const {
229 size_t ncells = indices.size();
230
231 // Not parallel-frendly, so we don't construct this with the neighbor search
232 std::vector<std::vector<std::pair<int, int> > > hosts(ncells);
233 for (size_t i = 0; i < ncells; ++i) {
234 hosts[i].push_back(std::make_pair(0, i)); // each point is its own 0-th nearest neighbor
235 const auto& current = indices[i];
236 int counter = 1;
237 for (auto x : current) {
238 hosts[x].push_back(std::make_pair(counter, i));
239 ++counter;
240 }
241 }
242
243 // Constructing the shared neighbor graph.
244 std::vector<std::vector<int> > edge_stores(ncells);
245 std::vector<std::vector<double> > weight_stores(ncells);
246
247#ifndef SCRAN_CUSTOM_PARALLEL
248 #pragma omp parallel num_threads(nthreads)
249 {
250#else
251 SCRAN_CUSTOM_PARALLEL(ncells, [&](size_t start, size_t end) -> void {
252#endif
253
254 std::vector<int> current_score(ncells);
255 std::vector<int> current_added;
256 current_added.reserve(ncells);
257
258#ifndef SCRAN_CUSTOM_PARALLEL
259 #pragma omp for
260 for (size_t j = 0; j < ncells; ++j) {
261#else
262 for (size_t j = start; j < end; ++j) {
263#endif
264
265 const auto& current_neighbors = indices[j];
266 for (int i = 0; i <= current_neighbors.size(); ++i) {
267 // First iteration treats 'j' as the zero-th neighbor.
268 // Remaining iterations go through the neighbors of 'j'.
269 const int cur_neighbor = (i==0 ? j : current_neighbors[i-1]);
270
271 // Going through all observations 'h' for which 'cur_neighbor'
272 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
273 // neighbor of both 'h' and 'j'.
274 for (const auto& h : hosts[cur_neighbor]) {
275 auto othernode = h.second;
276
277 if (othernode < j) { // avoid duplicates from symmetry in the SNN calculations.
278 int& existing_other = current_score[othernode];
279 if (weight_scheme == RANKED) {
280 // Recording the lowest combined rank per neighbor.
281 int currank = h.first + i;
282 if (existing_other == 0) {
283 existing_other = currank;
284 current_added.push_back(othernode);
285 } else if (existing_other > currank) {
286 existing_other = currank;
287 }
288 } else {
289 // Recording the number of shared neighbors.
290 if (existing_other==0) {
291 current_added.push_back(othernode);
292 }
293 ++existing_other;
294 }
295 }
296 }
297 }
298
299 // Converting to edges.
300 auto& current_edges = edge_stores[j];
301 current_edges.reserve(current_added.size() * 2);
302 auto& current_weights = weight_stores[j];
303 current_weights.reserve(current_added.size() * 2);
304
305 for (auto othernode : current_added) {
306 int& otherscore = current_score[othernode];
307 double finalscore;
308 if (weight_scheme == RANKED) {
309 finalscore = static_cast<double>(current_neighbors.size()) - 0.5 * static_cast<double>(otherscore);
310 } else {
311 finalscore = otherscore;
312 if (weight_scheme == JACCARD) {
313 finalscore = finalscore / (2 * (current_neighbors.size() + 1) - finalscore);
314 }
315 }
316
317 current_edges.push_back(j);
318 current_edges.push_back(othernode);
319 current_weights.push_back(std::max(finalscore, 1e-6)); // Ensuring that an edge with a positive weight is always reported.
320
321 // Resetting all those added to zero.
322 otherscore = 0;
323 }
324 current_added.clear();
325
326#ifndef SCRAN_CUSTOM_PARALLEL
327 }
328 }
329#else
330 }
331 }, nthreads);
332#endif
333
334 // Collating the total number of edges.
335 size_t nedges = 0;
336 for (const auto& w : weight_stores) {
337 nedges += w.size();
338 }
339
340 Results output;
341 output.ncells = ncells;
342
343 output.weights.reserve(nedges);
344 for (const auto& w : weight_stores) {
345 output.weights.insert(output.weights.end(), w.begin(), w.end());
346 }
347 weight_stores.clear();
348 weight_stores.shrink_to_fit(); // forcibly release memory so that we have some more space for edges.
349
350 output.edges.reserve(nedges * 2);
351 for (const auto& e : edge_stores) {
352 output.edges.insert(output.edges.end(), e.begin(), e.end());
353 }
354
355 return output;
356 }
