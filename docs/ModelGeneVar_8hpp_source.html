100 void run_blocked(const MAT* p, const B* block, std::vector<Stat*> means, std::vector<Stat*> variances, std::vector<Stat*> fitted, std::vector<Stat*> residuals) {
101 size_t NR = p->nrow(), NC = p->ncol();
102 std::vector<int> block_size(means.size());
103
104#ifdef SCRAN_LOGGER
105 SCRAN_LOGGER("scran::ModelGeneVar", "Estimating mean and variance of each gene");
106#endif
107
108 if (block) {
109 auto copy = block;
110 for (size_t j = 0; j < NC; ++j, ++copy) {
111 ++block_size[*copy];
112 }
113 feature_selection::BlockedVarianceFactory<true, Stat, B, decltype(block_size)> fact(NR, NC, means, variances, block, block_size);
114 tatami::apply<0>(p, fact);
115 } else {
116 block_size[0] = NC;
117 feature_selection::BlockedVarianceFactory<false, Stat, B, decltype(block_size)> fact(NR, NC, means, variances, block, block_size);
118 tatami::apply<0>(p, fact);
119 }
120
121 // Applying the trend fit to each block.
122#ifdef SCRAN_LOGGER
123 SCRAN_LOGGER("scran::ModelGeneVar", "Fitting a mean-variance trend across genes");
124#endif
125
126 for (size_t b = 0; b < block_size.size(); ++b) {
127 if (block_size[b] >= 2) {
128 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
129 } else {
130 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
131 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
132 }
133 }

135 return;

204 template<class MAT, typename B>
205 Results run_blocked(const MAT* mat, const B* block) {
206 int nblocks = 1;
207 if (block) {
208 if (mat->ncol()) {
209 nblocks = *std::max_element(block, block + mat->ncol()) + 1;
210 }
211 }
212
213 Results output(mat->nrow(), nblocks);
214 std::vector<double*> mean_ptr, var_ptr, fit_ptr, resid_ptr;
215 for (int b = 0; b < nblocks; ++b) {
216 mean_ptr.push_back(output.means[b].data());
217 var_ptr.push_back(output.variances[b].data());
218 fit_ptr.push_back(output.fitted[b].data());
219 resid_ptr.push_back(output.residuals[b].data());
220 }
221
222 run_blocked(mat, block, std::move(mean_ptr), std::move(var_ptr), std::move(fit_ptr), std::move(resid_ptr));
