310 void compute_thresholds(size_t n, double* ptr, double& lower_threshold, double& upper_threshold) {
311 if (!n) {
312 lower_threshold = std::numeric_limits<double>::quiet_NaN();
313 upper_threshold = std::numeric_limits<double>::quiet_NaN();
314 return;
315 }
316
317 if (log) {
318 auto copy = ptr;
319 for (size_t i = 0; i < n; ++i, ++copy) {
320 if (*copy > 0) {
321 *copy = std::log(*copy);
322 } else if (*copy == 0) {
323 *copy = -std::numeric_limits<double>::infinity();
324 } else {
325 throw std::runtime_error("cannot log-transform negative values");
326 }
327 }
328 }
329
330 // First, getting the median.
331 size_t halfway = n / 2;
332 double medtmp = compute_median(ptr, halfway, n);
333
334 if (log && std::isinf(medtmp)) {
335 if (lower) {
336 lower_threshold = 0;
337 } else {
338 lower_threshold = -std::numeric_limits<double>::infinity();
339 }
340
341 if (upper) {
342 upper_threshold = 0;
343 } else {
344 upper_threshold = std::numeric_limits<double>::infinity();
345 }
346
347 return;
348 }
349
350 // Now getting the MAD. No need to protect against -inf here;
351 // if the median is finite, the MAD must also be finite.
352 auto copy = ptr;
353 for (size_t i = 0; i < n; ++i, ++copy) {
354 *copy = std::abs(*copy - medtmp);
355 }
356 double madtmp = compute_median(ptr, halfway, n);
357
358 madtmp *= 1.4826; // for equivalence with the standard deviation under normality.
359
360 // Computing the outputs.
361 if (lower) {
362 lower_threshold = medtmp - nmads * madtmp;
363 if (log) {
364 lower_threshold = std::exp(lower_threshold);
365 }
366 } else {
367 lower_threshold = -std::numeric_limits<double>::infinity();
368 }
369
370 if (upper) {
371 upper_threshold = medtmp + nmads * madtmp;
372 if (log) {
373 upper_threshold = std::exp(upper_threshold);
374 }
375 } else {
376 upper_threshold = std::numeric_limits<double>::infinity();
377 }
378
379 return;
380 }

382 static double compute_median (double* ptr, size_t halfway, size_t n) {
383 std::nth_element(ptr, ptr + halfway, ptr + n);
384 double medtmp = *(ptr + halfway);
385
386 if (n % 2 == 0) {
387 std::nth_element(ptr, ptr + halfway - 1, ptr + n);
388 medtmp = (medtmp + *(ptr + halfway - 1))/2;
389 }
390
391 return medtmp;
392 }
