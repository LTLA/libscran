462 template<class Matrix, typename G, typename B, typename Stat>
463 void run_blocked(const Matrix* p, const G* group, const B* block,
464 std::vector<std::vector<Stat*> > means,
465 std::vector<std::vector<Stat*> > detected,
466 std::vector<std::vector<Stat*> > cohen,
467 std::vector<std::vector<Stat*> > auc,
468 std::vector<std::vector<Stat*> > lfc,
469 std::vector<std::vector<Stat*> > delta_detected)
470 const {
471 differential_analysis::EffectsCalculator runner(nthreads, threshold);
472
473 int ngroups = means.size();
474 int nblocks = (ngroups ? means[0].size() : 0); // no blocks = no groups.
475 Overlord<Stat> overlord(p->nrow(), ngroups, auc.empty());
476 auto state = runner.run_blocked(p, group, ngroups, block, nblocks, overlord);
477
478 int ncombos = ngroups * nblocks;
479 std::vector<Stat*> means2(ncombos), detected2(ncombos);
480 auto mIt = means2.begin(), dIt = detected2.begin();
481 for (int g = 0; g < ngroups; ++g) {
482 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
483 *mIt = means[g][b];
484 *dIt = detected[g][b];
485 }
486 }
487
488 process_simple_effects(state, means2, detected2, cohen, lfc, delta_detected);
489 summarize_auc(state, auc, overlord.auc_buffer);
490 }

522 template<typename Stat>
523 void process_simple_effects(
524 const differential_analysis::EffectsCalculator::State& state,
525 std::vector<Stat*>& means,
526 std::vector<Stat*>& detected,
527 std::vector<std::vector<Stat*> >& cohen,
528 std::vector<std::vector<Stat*> >& lfc,
529 std::vector<std::vector<Stat*> >& delta_detected)
530 const {
531 auto ngenes = state.ngenes;
532 const auto& level_size = state.level_size;
533 auto nlevels = level_size.size();
534 auto ngroups = state.ngroups;
535 auto nblocks = state.nblocks;
536 const auto* tmp_means = state.means.data();
537 const auto* tmp_variances = state.variances.data();
538 const auto* tmp_detected = state.detected.data();
539
540 // Transferring the block-wise statistics over.
541 {
542 auto my_means = tmp_means;
543 auto my_detected = tmp_detected;
544 for (size_t gene = 0; gene < ngenes; ++gene) {
545 for (size_t l = 0; l < nlevels; ++l, ++my_means, ++my_detected) {
546 means[l][gene] = *my_means;
547 detected[l][gene] = *my_detected;
548 }
549 }
550 }
551
552 size_t holding = ngenes * ngroups;
553 std::vector<double> full_cohen(cohen.size() ? holding : 0),
554 full_delta_detected(delta_detected.size() ? holding : 0),
555 full_lfc(lfc.size() ? holding : 0);
556
557 // Looping over each group and computing the various summaries. We do this on
558 // a per-group basis to avoid having to store the full group-by-group matrix of
559 // effect sizes that we would otherwise need as input to SummarizeEffects.
560 for (int group = 0; group < ngroups; ++group) {
561#ifndef SCRAN_CUSTOM_PARALLEL
562 #pragma omp parallel num_threads(nthreads)
563 {
564 std::vector<double> effect_buffer(ngroups);
565 #pragma omp for
566 for (size_t gene = 0; gene < ngenes; ++gene) {
567#else
568 SCRAN_CUSTOM_PARALLEL(ngenes, [&](size_t start, size_t end) -> void {
569 std::vector<double> effect_buffer(ngroups);
570 for (size_t gene = start; gene < end; ++gene) {
571#endif
572
573 size_t in_offset = nlevels * gene;
574 auto my_means = tmp_means + in_offset;
575 auto my_variances = tmp_variances + in_offset;
576 auto my_detected = tmp_detected + in_offset;
577
578 auto full_offset = gene * ngroups;
579 if (cohen.size()) {
580 auto cohen_ptr = full_cohen.data() + full_offset;
581 differential_analysis::compute_pairwise_cohens_d(group, my_means, my_variances, level_size, ngroups, nblocks, threshold, cohen_ptr);
582 differential_analysis::summarize_comparisons(ngroups, cohen_ptr, group, gene, cohen, effect_buffer);
583 }
584
585 if (delta_detected.size()) {
586 auto delta_detected_ptr = full_delta_detected.data() + full_offset;
587 differential_analysis::compute_pairwise_simple_diff(group, my_detected, level_size, ngroups, nblocks, delta_detected_ptr);
588 differential_analysis::summarize_comparisons(ngroups, delta_detected_ptr, group, gene, delta_detected, effect_buffer);
589 }
590
591 if (lfc.size()) {
592 auto lfc_ptr = full_lfc.data() + full_offset;
593 differential_analysis::compute_pairwise_simple_diff(group, my_means, level_size, ngroups, nblocks, lfc_ptr);
594 differential_analysis::summarize_comparisons(ngroups, lfc_ptr, group, gene, lfc, effect_buffer);
595 }
596
597#ifndef SCRAN_CUSTOM_PARALLEL
598 }
599 }
600#else
601 }
602 }, nthreads);
603#endif
604
605 if (cohen.size() && cohen[differential_analysis::summary::MIN_RANK].size()) {
606 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_cohen.data(), cohen[differential_analysis::summary::MIN_RANK][group], nthreads);
607 }
608
609 if (lfc.size() && lfc[differential_analysis::summary::MIN_RANK].size()) {
610 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_lfc.data(), lfc[differential_analysis::summary::MIN_RANK][group], nthreads);
611 }
612
613 if (delta_detected.size() && delta_detected[differential_analysis::summary::MIN_RANK].size()) {
614 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_delta_detected.data(), delta_detected[differential_analysis::summary::MIN_RANK][group], nthreads);
615 }
616 }
617 }

647 struct Results {
651 Results() {}
652
653 Results(
654 size_t ngenes,
655 int ngroups,
656 int nblocks,
657 const ComputeSummaries& do_cohen,
658 const ComputeSummaries& do_auc,
659 const ComputeSummaries& do_lfc,
660 const ComputeSummaries& do_delta_detected)
661 {
662 auto fill_inner = [&](int N, auto& type) {
663 type.reserve(N);
664 for (int n = 0; n < N; ++n) {
665 type.emplace_back(ngenes);
666 }
667 };
668
669 means.resize(ngroups);
670 detected.resize(ngroups);
671 for (int g = 0; g < ngroups; ++g) {
672 fill_inner(nblocks, means[g]);
673 fill_inner(nblocks, detected[g]);
674 }
675
676 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
677 bool has_any = false;
678 for (size_t i = 0; i < do_this.size(); ++i) {
679 if (do_this[i]) {
680 has_any = true;
681 break;
682 }
683 }
684
685 if (has_any) {
686 effect.resize(differential_analysis::n_summaries);
687 if (do_this[differential_analysis::MIN]) {
688 fill_inner(ngroups, effect[differential_analysis::MIN]);
689 }
690 if (do_this[differential_analysis::MEAN]) {
691 fill_inner(ngroups, effect[differential_analysis::MEAN]);
692 }
693 if (do_this[differential_analysis::MEDIAN]) {
694 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
695 }
696 if (do_this[differential_analysis::MAX]) {
697 fill_inner(ngroups, effect[differential_analysis::MAX]);
698 }
699 if (do_this[differential_analysis::MIN_RANK]) {
700 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
701 }
702 }
703 return;
704 };
705
706 fill_effect(do_cohen, cohen);
707 fill_effect(do_auc, auc);
708 fill_effect(do_lfc, lfc);
709 fill_effect(do_delta_detected, delta_detected);
710 return;
711 }
