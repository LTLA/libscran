86 void run(const tatami::Matrix<T, IDX>* mat, const Group* group, Out* output) {
87 size_t NR = mat->nrow(), NC = mat->ncol();
88 if (!NC) {
89 return;
90 }
91
92 // Aggregating each group to get a pseudo-bulk sample.
93 auto ngroups = *std::max_element(group, group + NC) + 1;
94 std::vector<double> combined(ngroups * NR);
95 {
96 std::vector<double*> sums(ngroups);
97 for (size_t i = 0; i < ngroups; ++i) {
98 sums[i] = combined.data() + i * NR;
99 }
100 AggregateAcrossCells aggregator;
101 aggregator.run(mat, group, std::move(sums), std::vector<int*>());
102 }
103
104 // Choosing one of them to be the reference. Here, we borrow some logic
105 // from edgeR and use the one with the largest sum of square roots,
106 // which provides a compromise between transcriptome coverage and
107 // complexity. The root ensures that we don't pick a sample that just
108 // has very high expression in a small subset of genes, while still
109 // remaining responsive to the overall coverage level.
110 size_t ref = 0;
111 double best = 0;
112
113 for (size_t i = 0; i < ngroups; ++i) {
114 auto start = combined.data() + i * NR;
115 double current = 0;
116 for (size_t j = 0; j < NR; ++j) {
117 current += std::sqrt(start[j]);
118 }
119 if (current > best) {
120 ref = i;
121 best = current;
122 }
123 }
124
125 // Computing median-based size factors.
126 tatami::ArrayView view(combined.data(), combined.size());
127 tatami::DenseColumnMatrix<T, IDX, decltype(view)> aggmat(NR, ngroups, std::move(view));
128 MedianSizeFactors med;
129 auto mres = med.run(&aggmat, combined.data() + ref * NR);
130
131 // Propagating to each cell via library size-based normalization.
132 auto aggcolsums = tatami::column_sums(&aggmat);
133 auto colsums = tatami::column_sums(mat);
134 for (size_t i = 0; i < NC; ++i) {
135 auto curgroup = group[i];
136 auto scale = static_cast<double>(colsums[i])/static_cast<double>(aggcolsums[curgroup]);
137 output[i] = scale * mres.factors[curgroup];
138 }
139
140 // Throwing in some centering.
141 if (center) {
142 CenterSizeFactors centerer;
143 centerer.run(NC, output);
144 }
145 }
