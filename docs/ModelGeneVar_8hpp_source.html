156
157 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
158 std::vector<Stat_> tmp_means(nblocks);
159 std::vector<Stat_> tmp_vars(nblocks);
160
161 std::vector<Data_> buffer(NC);
162 auto ext = tatami::consecutive_extractor<true, false>(mat, start, length);
163
164 for (Index_ r = start, end = start + length; r < end; ++r) {
165 auto ptr = ext->fetch(r, buffer.data());
166 feature_selection::blocked_variance_with_mean<blocked_>(ptr, NC, block, nblocks, block_size.data(), tmp_means.data(), tmp_vars.data());
167 for (size_t b = 0; b < nblocks; ++b) {
168 means[b][r] = tmp_means[b];
169 variances[b][r] = tmp_vars[b];
170 }
171 }
172 }, NR, num_threads);

203 void compute_dense_column(const tatami::Matrix<Data_, Index_>* mat, std::vector<Stat_*>& means, std::vector<Stat_*>& variances, const Block_* block, const std::vector<Index_>& block_size) const {
204 auto nblocks = block_size.size();
205 auto NR = mat->nrow(), NC = mat->ncol();
206
207 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
208 std::vector<Data_> buffer(length);
209 auto ext = tatami::consecutive_extractor<false, false>(mat, 0, NC, start, length);
210
211 // Shifting pointers to account for the new start point.
212 auto mcopy = means;
213 auto vcopy = variances;
214 for (Index_ b = 0; b < nblocks; ++b) {
215 mcopy[b] += start;
216 vcopy[b] += start;
217 }
218
219 std::vector<Index_> counts(nblocks);
220 for (Index_ c = 0; c < NC; ++c) {
221 auto ptr = ext->fetch(c, buffer.data());
222 auto b = feature_selection::get_block<blocked_>(c, block);
223 tatami::stats::variances::compute_running(ptr, length, mcopy[b], vcopy[b], counts[b]);
224 }
225
226 for (size_t b = 0; b < nblocks; ++b) {
227 tatami::stats::variances::finish_running(length, mcopy[b], vcopy[b], counts[b]);
228 }
229 }, NR, num_threads);

259 template<bool blocked_, typename Data_, typename Index_, typename Stat_, typename Block_>
260 void compute(const tatami::Matrix<Data_, Index_>* mat, std::vector<Stat_*>& means, std::vector<Stat_*>& variances, const Block_* block, const std::vector<Index_>& block_size) const {
261 if (mat->prefer_rows()) {
262 if (mat->sparse()) {
263 compute_sparse_row<blocked_>(mat, means, variances, block, block_size);
264 } else {
265 compute_dense_row<blocked_>(mat, means, variances, block, block_size);
266 }
267
268 } else {
269 // Set everything to zero before computing the running statistics.
270 auto NR = mat->nrow();
271 for (auto& mptr : means) {
272 std::fill(mptr, mptr + NR, 0);
273 }
274 for (auto& vptr : variances) {
275 std::fill(vptr, vptr + NR, 0);
276 }
277
278 if (mat->sparse()) {
279 compute_sparse_column<blocked_>(mat, means, variances, block, block_size);
280 } else {
281 compute_dense_column<blocked_>(mat, means, variances, block, block_size);
282 }
283 }

341 template<typename Value_, typename Index_, typename Block_, typename Stat_>
342 void run_blocked_with_average(
343 const tatami::Matrix<Value_, Index_>* mat,
344 const Block_* block,
345 std::vector<Stat_*> means,
346 std::vector<Stat_*> variances,
347 std::vector<Stat_*> fitted,
348 std::vector<Stat_*> residuals,
349 Stat_* ave_means,
350 Stat_* ave_variances,
351 Stat_* ave_fitted,
352 Stat_* ave_residuals)
353 const {
354 Index_ NR = mat->nrow(), NC = mat->ncol();
355 std::vector<Index_> block_size;
356
357 if (block) {
358 block_size = tabulate_ids(NC, block);
359 compute<true>(mat, means, variances, block, block_size);
360 } else {
361 block_size.push_back(NC); // everything is one big block.
362 compute<false>(mat, means, variances, block, block_size);
363 }
364
365 // Applying the trend fit to each block.
366 FitTrendVar fit;
367 fit.set_span(span);
368 fit.set_minimum_mean(min_mean);
369 fit.set_use_fixed_width(use_fixed_width);
370 fit.set_fixed_width(fixed_width);
371 fit.set_minimum_window_count(minimum_window_count);
372
373 for (size_t b = 0; b < block_size.size(); ++b) {
374 if (block_size[b] >= 2) {
375 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
376 } else {
377 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
378 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
379 }
380 }
381
382 // Computing averages under different policies.
383 if (ave_means || ave_variances || ave_fitted || ave_residuals) {
384 if (block_weight_policy == WeightPolicy::EQUAL) {
385 if (ave_means) {
386 average_vectors(NR, means, ave_means);
387 }
388 if (ave_variances) {
389 average_vectors(NR, variances, ave_variances);
390 }
391 if (ave_fitted) {
392 average_vectors(NR, fitted, ave_fitted);
393 }
394 if (ave_residuals) {
395 average_vectors(NR, residuals, ave_residuals);
396 }
397
398 } else {
399 std::vector<double> block_weight(block_size.begin(), block_size.end());
400 if (block_weight_policy == WeightPolicy::VARIABLE) {
401 for (auto& bw : block_weight) {
402 bw = variable_block_weight(bw, variable_block_weight_parameters);
403 }
404 }
405
406 if (ave_means) {
407 average_vectors_weighted(NR, means, block_weight.data(), ave_means);
408 }
409 if (ave_variances) {
410 average_vectors_weighted(NR, variances, block_weight.data(), ave_variances);
411 }
412 if (ave_fitted) {
413 average_vectors_weighted(NR, fitted, block_weight.data(), ave_fitted);
414 }
415 if (ave_residuals) {
416 average_vectors_weighted(NR, residuals, block_weight.data(), ave_residuals);
417 }
418 }
419 }
420
421 return;
422 }
