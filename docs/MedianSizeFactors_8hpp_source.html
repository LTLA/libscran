169 template<typename T, typename IDX, typename Ref, typename Out>
170 void run(const tatami::Matrix<T, IDX>* mat, const Ref* ref, Out* output) const {
171 size_t NR = mat->nrow(), NC = mat->ncol();
172 Factory<T, Ref, Out> fact(NR, NC, ref, output);
173 tatami::apply<1>(mat, fact, num_threads);
174
175 /* Mild squeezing towards library size-derived factors. Basically,
176 * we're adding a scaled version of the reference profile to each
177 * column, before normalizing against the reference profile. Given gene
178 * i and cell j:
179 *
180 * ratio_{ij} = (y_{ij} + ref_i * extra_j) / ref_i
181 * = (y_{ij} / ref_i) + extra_j
182 *
183 * which means that the "shrunken" size factor is:
184 *
185 * median(ratio_{ij}) = median(y_{ij} / ref_i) + extra_j
186 *
187 * This allows us to avoid the actual addition, as we can compute the
188 * unshrunken size factor first and then add the j-specific scaling
189 * later. This is important as otherwise we'd need to make two passes
190 * over the matrix; once to get the mean library size to compute
191 * extra_j, and then again to compute the shrunken factors.
192 *
193 * Incidentally, extra_j is defined as:
194 *
195 * extra_j = p * t_j / T / R
196 *
197 * where p is the constant prior count, t_j is the library size for j,
198 * T is the mean library size across all j, and R is the library size
199 * for the reference profile. Basically, p * ref_i / R is how we
200 * "spread out" the prior count across all genes based on their
201 * relative abundance in the reference profile, while t_j / T
202 * represents the library size factor that we are shrinking towards.
203 *
204 * The addition of extra_j means that the shrunken size factor is
205 * slightly too big to normalize against the reference. To adjust for
206 * this, consider some hypothetical position-invariant function f()
207 * such that f(y_{ij} / ref_i) yields the true size factor x_j. In this
208 * case, our shrunken size factor can be written as
209 *
210 * f(ratio_{ij}) = x_j + extra_j
211 *
212 * To correct our shrunken size factor to x_j, we need to divide it by:
213 *
214 * (x_j + extra_j) / x_j = 1 + (extra_j / x_j)
215 *
216 * As an approximation, we assume x_j =~ t_j / R, i.e., library size
217 * normalization against the reference. This simplifies the correction:
218 *
219 * 1 + (p / T)
220 *
221 * which is what we apply to our actual median-based shrunken factor.
222 */
223 if (prior_count && NR && NC) {
224 const auto& sums = fact.sums;
225 double mean = std::accumulate(sums.begin(), sums.end(), static_cast<T>(0));
226 mean /= NC;
227
228 double reftotal = std::accumulate(ref, ref + NR, static_cast<double>(0));
229
230 if (mean && reftotal) {
231 double scaling = prior_count / mean;
232 for (size_t i = 0; i < NC; ++i) {
233 output[i] += sums[i] * scaling / reftotal;
234 output[i] /= 1.0 + scaling;
235 }
236 }
237 }
238
239 // Throwing in some centering.
240 if (center) {
241 CenterSizeFactors centerer;
242 centerer.run(NC, output);
243 }
244
245 return;
246 }
