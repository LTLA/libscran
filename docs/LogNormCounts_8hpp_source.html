268 template<class MAT, class V, typename B>
269 std::shared_ptr<MAT> run_blocked(std::shared_ptr<MAT> mat, V size_factors, const B* block) const {
270 // One might ask why we don't require a pointer for size_factors here.
271 // It's because size_factors need to be moved into the Delayed operation
272 // anyway, so we might as well ask the user to construct a vector for us.
273 if (size_factors.size() != mat->ncol()) {
274 throw std::runtime_error("number of size factors and columns are not equal");
275 }
276
277 typename CenterSizeFactors::Results cresults;
278 if (center) {
279 cresults = centerer.run_blocked(size_factors.size(), size_factors.data(), block);
280 } else {
281 for (size_t i = 0, end = size_factors.size(); i < end; ++i) {
282 if (size_factors[i] < 0) {
283 throw std::runtime_error("negative size factors detected");
284 }
285 if (size_factors[i] == 0) {
286 cresults.has_zero = true;
287 } else if (!std::isfinite(size_factors[i])) {
288 cresults.has_non_finite = true;
289 }
290 }
291 }
292
293 if (cresults.has_zero) {
294 if (!handle_zeros) {
295 throw std::runtime_error("all size factors should be positive");
296 } else {
297 sanitize_zeros(size_factors);
298 }
299 }
300
301 if (cresults.has_non_finite) {
302 if (!handle_non_finite) {
303 throw std::runtime_error("all size factors should be finite");
304 } else {
305 sanitize_non_finite(size_factors);
306 }
307 }
308
309 double current_pseudo = pseudo_count;
310 if (choose_pseudo_count) {
311 current_pseudo = pseudo_chooser.run(size_factors.size(), size_factors.data());
312 }
313
314 if (sparse_addition && current_pseudo != 1) {
315 for (auto& d : size_factors) {
316 d *= current_pseudo;
317 }
318 current_pseudo = 1; // effectively 1 now.
319 }
320
321 typedef typename MAT::value_type Value_;
322 auto div = tatami::make_DelayedUnaryIsometricOp(std::move(mat), tatami::make_DelayedDivideVectorHelper<true, 1, Value_>(std::move(size_factors)));
323
324 if (current_pseudo == 1) {
325 return tatami::make_DelayedUnaryIsometricOp(std::move(div), tatami::DelayedLog1pHelper<Value_>(2.0));
326 } else {
327 auto add = tatami::make_DelayedUnaryIsometricOp(std::move(div), tatami::make_DelayedAddScalarHelper<Value_>(current_pseudo));
328 return tatami::make_DelayedUnaryIsometricOp(std::move(add), tatami::DelayedLogHelper<Value_>(2.0));
329 }
330 }
