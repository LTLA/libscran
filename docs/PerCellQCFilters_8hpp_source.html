215#endif
216 auto res = fun(ncells, detected, filter_by_detected);
217 output.detected = res.lower;
218 for (size_t i = 0; i < ncells; ++i) {
219 overall_filter[i] |= filter_by_detected[i];
220 }
221 }

223 // Filtering to remove outliers on the subset proportions.
224#ifdef SCRAN_LOGGER

226#endif
227
228 size_t nsubsets = subset_proportions.size();
229 if (filter_by_subset_proportions.size() != nsubsets) {
230 throw std::runtime_error("mismatching number of input/outputs for subset proportion filters");

234 output.subset_proportions.resize(nsubsets);

238 auto res = fun(ncells, subset_proportions[s], dump);
239 output.subset_proportions[s] = res.upper;
