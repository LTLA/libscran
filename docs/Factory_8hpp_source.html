45 template<typename Row>
46 void compute(size_t i, const Row& row) {
47 constexpr bool is_sparse = !std::is_pointer<Row>::value;
48
49 if (this->sums.size()) {
50 std::fill(tmp_sums.begin(), tmp_sums.end(), 0);
51
52 if constexpr(is_sparse) {
53 for (size_t i = 0; i < row.number; ++i) {
54 tmp_sums[factor[row.index[i]]] += row.value[i];
55 }
56 } else {
57 for (size_t j = 0; j < NC; ++j) {
58 tmp_sums[factor[j]] += row[j];
59 }
60 }
61
62 // Computing before transferring for more cache-friendliness.
63 for (size_t l = 0; l < tmp_sums.size(); ++l) {
64 sums[l][i] = tmp_sums[l];
65 }
66 }
67
68 if (this->detected.size()) {
69 std::fill(tmp_detected.begin(), tmp_detected.end(), 0);
70
71 if constexpr(is_sparse) {
72 for (size_t i = 0; i < row.number; ++i) {
73 tmp_detected[factor[row.index[i]]] += (row.value[i] > 0);
74 }
75 } else {
76 for (size_t j = 0; j < NC; ++j) {
77 tmp_detected[factor[j]] += (row[j] > 0);
78 }
79 }
80
81 for (size_t l = 0; l < tmp_detected.size(); ++l) {
82 detected[l][i] = tmp_detected[l];
83 }
84 }
85 }

114 template<typename Column>
115 void add(const Column& col) {
116 constexpr bool is_sparse = !std::is_pointer<Column>::value;
117 auto current = factor[counter];
118
119 if (sums.size()) {
120 auto& cursum = sums[current];
121 if constexpr(is_sparse) {
122 for (size_t i = 0; i < col.number; ++i) {
123 cursum[col.index[i]] += col.value[i];
124 }
125 } else {
126 for (size_t i = start; i < end; ++i) {
127 cursum[i] += col[i - start];
128 }
129 }
130 }
131
132 if (detected.size()) {
133 auto& curdetected = detected[current];
134 if constexpr(is_sparse) {
135 for (size_t i = 0; i < col.number; ++i) {
136 curdetected[col.index[i]] += (col.value[i] > 0);
137 }
138 } else {
139 for (size_t i = start; i < end; ++i) {
140 curdetected[i] += (col[i - start] > 0);
141 }
142 }
143 }
144
145 ++counter;
146 return;
147 }
