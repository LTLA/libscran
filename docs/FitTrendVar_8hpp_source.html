196 void run(size_t n, const double* mean, const double* variance, double* fitted, double* residuals) const {
197 std::vector<double> xbuffer(n), ybuffer(n);
198
199 size_t counter = 0;
200 for (size_t i = 0; i < n; ++i) {
201 if (!filter || mean[i] >= min_mean) {
202 xbuffer[counter] = mean[i];
203 if (transform) {
204 ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
205 } else {
206 ybuffer[counter] = variance[i];
207 }
208 ++counter;
209 }
210 }
211
212 if (counter < 2) {
213 throw std::runtime_error("not enough observations above the minimum mean");
214 }
215
216 // Determining the left edge. This needs to be done before
217 // run_in_place, which mutates the input xbuffer.
218 size_t left_index = std::min_element(xbuffer.begin(), xbuffer.begin() + counter) - xbuffer.begin();
219 double left_x = xbuffer[left_index];
220
221 WeightedLowess::WeightedLowess<> smoother;
222 if (use_fixed_width) {
223 smoother.set_span(minimum_window_count);
224 smoother.set_span_as_proportion(false);
225 smoother.set_min_width(fixed_width);
226 } else {
227 smoother.set_span(span);
228 }
229
230 std::vector<double> fbuffer(counter), rbuffer(counter);
231 smoother.run(counter, xbuffer.data(), ybuffer.data(), NULL, fbuffer.data(), rbuffer.data());
232
233 // Identifying the left-most fitted value.
234 double left_fitted = (transform ? quad(fbuffer[left_index]) : fbuffer[left_index]);
235
236 counter = 0;
237 for (size_t i = 0; i < n; ++i) {
238 if (!filter || mean[i] >= min_mean) {
239 fitted[i] = (transform ? quad(fbuffer[counter]) : fbuffer[counter]);
240 ++counter;
241 } else {
242 fitted[i] = mean[i] / left_x * left_fitted; // draw a y = x line to the origin from the left of the fitted trend.
243 }
244 residuals[i] = variance[i] - fitted[i];
245 }
246 return;
247 }
