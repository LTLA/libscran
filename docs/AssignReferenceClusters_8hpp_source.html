230 size_t nclusters = built.size();
231 int ndim = (built.size() ? built.front()->ndim() : 0); // setting to zero doesn't matter if there are no clusters.
232
233 // This part is pretty much stolen from singlepp.
234 std::vector<Index> search(nclusters);
235 std::vector<Float> left(nclusters), right(nclusters);
236
237 for (size_t c = 0; c < nclusters; ++c) {
238 size_t denom = built[c]->nobs() - 1;
239 double prod = denom * quantile;
240 auto k = std::ceil(prod) + 1;
241 search[c] = k;
242 left[c] = static_cast<double>(k - 1) - prod;
243 right[c] = prod - static_cast<double>(k - 2);
244 }
