303 Thresholds output;
304 std::vector<double> workspace(n);
305 std::vector<int> starts;
306 if (block) {

308 }
309
310 // Filtering on the number of detected features.
311 {
312 ComputeMedianMad meddler;
313 meddler.set_log(true);
314 auto detected_res = meddler.run_blocked(n, block, starts, buffers.detected, workspace.data());
315
316 ChooseOutlierFilters filter;
317 filter.set_num_mads(detected_num_mads);
318 filter.set_min_diff(-std::log(1 - min_detected_drop));
319 filter.set_upper(false);
320 auto detected_filt = filter.run(std::move(detected_res));
321
322 output.detected = std::move(detected_filt.lower);
323 }
324
325 // Filtering on the IgG subset totals.
326 {
327 ComputeMedianMad meddler;
328 meddler.set_log(true);
329
330 ChooseOutlierFilters filter;
331 filter.set_num_mads(subset_num_mads);
332 filter.set_lower(false);
333
334 size_t nsubsets = buffers.subset_totals.size();
335 for (size_t s = 0; s < nsubsets; ++s) {
336 auto subset_res = meddler.run_blocked(n, block, starts, buffers.subset_totals[s], workspace.data());
337 auto subset_filt = filter.run(std::move(subset_res));
338 output.subset_totals.push_back(std::move(subset_filt.upper));
339 }
