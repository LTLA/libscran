108 void run(const tatami::Matrix<T, IDX>* mat, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
109 irlba::EigenThreadScope t(nthreads);
110 irlba::Irlba irb;
111 irb.set_number(rank);
112
113 if (mat->sparse()) {
114 Eigen::VectorXd center_v(mat->nrow()), scale_v(mat->nrow());
115 auto emat = create_custom_sparse_matrix(mat, center_v, scale_v, total_var);
116
117 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
118 if (scale) {
119 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
120 irb.run(scaled, pcs, rotation, variance_explained);
121 } else {
122 irb.run(centered, pcs, rotation, variance_explained);
123 }
124 } else {
125 auto emat = create_eigen_matrix_dense(mat, total_var);
126 irb.run(emat, pcs, rotation, variance_explained); // already centered and scaled, if relevant.
127 }
128
129 pca_utils::clean_up(mat->ncol(), pcs, variance_explained);
130 if (transpose) {
131 pcs.adjointInPlace();
132 }
133
134 return;
135 }
