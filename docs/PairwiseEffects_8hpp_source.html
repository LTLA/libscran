278 template<typename Data_, typename Index_, typename Group_, typename Block_, typename Stat_>
279 void run_blocked(
280 const tatami::Matrix<Data_, Index_>* p,
281 const Group_* group,
282 const Block_* block,
283 std::vector<std::vector<Stat_*> > means,
284 std::vector<std::vector<Stat_*> > detected,
285 Stat_* cohen,
286 Stat_* auc,
287 Stat_* lfc,
288 Stat_* delta_detected)
289 const {
290 if (block == NULL) {
291 run(p, group, fetch_first(means), fetch_first(detected), cohen, auc, lfc, delta_detected);
292 return;
293 }
294
295 int ngroups = means.size();
296 int nblocks = (ngroups ? means[0].size() : 0); // no blocks = no groups.
297
298 int ncombos = ngroups * nblocks;
299 std::vector<Stat_*> means2(ncombos), detected2(ncombos);
300 {
301 auto mIt = means2.begin(), dIt = detected2.begin();
302 for (int g = 0; g < ngroups; ++g) {
303 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
304 *mIt = means[g][b];
305 *dIt = detected[g][b];
306 }
307 }
308 }
309
310 differential_analysis::MatrixCalculator runner(nthreads, threshold);
311 Overlord overlord(auc);
312 auto state = runner.run_blocked(p, group, ngroups, block, nblocks, overlord);
313 process_simple_effects(p->nrow(), ngroups, nblocks, state, means2, detected2, cohen, lfc, delta_detected);
314 }

332 template<typename Stat_>
333 void process_simple_effects(
334 size_t ngenes, // using size_t consistently here, to avoid integer overflow bugs when computing products.
335 size_t ngroups,
336 size_t nblocks,
337 const differential_analysis::MatrixCalculator::State& state,
338 std::vector<Stat_*>& means,
339 std::vector<Stat_*>& detected,
340 Stat_* cohen,
341 Stat_* lfc,
342 Stat_* delta_detected)
343 const {
344 const auto& level_size = state.level_size;
345 size_t nlevels = level_size.size();
346
347 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
348 auto in_offset = nlevels * start;
349 const auto* tmp_means = state.means.data() + in_offset;
350 const auto* tmp_variances = state.variances.data() + in_offset;
351 const auto* tmp_detected = state.detected.data() + in_offset;
352
353 size_t squared = ngroups * ngroups;
354 size_t out_offset = start * squared;
355 for (size_t gene = start, end = start + length; gene < end; ++gene, out_offset += squared) {
356 for (size_t l = 0; l < nlevels; ++l) {
357 means[l][gene] = tmp_means[l];
358 detected[l][gene] = tmp_detected[l];
359 }
360
361 if (cohen != NULL) {
362 differential_analysis::compute_pairwise_cohens_d(tmp_means, tmp_variances, level_size, ngroups, nblocks, threshold, cohen + out_offset);
363 }
364
365 if (delta_detected != NULL) {
366 differential_analysis::compute_pairwise_simple_diff(tmp_detected, level_size, ngroups, nblocks, delta_detected + out_offset);
367 }
368
369 if (lfc != NULL) {
370 differential_analysis::compute_pairwise_simple_diff(tmp_means, level_size, ngroups, nblocks, lfc + out_offset);
371 }
372
373 tmp_means += nlevels;
374 tmp_variances += nlevels;
375 tmp_detected += nlevels;
376 }
377 }, ngenes, nthreads);
378 }
