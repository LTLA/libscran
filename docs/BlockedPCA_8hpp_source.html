18template<class Matrix, typename Block>
19struct BlockedEigenMatrix {
20 BlockedEigenMatrix(const Matrix* m, const Block* b, const Eigen::MatrixXd* mx) : mat(m), block(b), means(mx) {}
21
22 auto rows() const { return mat->rows(); }
23 auto cols() const { return mat->cols(); }
24
25 template<class Right>
26 void multiply(const Right& rhs, Eigen::VectorXd& output) const {
27 output.noalias() = (*mat) * rhs;
28 Eigen::MatrixXd sub = (*means) * rhs;
29 for (Eigen::Index i = 0; i < output.size(); ++i) {
30 output[i] -= sub(block[i], 0);
31 }
32 return;
33 }
34
35 template<class Right>
36 void adjoint_multiply(const Right& rhs, Eigen::VectorXd& output) const {
37 output.noalias() = mat->adjoint() * rhs;
38
39 Eigen::VectorXd aggr(means->rows());
40 aggr.setZero();
41 for (Eigen::Index i = 0; i < rhs.size(); ++i) {
42 aggr[block[i]] += rhs[i];
43 }
44
45 output.noalias() -= means->adjoint() * aggr;
46 return;
47 }
48
49 Eigen::MatrixXd realize() const {
50 Eigen::MatrixXd output(*mat);
51 for (Eigen::Index c = 0; c < output.cols(); ++c) {
52 for (Eigen::Index r = 0; r < output.rows(); ++r) {
53 output(r, c) -= (*means)(block[r], c);
54 }
55 }
56 return output;
57 }
58private:
59 const Matrix* mat;
60 const Block* block;
61 const Eigen::MatrixXd* means;
62};

88 template<typename T, typename IDX, typename Block>
89 void run(const tatami::Matrix<T, IDX>* mat, const Block* block, Eigen::MatrixXd& pcs, Eigen::VectorXd& variance_explained, double& total_var) {
90 irb.set_number(rank);
91
92 const size_t NC = mat->ncol();
93 const int nblocks = (NC ? *std::max_element(block, block + NC) + 1 : 1);
94 std::vector<int> block_size(nblocks);
95 for (size_t j = 0; j < NC; ++j) {
96 ++block_size[block[j]];
97 }
98
99 if (mat->sparse()) {
100 Eigen::MatrixXd center_m(nblocks, mat->nrow());
101 Eigen::VectorXd scale_v(mat->nrow());
102 auto emat = create_eigen_matrix_sparse(mat, center_m, scale_v, block, block_size, total_var);
103
104 BlockedEigenMatrix<decltype(emat), Block> centered(&emat, block, &center_m);
105 if (scale) {
106 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
107 auto result = irb.run(scaled);
108 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
109 } else {
110 auto result = irb.run(centered);
111 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
112 }
113 } else {
114 auto emat = create_eigen_matrix_dense(mat, block, block_size, total_var);
115 auto result = irb.run(emat); // already centered and scaled, if relevant.
116 pca_utils::clean_up(mat->ncol(), result.U, result.D, pcs, variance_explained);
117 }
118
119 return;
120 }

149 template<typename T, typename IDX, typename Block>
150 Eigen::SparseMatrix<double> create_eigen_matrix_sparse(const tatami::Matrix<T, IDX>* mat,
151 Eigen::MatrixXd& center_m,
152 Eigen::VectorXd& scale_v,
153 const Block* block,
154 const std::vector<int>& block_size,
155 double& total_var)
156 {
157 size_t NR = mat->nrow(), NC = mat->ncol();
158 total_var = 0;
159
160 Eigen::SparseMatrix<double> A(NC, NR); // transposed; we want genes in the columns.
161 std::vector<std::vector<double> > values;
162 std::vector<std::vector<int> > indices;
163
164 const size_t nblocks = block_size.size();
165 double * mbuffer = center_m.data();
166
167#ifdef SCRAN_LOGGER
168 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
169#endif
170
171 if (mat->prefer_rows()) {
172 std::vector<double> xbuffer(NC);
173 std::vector<int> ibuffer(NC);
174 std::vector<int> nnzeros(NR);
175 values.reserve(NR);
176 indices.reserve(NR);
177
178 for (size_t r = 0; r < NR; ++r, mbuffer += nblocks) {
179 auto range = mat->sparse_row(r, xbuffer.data(), ibuffer.data());
180
181 // Computing the block-wise means.
182 std::fill(mbuffer, mbuffer + nblocks, 0);
183 for (size_t i = 0; i < range.number; ++i) {
184 mbuffer[block[range.index[i]]] += range.value[i];
185 }
186 for (size_t b = 0; b < nblocks; ++b) {
187 mbuffer[b] /= block_size[b];
188 }
189
190 // Computing the variance from the sum of squared differences.
191 // This is technically not the correct variance estimate if
192 // we were to consider the blocks, but it's what the PCA sees.
193 double& proxyvar = scale_v[r];
194 proxyvar = 0;
195 {
196 auto block_copy = block_size;
197 for (size_t i = 0; i < range.number; ++i) {
198 Block curb = block[range.index[i]];
199 double diff = range.value[i] - mbuffer[curb];
200 proxyvar += diff * diff;
201 --block_copy[curb];
202 }
203
204 for (size_t b = 0; b < nblocks; ++b) {
205 proxyvar += mbuffer[b] * mbuffer[b] * block_copy[b];
206 }
207
208 proxyvar /= NC - 1;
209 }
210
211 nnzeros[r] = range.number;
212 values.emplace_back(range.value, range.value + range.number);
213 indices.emplace_back(range.index, range.index + range.number);
214 }
215
216 pca_utils::set_scale(scale, scale_v, total_var);
217 pca_utils::fill_sparse_matrix<true>(A, indices, values, nnzeros);
218 } else {
219 std::vector<double> xbuffer(NR);
220 std::vector<int> ibuffer(NR);
221 values.reserve(NC);
222 indices.reserve(NC);
223 std::vector<int> nnzeros(NR);
224
225 std::vector<std::vector<double> > tmp_means(nblocks, std::vector<double>(NR));
226 std::vector<std::vector<int> > tmp_nonzeros(nblocks, std::vector<int>(NR));
227
228 for (size_t c = 0; c < NC; ++c) {
229 auto range = mat->sparse_column(c, xbuffer.data(), ibuffer.data());
230 values.emplace_back(range.value, range.value + range.number);
231 indices.emplace_back(range.index, range.index + range.number);
232
233 // Collecting values for the means.
234 Block curb = block[c];
235 auto& cur_means = tmp_means[curb];
236 auto& cur_nonzeros = tmp_nonzeros[curb];
237 for (size_t i = 0; i < range.number; ++i) {
238 auto r = range.index[i];
239 cur_means[r] += range.value[i];
240 ++cur_nonzeros[r];
241 ++nnzeros[r];
242 }
243 }
244
245 // Finalizing the means.
246 for (size_t b = 0; b < nblocks; ++b) {
247 auto& cur_means = tmp_means[b];
248 for (size_t r = 0; r < cur_means.size(); ++r) {
249 cur_means[r] /= block_size[b];
250 center_m(b, r) = cur_means[r];
251 }
252 }
253
254 // Computing the pseudo-variances for each gene.
255 {
256 scale_v.setZero();
257
258 for (size_t c = 0; c < NC; ++c) {
259 const auto& cur_means = tmp_means[block[c]];
260 const auto& cur_vals = values[c];
261 const auto& cur_idx = indices[c];
262
263 for (size_t i = 0; i < cur_idx.size(); ++i) {
264 auto r = cur_idx[i];
265 double diff = cur_vals[i] - cur_means[r];
266 scale_v[r] += diff * diff;
267 }
268 }
269
270 for (size_t b = 0; b < nblocks; ++b) {
271 const auto& cur_means = tmp_means[b];
272 const auto& cur_nonzeros = tmp_nonzeros[b];
273 for (size_t r = 0; r < NR; ++r) {
274 scale_v[r] += cur_means[r] * cur_means[r] * (block_size[b] - cur_nonzeros[r]);
275 }
276 }
277
278 for (size_t r = 0; r < NR; ++r) {
279 scale_v[r] /= NC - 1;
280 }
281 }
282
283 pca_utils::set_scale(scale, scale_v, total_var);
284 pca_utils::fill_sparse_matrix<false>(A, indices, values, nnzeros);
285 }
286
287 A.makeCompressed();
288 return A;
289 }

292 template<typename T, typename IDX, typename Block>
293 Eigen::MatrixXd create_eigen_matrix_dense(
294 const tatami::Matrix<T, IDX>* mat,
295 const Block* block,
296 const std::vector<int>& block_size,
297 double& total_var)
298 {
299 size_t NR = mat->nrow(), NC = mat->ncol();
300 total_var = 0;
301
302 Eigen::MatrixXd output(NC, NR); // transposed.
303 std::vector<double> xbuffer(NC);
304 double* outIt = output.data();
305
306 int nblocks = block_size.size();
307 std::vector<double> mean_buffer(nblocks);
308
309#ifdef SCRAN_LOGGER
310 SCRAN_LOGGER("scran::RunPCA", "Preparing the input matrix");
311#endif
312
313 for (size_t r = 0; r < NR; ++r, outIt += NC) {
314 auto ptr = mat->row_copy(r, outIt);
315
316 // Compute the means of each block.
317 std::fill(mean_buffer.begin(), mean_buffer.end(), 0);
318 {
319 auto copy = outIt;
320 auto bcopy = block;
321 for (size_t c = 0; c < NC; ++c, ++copy, ++bcopy) {
322 mean_buffer[*bcopy] += *copy;
323 }
324 }
325 for (int b = 0; b < nblocks; ++b) {
326 if (block_size[b]) {
327 mean_buffer[b] /= block_size[b];
328 }
329 }
330
331 // We don't actually compute the blockwise variance, but instead
332 // the squared sum of deltas, which is what PCA actually sees.
333 double proxyvar = 0;
334 {
335 auto copy = outIt;
336 auto bcopy = block;
337 for (size_t c = 0; c < NC; ++c, ++copy, ++bcopy) {
338 *copy -= mean_buffer[*bcopy];
339 proxyvar += *copy * *copy;
340 }
341 proxyvar /= NC - 1;
342 }
343
344 pca_utils::apply_scale(scale, proxyvar, NC, outIt, total_var);
345 }
346
347 return output;
348 }
