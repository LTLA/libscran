368 for (size_t i = 0; i < range.number; ++i) {
369 current += range.value[i];
370 }
371 }
372
373 if (output.detected) {
374 auto& current = output.detected[c];
375 for (size_t i = 0; i < range.number; ++i) {
376 current += (range.value[i] != 0);
377 }
378 }
379
380 if (output.max_index || output.max_count) {
381 auto max_count = PerCellQcMetrics::pick_fill_value<Float>();
382 Integer max_index = 0;
383 for (size_t i = 0; i < range.number; ++i) {
384 if (max_count < range.value[i]) {
385 max_count = range.value[i];
386 max_index = range.index[i];
387 }
388 }
389
390 if (max_count <= 0 && range.number < NR) {
391 // Zero is the max.
392 max_count = 0;
393
394 // Finding the index of the first zero by tracking all
395 // indices with non-zero values. This isn't the fastest
396 // approach but it's simple and avoids assuming that
397 // indices are sorted. Hopefully we don't have to hit
398 // this section often.
399 if (output.max_index) {
400 for (size_t i = 0; i < range.number; ++i) {
401 if (range.value[i]) {
402 internal_is_nonzero[range.index[i]] = 1;
403 }
404 }
405 for (size_t r = 0; r < NR; ++r) {
406 if (internal_is_nonzero[r] == 0) {
407 max_index = r;
408 break;
409 }
410 }
411 for (size_t i = 0; i < range.number; ++i) { // setting back to zero.
412 internal_is_nonzero[range.index[i]] = 0;
413 }
414 }
415 }
416
417 if (output.max_index) {
418 output.max_index[c] = max_index;
419 }
420 if (output.max_count) {
421 output.max_count[c] = max_count;
422 }
423 }
424

426 for (size_t s = 0; s < nsubsets; ++s) {
427 const auto& sub = (*subsets_ptr)[s];
428
429 if (!output.subset_total.empty() && output.subset_total[s]) {
430 auto& current = output.subset_total[s][c];
431 for (size_t i = 0; i < range.number; ++i) {

672 template<class Matrix, typename Subset = const uint8_t*, typename Float, typename Integer>
673 void run(const Matrix* mat, const std::vector<Subset>& subsets, Buffers<Float, Integer> output) const {
674 if (!output.already_zeroed) {
675 size_t n = mat->ncol();
676 auto check_and_fill = [&](auto* ptr, auto value) -> void {
677 if (ptr) {
678 std::fill(ptr, ptr + n, value);
679 }
680 };
681
682 check_and_fill(output.total, static_cast<Float>(0));
683 check_and_fill(output.detected, static_cast<Integer>(0));
684 check_and_fill(output.max_count, pick_fill_value<Float>());
685 check_and_fill(output.max_index, static_cast<Integer>(0));
686
687 for (size_t s = 0; s < subsets.size(); ++s) {
688 if (!output.subset_total.empty()) {
689 check_and_fill(output.subset_total[s], static_cast<Float>(0));
690 }
691 if (!output.subset_detected.empty()) {
692 check_and_fill(output.subset_detected[s], static_cast<Integer>(0));
693 }
694 }
695 }
696
697 size_t nr = mat->nrow(), nc = mat->ncol();
698 Factory fact(nr, nc, &subsets, std::move(output));
699 tatami::apply<1>(mat, fact, num_threads);
700 return;
701 }
