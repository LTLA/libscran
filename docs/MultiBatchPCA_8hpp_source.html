29struct MultiBatchEigenMatrix {
30 MultiBatchEigenMatrix(const Matrix* m, const Eigen::VectorXd* w, const Eigen::VectorXd* mx) : mat(m), weights(w), means(mx) {}
31
32 auto rows() const { return mat->rows(); }
33 auto cols() const { return mat->cols(); }
34
35 template<class Right>
36 void multiply(const Right& rhs, Eigen::VectorXd& output) const {
37 if constexpr(irlba::has_multiply_method<Matrix>::value) {
38 output.noalias() = *mat * rhs;
39 } else {
40 mat->multiply(rhs, output);
41 }
42
43 double sub = means->dot(rhs);
44 for (Eigen::Index i = 0; i < output.size(); ++i) {
45 auto& val = output.coeffRef(i);
46 val -= sub;
47 val *= weights->coeff(i);
48 }
49 }
50
51 template<class Right>
52 void adjoint_multiply(const Right& rhs, Eigen::VectorXd& output) const {
53 Eigen::VectorXd combined = weights->cwiseProduct(rhs);
54
55 if constexpr(irlba::has_adjoint_multiply_method<Matrix>::value) {
56 output.noalias() = mat->adjoint() * combined;
57 } else {
58 mat->adjoint_multiply(combined, output);
59 }
60
61 double sum = combined.sum();
62 for (Eigen::Index i = 0; i < output.size(); ++i) {
63 output.coeffRef(i) -= means->coeff(i) * sum;
64 }
65 return;
66 }
67
68 Eigen::MatrixXd realize() const {
69 Eigen::MatrixXd output;
70 if constexpr(irlba::has_realize_method<Matrix>::value) {
71 output = mat->realize();
72 } else {
73 output = Eigen::MatrixXd(*mat);
74 }
75
76 for (Eigen::Index c = 0; c < output.cols(); ++c) {
77 for (Eigen::Index r = 0; r < output.rows(); ++r) {
78 auto& val = output.coeffRef(r, c);
79 val -= means->coeff(c);
80 val *= weights->coeff(r);
81 }
82 }
83 return output;
84 }
85private:
86 const Matrix* mat;
87 const Eigen::VectorXd* weights;
88 const Eigen::VectorXd* means;
89};

194 template<class Matrix>
195 void reapply(const Matrix& emat,
196 const Eigen::VectorXd& center_v,
197 const Eigen::VectorXd& scale_v,
198 Eigen::MatrixXd& pcs,
199 Eigen::MatrixXd& rotation,
200 Eigen::VectorXd& variance_explained)
201 const {
202 // Dividing by the scaling factor to mimic the division of 'emat'
203 // by the scaling factor (after centering).
204 auto rIt = rotation.data();
205 for (size_t i = 0, iend = rotation.cols(); i < iend; ++i) {
206 auto sIt = scale_v.data();
207 for (size_t j = 0, jend = rotation.rows(); j < jend; ++j, ++rIt, ++sIt) {
208 (*rIt) /= *sIt;
209 }
210 }
211
212 reapply(emat, center_v, pcs, rotation, variance_explained);
213 }

215 template<class Matrix>
216 void reapply(const Matrix& emat,
217 const Eigen::VectorXd& center_v,
218 Eigen::MatrixXd& pcs,
219 const Eigen::MatrixXd& rotation,
220 Eigen::VectorXd& variance_explained)
221 const {
222 if constexpr(irlba::has_multiply_method<Matrix>::value) {
223 pcs.noalias() = emat * rotation;
224 } else {
225 pcs.resize(emat.rows(), rotation.cols());
226 emat.multiply(rotation, pcs);
227 }
228
229 // Effective centering because I don't want to modify 'emat'.
230 auto pIt = pcs.data();
231 for (size_t i = 0, iend = pcs.cols(); i < iend; ++i) {
232 double meanval = center_v.dot(rotation.col(i));
233 for (size_t j = 0, jend = pcs.rows(); j < jend; ++j, ++pIt) {
234 *pIt -= meanval;
235 }
236 }
237
238 // Variance is a somewhat murky concept with weights, so we just square
239 // it and assume that only the relative value matters.
240 for (auto& d : variance_explained) {
241 d = d * d;
242 }
243 return;
244 }

247 template<typename T, typename IDX, typename Batch>
248 void run(const tatami::Matrix<T, IDX>* mat, const Batch* batch, Eigen::MatrixXd& pcs, Eigen::MatrixXd& rotation, Eigen::VectorXd& variance_explained, double& total_var) const {
249 const size_t NC = mat->ncol();
250 const auto& batch_size = block_sizes(NC, batch);
251 const size_t nbatchs = batch_size.size();
252
253 // Computing weights.
254 Eigen::VectorXd weights(NC);
255 for (size_t i = 0; i < NC; ++i) {
256 weights[i] = 1/std::sqrt(static_cast<double>(batch_size[batch[i]]));
257 }
258
259 Eigen::VectorXd center_v(mat->nrow());
260 Eigen::VectorXd scale_v(mat->nrow());
261
262 // Remember, we want to run the PCA on the modified matrix,
263 // but we want to apply the rotation vectors to the original matrix
264 // (after any centering/scaling but without the batch weights).
265 auto executor = [&](const auto& emat) -> void {
266 irlba::EigenThreadScope t(nthreads);
267 irlba::Irlba irb;
268 irb.set_number(rank);
269
270 MultiBatchEigenMatrix<typename std::remove_reference<decltype(emat)>::type> thing(&emat, &weights, &center_v);
271 if (scale) {
272 irb.run(irlba::Scaled<decltype(thing)>(&thing, &scale_v), pcs, rotation, variance_explained);
273 reapply(emat, center_v, scale_v, pcs, rotation, variance_explained);
274 } else {
275 irb.run(thing, pcs, rotation, variance_explained);
276 reapply(emat, center_v, pcs, rotation, variance_explained);
277 }
278 };
279
280 if (mat->sparse()) {
281 auto emat = create_custom_sparse_matrix(mat, center_v, scale_v, batch, batch_size, total_var);
282 executor(emat);
283 } else {
284 auto emat = create_eigen_matrix_dense(mat, center_v, scale_v, batch, batch_size, total_var);
285 executor(emat);
286 }
287
288 if (transpose) {
289 pcs.adjointInPlace();
290 }
291
292 return;
293 }

389 template<typename T, typename IDX, typename Batch>
390 pca_utils::CustomSparseMatrix create_custom_sparse_matrix(const tatami::Matrix<T, IDX>* mat,
391 Eigen::VectorXd& center_v,
392 Eigen::VectorXd& scale_v,
393 const Batch* batch,
394 const std::vector<int>& batch_size,
395 double& total_var)
396 const {
397
398 size_t NR = mat->nrow(), NC = mat->ncol();
399 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
400 auto& ptrs = extracted.ptrs;
401 auto& values = extracted.values;
402 auto& indices = extracted.indices;
403
404 // Computing grand means and variances with weights.
405 {
406 size_t nbatchs = batch_size.size();
407
408#ifndef SCRAN_CUSTOM_PARALLEL
409 #pragma omp parallel num_threads(nthreads)
410 {
411#else
412 SCRAN_CUSTOM_PARALLEL(NR, [&](size_t first, size_t last) -> void {
413#endif
414
415 std::vector<double> batch_means(nbatchs);
416 std::vector<int> batch_count(nbatchs);
417
418#ifndef SCRAN_CUSTOM_PARALLEL
419 #pragma omp for
420 for (size_t r = 0; r < NR; ++r) {
421#else
422 for (size_t r = first; r < last; ++r) {
423#endif
424
425 auto offset = ptrs[r];
426 size_t num_entries = ptrs[r+1] - offset;
427 auto value_ptr = values.data() + offset;
428 auto index_ptr = indices.data() + offset;
429
430 // Computing the grand mean across all batchs.
431 std::fill(batch_means.begin(), batch_means.end(), 0);
432 std::fill(batch_count.begin(), batch_count.end(), 0);
433 for (size_t i = 0; i < num_entries; ++i) {
434 auto b = batch[index_ptr[i]];
435 batch_means[b] += value_ptr[i];
436 ++batch_count[b];
437 }
438
439 double& grand_mean = center_v[r];
440 grand_mean = 0;
441 for (size_t b = 0; b < nbatchs; ++b) {
442 auto bsize = batch_size[b];
443 if (bsize) {
444 grand_mean += batch_means[b] / bsize;
445 }
446 }
447 grand_mean /= nbatchs;
448
449 // Computing pseudo-variances where each batch's contribution
450 // is weighted inversely proportional to its size. This aims to
451 // match up with the variances used in the PCA but not the
452 // variances of the output components (where weightings are not used).
453 double& proxyvar = scale_v[r];
454 proxyvar = 0;
455 for (size_t b = 0; b < nbatchs; ++b) {
456 double zero_sum = (batch_size[b] - batch_count[b]) * grand_mean * grand_mean;
457 proxyvar += zero_sum / batch_size[b];
458 }
459
460 for (size_t i = 0; i < num_entries; ++i) {
461 double diff = value_ptr[i] - grand_mean;
462 auto bsize = batch_size[batch[index_ptr[i]]];
463 if (bsize) {
464 proxyvar += diff * diff / bsize;
465 }
466 }
467
468#ifndef SCRAN_CUSTOM_PARALLEL
469 }
470 }
471#else
472 }
473 }, nthreads);
474#endif
475
476 total_var = pca_utils::process_scale_vector(scale, scale_v);
477 }
478
479 // Actually filling the sparse matrix. Note that this is transposed
480 // because we want genes in the columns.
481 pca_utils::CustomSparseMatrix A(NC, NR, nthreads);
482#ifdef TEST_SCRAN_CUSTOM_SPARSE_MATRIX
483 if (use_eigen) {
484 A.use_eigen();
485 }
486#endif
487 A.fill_direct(std::move(values), std::move(indices), std::move(ptrs));
488
489 return A;
490 }

493 template<typename T, typename IDX, typename Batch>
494 Eigen::MatrixXd create_eigen_matrix_dense(
495 const tatami::Matrix<T, IDX>* mat,
496 Eigen::VectorXd& center_v,
497 Eigen::VectorXd& scale_v,
498 const Batch* batch,
499 const std::vector<int>& batch_size,
500 double& total_var)
501 const {
502
503 // Extract a column-major matrix with genes in columns.
504 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads);
505
506 // Looping across genes (i.e., columns) to fill center_v and scale_v.
507 size_t NC = emat.cols(), NR = emat.rows();
508 size_t nbatchs = batch_size.size();
509
510#ifndef SCRAN_CUSTOM_PARALLEL
511 #pragma omp parallel num_threads(nthreads)
512 {
513 std::vector<double> mean_buffer(nbatchs);
514 #pragma omp for
515 for (size_t c = 0; c < NC; ++c) {
516#else
517 SCRAN_CUSTOM_PARALLEL(NC, [&](size_t first, size_t last) -> void {
518 std::vector<double> mean_buffer(nbatchs);
519 for (size_t c = first; c < last; ++c) {
520#endif
521
522 auto ptr = emat.data() + c * NR;
523 std::fill(mean_buffer.begin(), mean_buffer.end(), 0.0);
524 for (size_t r = 0; r < NR; ++r) {
525 mean_buffer[batch[r]] += ptr[r];
526 }
527
528 double& grand_mean = center_v[c];
529 grand_mean = 0;
530 int non_empty_batches = 0;
531 for (size_t b = 0; b < nbatchs; ++b) {
532 const auto& bsize = batch_size[b];
533 if (bsize) {
534 grand_mean += mean_buffer[b] / bsize;
535 ++non_empty_batches;
536 }
537 }
538 grand_mean /= non_empty_batches;
539
540 // We don't actually compute the batchwise variance, but instead
541 // the weighted sum of squared deltas, which is what PCA actually sees.
542 double& proxyvar = scale_v[c];
543 proxyvar = 0;
544 for (size_t r = 0; r < NR; ++r) {
545 const auto& bsize = batch_size[batch[r]];
546 if (bsize) {
547 double diff = ptr[r] - grand_mean;
548 proxyvar += diff * diff / bsize;
549 }
550 }
551
552#ifndef SCRAN_CUSTOM_PARALLEL
553 }
554 }
555#else
556 }
557 }, nthreads);
558#endif
559
560 total_var = pca_utils::process_scale_vector(scale, scale_v);
561 return emat;
562 }
