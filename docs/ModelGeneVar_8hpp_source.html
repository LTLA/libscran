106 void run_blocked(const MAT* mat, const B* block, std::vector<Stat*> means, std::vector<Stat*> variances, std::vector<Stat*> fitted, std::vector<Stat*> residuals) {
107 size_t NR = mat->nrow(), NC = mat->ncol();
108 std::vector<int> block_size(means.size());
109
110#ifdef SCRAN_LOGGER
111 SCRAN_LOGGER("scran::ModelGeneVar", "Estimating mean and variance of each gene");
112#endif
113
114 if (block) {
115 auto copy = block;
116 for (size_t j = 0; j < NC; ++j, ++copy) {
117 ++block_size[*copy];
118 }
119 feature_selection::BlockedVarianceFactory<true, Stat, B, decltype(block_size)> fact(NR, NC, means, variances, block, &block_size);
120 tatami::apply<0>(mat, fact);
121 } else {
122 block_size[0] = NC;
123 feature_selection::BlockedVarianceFactory<false, Stat, B, decltype(block_size)> fact(NR, NC, means, variances, block, &block_size);
124 tatami::apply<0>(mat, fact);
125 }
126
127 // Applying the trend fit to each block.
128#ifdef SCRAN_LOGGER
129 SCRAN_LOGGER("scran::ModelGeneVar", "Fitting a mean-variance trend across genes");
130#endif
131
132 for (size_t b = 0; b < block_size.size(); ++b) {
133 if (block_size[b] >= 2) {
134 fit.run(NR, means[b], variances[b], fitted[b], residuals[b]);
135 } else {
136 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
137 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
138 }
139 }
140
141 return;
142 }

218 Results run_blocked(const MAT* mat, const B* block) {
219 int nblocks = 1;
220 if (block) {
221 if (mat->ncol()) {
222 nblocks = *std::max_element(block, block + mat->ncol()) + 1;
223 }
224 }
225

227 std::vector<double*> mean_ptr, var_ptr, fit_ptr, resid_ptr;
228 for (int b = 0; b < nblocks; ++b) {
229 mean_ptr.push_back(output.means[b].data());
230 var_ptr.push_back(output.variances[b].data());
231 fit_ptr.push_back(output.fitted[b].data());
232 resid_ptr.push_back(output.residuals[b].data());
233 }
