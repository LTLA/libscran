231 template<class Function, typename D, typename PPTR, typename X>
232 Thresholds run_internal(const Function& fun, size_t ncells, const D* detected, std::vector<PPTR> subset_totals,
233 X* filter_by_detected, std::vector<X*> filter_by_subset_totals, X* overall_filter)
234 const {
235 Thresholds output;
236 std::vector<double> buffer(ncells);
237
238 // Filtering to remove outliers on the log-detected number.
239 {
240 IsOutlier outliers;
241 outliers.set_nmads(detected_nmads).set_lower(true).set_upper(false).set_log(true);
242 if (min_detected_drop > 0) {
243 if (min_detected_drop >= 1) {
244 throw std::runtime_error("minimum drop in the detected features should lie in [0, 1)");
245 }
246 outliers.set_min_diff(-std::log(1 - min_detected_drop));
247 }
248
249 auto res = fun(outliers, ncells, detected, filter_by_detected, buffer);
250 output.detected = res.lower;
251 for (size_t i = 0; i < ncells; ++i) {
252 overall_filter[i] |= filter_by_detected[i];
253 }
254 }
255
256 // Filtering to remove outliers on the subset totals.
257 {
258 size_t nsubsets = subset_totals.size();
259 if (filter_by_subset_totals.size() != nsubsets) {
260 throw std::runtime_error("mismatching number of input/outputs for subset filters");
261 }
262
263 IsOutlier outliers;
264 outliers.set_nmads(subset_nmads).set_upper(true).set_lower(false).set_log(true);
265 output.subset_totals.resize(nsubsets);
266
267 for (size_t s = 0; s < subset_totals.size(); ++s) {
268 auto dump = filter_by_subset_totals[s];
269 auto res = fun(outliers, ncells, subset_totals[s], dump, buffer);
270 output.subset_totals[s] = res.upper;
271
272 for (size_t i = 0; i < ncells; ++i) {
273 overall_filter[i] |= dump[i];
274 }
275 }
276 }
277
278 return output;
279 }
