125 template<typename Index, typename Float, typename Cluster>
126 static std::tuple<Cluster, Float, Float> assign(const std::vector<std::pair<Index, Float> >& neighbors, const Cluster* clusters, Cluster nclusters, int* buffer) {
127 std::fill(buffer, buffer + nclusters, 0);
128 for (const auto& n : neighbors) {
129 ++(buffer[clusters[n.first]]);
130 }
131
132 Cluster best = 0, second = 0;
133 bool no_second = true;
134 for (Cluster c = 1; c < nclusters; ++c) {
135 if (buffer[best] < buffer[c]) {
136 second = best;
137 best = c;
138 no_second = false;
139 } else if (no_second || buffer[second] < buffer[c]) {
140 second = c;
141 no_second = false;
142 }
143 }
144
145 Float best_prop = (nclusters ? static_cast<Float>(buffer[best]) / neighbors.size() : std::numeric_limits<Float>::quiet_NaN());
146 Float second_prop = (nclusters > 1 ? static_cast<Float>(buffer[second]) / neighbors.size() : std::numeric_limits<Float>::quiet_NaN());
147 return std::make_tuple(best, best_prop, second_prop);
148 }
