220 template<typename Data_, typename Index_, typename Block_>
221 void run_sparse_simple(
222 const tatami::Matrix<Data_, Index_>* mat,
223 const Block_* block,
224 const pca_utils::BlockingDetails<true> block_details,
225 const irlba::Irlba& irb,
226 Eigen::MatrixXd& pcs,
227 Eigen::MatrixXd& rotation,
228 Eigen::VectorXd& variance_explained,
229 Eigen::VectorXd& center_v,
230 Eigen::VectorXd& scale_v,
231 double& total_var)
232 const {
233 auto extracted = pca_utils::extract_sparse_for_pca(mat, nthreads); // row-major extraction.
234 pca_utils::SparseMatrix emat(mat->ncol(), mat->nrow(), std::move(extracted.values), std::move(extracted.indices), std::move(extracted.ptrs), nthreads); // CSC with genes in columns.
235
236 size_t ngenes = emat.cols();
237 center_v.resize(ngenes);
238 scale_v.resize(ngenes);
239
240 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
241 const auto& values = emat.get_values();
242 const auto& indices = emat.get_indices();
243 const auto& ptrs = emat.get_pointers();
244
245 const auto& block_size = block_details.block_size;
246 size_t nblocks = block_size.size();
247 std::vector<int> block_count(nblocks);
248 const auto& block_weight = block_details.per_element_weight;
249
250 for (size_t r = start, end = start + length; r < end; ++r) {
251 auto offset = ptrs[r];
252 size_t num_entries = ptrs[r+1] - offset;
253 auto value_ptr = values.data() + offset;
254 auto index_ptr = indices.data() + offset;
255
256 std::fill(block_count.begin(), block_count.end(), 0);
257
258 // Computing the grand mean across all blocks.
259 double& grand_mean = center_v[r];
260 grand_mean = 0;
261 for (size_t i = 0; i < num_entries; ++i) {
262 auto b = block[index_ptr[i]];
263 grand_mean += value_ptr[i] * block_weight[b];
264 ++(block_count[b]);
265 }
266 grand_mean /= block_details.total_block_weight;
267
268 // Computing pseudo-variances where each block's contribution
269 // is weighted inversely proportional to its size. This aims to
270 // match up with the variances used in the PCA but not the
271 // variances of the output components (where weightings are not used).
272 double& proxyvar = scale_v[r];
273 proxyvar = 0;
274 for (size_t b = 0; b < nblocks; ++b) {
275 double zero_sum = (block_size[b] - block_count[b]) * grand_mean * grand_mean;
276 proxyvar += zero_sum * block_weight[b];
277 }
278
279 for (size_t i = 0; i < num_entries; ++i) {
280 double diff = value_ptr[i] - grand_mean;
281 proxyvar += diff * diff * block_weight[block[index_ptr[i]]];
282 }
283
284 proxyvar /= emat.rows() - 1;
285 }
286 }, ngenes, nthreads);
287
288 total_var = pca_utils::process_scale_vector(scale, scale_v);
289
290 // Now actually performing the PCA.
291 irlba::Centered<decltype(emat)> centered(&emat, &center_v);
292 if (scale) {
293 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);
294 pca_utils::SampleScaledWrapper<decltype(scaled)> weighted(&scaled, &(block_details.expanded_weights));
295 irb.run(weighted, pcs, rotation, variance_explained);
296 } else {
297 pca_utils::SampleScaledWrapper<decltype(centered)> weighted(&centered, &(block_details.expanded_weights));
298 irb.run(weighted, pcs, rotation, variance_explained);
299 }
300
301 // This transposes 'pcs' to be a NDIM * NCELLS matrix.
302 pca_utils::project_sparse_matrix(emat, pcs, rotation, scale, scale_v, nthreads);
303
304 pca_utils::clean_up_projected<true>(pcs, variance_explained);
305 if (!transpose) {
306 pcs.adjointInPlace();
307 }
308 }

310 template<typename Data_, typename Index_, typename Block_>
311 void run_dense_simple(
312 const tatami::Matrix<Data_, Index_>* mat,
313 const Block_* block,
314 const pca_utils::BlockingDetails<true>& block_details,
315 const irlba::Irlba& irb,
316 Eigen::MatrixXd& pcs,
317 Eigen::MatrixXd& rotation,
318 Eigen::VectorXd& variance_explained,
319 Eigen::VectorXd& center_v,
320 Eigen::VectorXd& scale_v,
321 double& total_var)
322 const {
323 auto emat = pca_utils::extract_dense_for_pca(mat, nthreads); // row-major extraction.
324
325 size_t ngenes = emat.cols();
326 center_v.resize(ngenes);
327 scale_v.resize(ngenes);
328
329 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
330 size_t nblocks = block_details.num_blocks();
331 std::vector<double> mean_buffer(nblocks);
332 const auto& block_weight = block_details.per_element_weight;
333 size_t ncells = emat.rows();
334
335 for (size_t c = start, end = start + length; c < end; ++c) {
336 auto ptr = emat.data() + c * ncells;
337
338 double& grand_mean = center_v[c];
339 grand_mean = 0;
340 for (size_t r = 0; r < ncells; ++r) {
341 grand_mean += ptr[r] * block_weight[block[r]];
342 }
343 grand_mean /= block_details.total_block_weight;
344
345 // We don't actually compute the batchwise variance, but instead
346 // the weighted sum of squared deltas, which is what PCA actually sees.
347 double& proxyvar = scale_v[c];
348 proxyvar = 0;
349 for (size_t r = 0; r < ncells; ++r) {
350 double diff = ptr[r] - grand_mean;
351 proxyvar += diff * diff * block_weight[block[r]];
352 }
353
354 proxyvar /= emat.rows() - 1;
355 }
356 }, ngenes, nthreads);
357
358 total_var = pca_utils::process_scale_vector(scale, scale_v);
359
360 // Applying the centering and scaling now so we can do the PCA with fewer wrappers.
361 pca_utils::apply_center_and_scale_to_dense_matrix(emat, center_v, scale, scale_v, nthreads);
362
363 pca_utils::SampleScaledWrapper<decltype(emat)> weighted(&emat, &(block_details.expanded_weights));
364 irb.run(weighted, pcs, rotation, variance_explained);
365
366 pcs.noalias() = emat * rotation;
367 pca_utils::clean_up_projected<false>(pcs, variance_explained);
368 if (transpose) {
369 pcs.adjointInPlace();
370 }
371 }

375 void run_residuals_internal(
376 const Matrix_& emat,
377 const Block_* block,

395 irb.run(weighted, pcs, rotation, variance_explained);
396 }
397
398 } else {
399 if (scale) {
400 irlba::Scaled<decltype(centered)> scaled(&centered, &scale_v);

452 void run_dense_residuals(
453 const tatami::Matrix<Data_, Index_>* mat,
454 const Block_* block,
455 const pca_utils::BlockingDetails<weight_>& block_details,

479 irb,

498 template<typename Data_, typename Index_, typename Block_>
499 void run_internal(
500 const tatami::Matrix<Data_, Index_>* mat,
501 const Block_* block,
502 Eigen::MatrixXd& pcs,
503 Eigen::MatrixXd& rotation,
504 Eigen::VectorXd& variance_explained,
505 Eigen::MatrixXd& center_m,
506 Eigen::VectorXd& scale_v,
507 double& total_var)
508 const {
509 irlba::EigenThreadScope t(nthreads);
510 irlba::Irlba irb;
511 irb.set_number(rank);
512
513 if (use_residuals) {
514 if (block_weight_policy == WeightPolicy::NONE) {
515 auto bdetails = pca_utils::compute_blocking_details(mat->ncol(), block);
516 if (mat->sparse()) {
517 run_sparse_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
518 } else {
519 run_dense_residuals<false>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
520 }
521
522 } else {
523 auto bdetails = pca_utils::compute_blocking_details(mat->ncol(), block, block_weight_policy, variable_block_weight_parameters);
524 if (mat->sparse()) {
525 run_sparse_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
526 } else {
527 run_dense_residuals<true>(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_m, scale_v, total_var);
528 }
529 }
530
531 } else {
532 if (block_weight_policy == WeightPolicy::NONE) {
533 throw std::runtime_error("block weight policy cannot be NONE when 'use_residuals = true', use SimplePca instead");
534 }
535
536 auto bdetails = pca_utils::compute_blocking_details(mat->ncol(), block, block_weight_policy, variable_block_weight_parameters);
537
538 Eigen::VectorXd center_v;
539 if (mat->sparse()) {
540 run_sparse_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_v, scale_v, total_var);
541 } else {
542 run_dense_simple(mat, block, bdetails, irb, pcs, rotation, variance_explained, center_v, scale_v, total_var);
543 }
544
545 if (return_center) {
546 center_m.resize(1, center_v.size());
547 center_m.row(0) = center_v;
548 }
549 }
550 }

628 template<typename T, typename IDX, typename Batch>
629 Results run(const tatami::Matrix<T, IDX>* mat, const Batch* batch) const {
630 Results output;
631 Eigen::MatrixXd rotation, center_m;
632 Eigen::VectorXd scale_v;
633
634 run_internal(mat, batch, output.pcs, rotation, output.variance_explained, center_m, scale_v, output.total_variance);
635
636 // Shifting them if we want to keep them.
637 if (return_rotation) {
638 output.rotation = std::move(rotation);
639 }
640 if (return_center) {
641 output.center = center_m.adjoint();
642 }
643 if (return_scale) {
644 output.scale = std::move(scale_v);
645 }
646
647 return output;
648 }
