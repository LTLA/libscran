18struct SparseComponents {
19 std::vector<size_t> ptrs;
20 std::vector<double> values;
21 std::vector<int> indices;
22};

24/*
25 * We use a two-pass philosophy to create a CSR matrix, which is a little
26 * slower but reduces memory usage. Otherwise, we'd either have to have to
27 * store two realized copies of a double-precision sparse matrix, if we
28 * extracted each row/column into its own vector, before consolidating into a
29 * single CSR matrix; or we'd have to suffer repeated reallocations or
30 * overallocations to accommodate an unknown total number of non-zeros.
31*/
32
33namespace extract_for_pca_internal {

35template<typename T, typename IDX>
36SparseComponents sparse_by_row(const tatami::Matrix<T, IDX>* mat, int nthreads) {
37 SparseComponents output;
38 auto NR = mat->nrow(), NC = mat->ncol();
39 auto& ptrs = output.ptrs;
40 ptrs.resize(NR + 1);
41
42 /*** First round, to fetch the number of zeros in each row. ***/
43 tatami::parallelize([&](size_t, IDX start, IDX length) -> void {
44 tatami::Options opt;
45 opt.sparse_extract_value = false;
46 opt.sparse_extract_index = false;
47 auto ext = tatami::consecutive_extractor<true, true>(mat, start, length, opt);
48
49 for (IDX r = start, end = start + length; r < end; ++r) {
50 auto range = ext->fetch(r, NULL, NULL);
51 ptrs[r + 1] = range.number;
52 }
53 }, NR, nthreads);
54
55 /*** Second round, to populate the vectors. ***/
56 for (size_t r = 0; r < NR; ++r) {
57 ptrs[r + 1] += ptrs[r];
58 }
59 output.values.resize(ptrs.back());
60 output.indices.resize(ptrs.back());
61
62 tatami::parallelize([&](size_t, IDX start, IDX length) -> void {
63 auto ext = tatami::consecutive_extractor<true, true>(mat, start, length);
64 for (IDX r = start, end = start + length; r < end; ++r) {
65 auto offset = ptrs[r];
66 ext->fetch_copy(r, output.values.data() + offset, output.indices.data() + offset);
67 }
68 }, NR, nthreads);
69
70 return output;
71}

73template<typename T, typename IDX>
74SparseComponents sparse_by_column(const tatami::Matrix<T, IDX>* mat, int nthreads) {
75 auto NR = mat->nrow(), NC = mat->ncol();
76
77 /*** First round, to fetch the number of zeros in each row. ***/
78 std::vector<std::vector<size_t> > threaded_nonzeros_per_row(nthreads);
79 for (auto& nonzeros : threaded_nonzeros_per_row) {
80 nonzeros.resize(NR);
81 }
82
83 tatami::parallelize([&](size_t t, IDX start, IDX length) -> void {
84 tatami::Options opt;
85 opt.sparse_extract_value = false;
86 opt.sparse_ordered_index = false;
87 auto ext = tatami::consecutive_extractor<false, true>(mat, start, length, opt);
88
89 std::vector<IDX> ibuffer(NR);
90 auto& nonzeros = threaded_nonzeros_per_row[t];
91 for (IDX c = start, end = start + length; c < end; ++c) {
92 auto range = ext->fetch(c, NULL, ibuffer.data());
93 for (IDX j = 0; j < range.number; ++j) {
94 ++nonzeros[range.index[j]];
95 }
96 }
97 }, NC, nthreads);
98
99 // There had better be at least one thread!
100 std::vector<size_t> nonzeros_per_row = std::move(threaded_nonzeros_per_row[0]);
101 for (int t = 1; t < nthreads; ++t) {
102 auto it = nonzeros_per_row.begin();
103 for (auto x : threaded_nonzeros_per_row[t]) {
104 *it += x;
105 ++it;
106 }
107 }
108
109 /*** Second round, to populate the vectors. ***/
110 SparseComponents output;
111 output.ptrs.resize(NR + 1);
112 size_t total_nzeros = 0;
113 for (size_t r = 0; r < NR; ++r) {
114 total_nzeros += nonzeros_per_row[r];
115 output.ptrs[r + 1] = total_nzeros;
116 }
117 output.values.resize(total_nzeros);
118 output.indices.resize(total_nzeros);
119
120 // Splitting by row this time, as it can't otherwise be safely parallelized.
121 auto ptr_copy = output.ptrs;
122 tatami::parallelize([&](size_t t, IDX start, IDX length) -> void {
123 tatami::Options opt;
124 opt.sparse_ordered_index = false;
125 auto ext = tatami::consecutive_extractor<false, true>(mat, 0, NC, start, length, opt);
126
127 std::vector<T> vbuffer(length);
128 std::vector<IDX> ibuffer(length);
129
130 for (size_t c = 0; c < NC; ++c) {
131 auto range = ext->fetch(c, vbuffer.data(), ibuffer.data());
132 for (size_t i = 0; i < range.number; ++i) {
133 auto r = range.index[i];
134 auto& offset = ptr_copy[r];
135 output.values[offset] = range.value[i];
136 output.indices[offset] = c;
137 ++offset;
138 }
139 }
140 }, NR, nthreads);
141
142 return output;
143}

161template<typename T, typename IDX>
162Eigen::MatrixXd dense_by_row(const tatami::Matrix<T, IDX>* mat, int nthreads) {
163 auto NR = mat->nrow(), NC = mat->ncol();
164 Eigen::MatrixXd output(NC, NR); // transposed, we want our genes in the columns.
165 auto ptr = output.data();
166
167 tatami::parallelize([&](size_t, IDX start, IDX length) -> void {
168 auto ext = tatami::consecutive_extractor<true, false>(mat, start, length);
169 for (IDX r = start, end = start + length; r < end; ++r) {
170 ext->fetch_copy(r, ptr + r * NC);
171 }
172 }, NR, nthreads);
173
174 return output;
175}

177template<typename T, typename IDX>
178Eigen::MatrixXd dense_by_column(const tatami::Matrix<T, IDX>* mat, int nthreads) {
179 auto NR = mat->nrow(), NC = mat->ncol();
180 Eigen::MatrixXd output(NC, NR); // transposed, we want our genes in the columns.
181
182 tatami::parallelize([&](size_t, IDX start, IDX length) -> void {
183 auto ext = tatami::consecutive_extractor<false, false>(mat, 0, NC, start, length);
184 std::vector<T> buffer(length);
185
186 for (size_t c = 0; c < NC; ++c) {
187 auto ptr = ext->fetch(c, buffer.data());
188 for (size_t r = 0; r < length; ++r) {
189 output(c, r + start) = ptr[r];
190 }
191 }
192 }, NR, nthreads);
193
194 return output;
195}
