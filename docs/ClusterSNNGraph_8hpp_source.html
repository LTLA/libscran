272 igraph_vector_t membership, modularity;

278
279 // I just can't be bothered to do anything fancier here, so this is what we've got.
280 igraph_rng_seed(igraph_rng_default(), seed);
281
282#ifdef SCRAN_LOGGER
283 SCRAN_LOGGER("scran::ClusterSNNGraph", "Performing multilevel community detection");
284#endif
285 MultiLevelResult output;
286 output.status = igraph_community_multilevel(&graph_info.graph, &graph_info.weights, resolution, &membership, &memberships, &modularity);
287
288 if (!output.status) {
289 output.max = igraph_vector_which_max(&modularity);
290
291 size_t nmods = igraph_vector_size(&modularity);
292 output.modularity.resize(nmods);
293 for (size_t i = 0; i < nmods; ++i) {
294 output.modularity[i] = VECTOR(modularity)[i];
295 }
296
297 size_t ncells = igraph_vcount(&graph_info.graph);
298 size_t nlevels = igraph_matrix_nrow(&memberships);
299 output.membership.resize(nlevels);
300
301 for (size_t i = 0; i < nlevels; ++i) {
302 auto& current = output.membership[i];
303 current.resize(ncells);
304 for (size_t j = 0; j < ncells; ++j) {
305 current[j] = MATRIX(memberships, i, j);
306 }
307 }
308 }
309
310 igraph_vector_destroy(&modularity);
311 igraph_vector_destroy(&membership);
312 igraph_matrix_destroy(&memberships);
