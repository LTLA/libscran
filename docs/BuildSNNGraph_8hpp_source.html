178 std::deque<WeightedEdge> run(const std::vector<std::vector<int> >& indices) const {
179 size_t ncells = indices.size();
180
181 // Not parallel-frendly, so we don't construct this with the neighbor search
182 std::vector<std::vector<std::pair<int, size_t> > > hosts(ncells);
183 for (size_t i = 0; i < ncells; ++i) {
184 hosts[i].push_back(std::make_pair(0, i)); // each point is its own 0-th nearest neighbor
185 const auto& current = indices[i];
186 int counter = 1;
187 for (auto x : current) {
188 hosts[x].push_back(std::make_pair(counter, i));
189 ++counter;
190 }
191 }
192
193 // Constructing the shared neighbor graph.
194#ifdef SCRAN_LOGGER
195 SCRAN_LOGGER("scran::BuildSNNGraph", "Computing shared neighbor weights");
196#endif
197 std::vector<size_t> current_added;
198 current_added.reserve(ncells);
199 std::vector<int> current_score(ncells);
200 std::deque<WeightedEdge> store;
201
202 for (size_t j = 0; j < ncells; ++j) {
203 const auto& current_neighbors = indices[j];
204
205 for (int i = 0; i <= current_neighbors.size(); ++i) {
206 // First iteration treats 'j' as the zero-th neighbor.
207 // Remaining iterations go through the neighbors of 'j'.
208 const int cur_neighbor = (i==0 ? j : current_neighbors[i-1]);
209
210 // Going through all observations 'h' for which 'cur_neighbor'
211 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
212 // neighbor of both 'h' and 'j'.
213 for (auto& h : hosts[cur_neighbor]) {
214 const auto& othernode = h.second;
215
216 if (othernode < j) { // avoid duplicates from symmetry in the SNN calculations.
217 int& existing_other = current_score[othernode];
218 if (weight_scheme == RANKED) {
219 // Recording the lowest combined rank per neighbor.
220 int currank = h.first + i;
221 if (existing_other == 0) {
222 existing_other = currank;
223 current_added.push_back(othernode);
224 } else if (existing_other > currank) {
225 existing_other = currank;
226 }
227 } else {
228 // Recording the number of shared neighbors.
229 if (existing_other==0) {
230 current_added.push_back(othernode);
231 }
232 ++existing_other;
233 }
234 }
235 }
236 }
237
238 // Converting to edges.
239 for (auto othernode : current_added) {
240 int& otherscore=current_score[othernode];
241 double finalscore;
242 if (weight_scheme == RANKED) {
243 finalscore = static_cast<double>(current_neighbors.size()) - 0.5 * static_cast<double>(otherscore);
244 } else {
245 finalscore = otherscore;
246 if (weight_scheme == JACCARD) {
247 finalscore = finalscore / (2 * (current_neighbors.size() + 1) - finalscore);
248 }
249 }
250
251 // Ensuring that an edge with a positive weight is always reported.
252 store.push_back(WeightedEdge(j, othernode, std::max(finalscore, 1e-6)));
253
254 // Resetting all those added to zero.
255 otherscore=0;
256 }
257 current_added.clear();
258 }
259
260 return store;
261 }
