373 template<class MAT, typename G, typename B, typename Stat>
374 void run_blocked_internal(const MAT* p, const G* group, const B* block, int ngroups, int nblocks,
375 std::vector<std::vector<Stat*> >& means,
376 std::vector<std::vector<Stat*> >& detected,
377 std::vector<std::vector<Stat*> >& cohen,
378 std::vector<std::vector<Stat*> >& auc)
379 {
380 int ncombos = ngroups * nblocks;
381 std::vector<int> combos(p->ncol());
382 std::vector<int> combo_size(ncombos);
383
384 for (size_t i = 0; i < combos.size(); ++i) {
385 combos[i] = group[i] * nblocks + block[i];
386 ++(combo_size[combos[i]]);
387 }
388
389 std::vector<Stat*> means2(ncombos), detected2(ncombos);
390 auto mIt = means2.begin(), dIt = detected2.begin();
391 for (int g = 0; g < ngroups; ++g) {
392 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
393 *mIt = means[g][b];
394 *dIt = detected[g][b];
395 }
396 }
397
398 core(p, combos.data(), combo_size, group, ngroups, block, nblocks, means2, detected2, cohen, auc);
399 }

401 template<class MAT, typename L, class Ls, typename G, typename B, typename Stat>
402 void core(const MAT* p,
403 const L* level, const Ls& level_size,
404 const G* group, int ngroups,
405 const B* block, int nblocks,
406 std::vector<Stat*>& means,
407 std::vector<Stat*>& detected,
408 std::vector<std::vector<Stat*> >& cohen,
409 std::vector<std::vector<Stat*> >& auc)
410 {
411 const bool do_cd = !cohen.empty();
412 std::vector<Stat> cohens_d(do_cd ? p->nrow() * ngroups * ngroups : 0);
413 Stat* cohens_ptr = do_cd ? cohens_d.data() : NULL;
414
415 const bool do_wilcox = !auc.empty();
416 std::vector<Stat> wilcox_auc(do_wilcox ? p->nrow() * ngroups * ngroups : 0);
417
418#ifdef SCRAN_LOGGER
419 SCRAN_LOGGER("scran::ScoreMarkers", "Performing pairwise comparisons between groups of cells");
420#endif
421 if (!do_wilcox) {
422 std::vector<Stat*> effects{cohens_ptr};
423 differential_analysis::BidimensionalFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, ngroups, nblocks, threshold);
424 tatami::apply<0>(p, fact);
425 } else {
426 std::vector<Stat*> effects{cohens_ptr, wilcox_auc.data()};
427
428 // Need to remake this, as there's no guarantee that 'blocks' exists.
429 std::vector<B> tmp_blocks;
430 if (!block) {
431 tmp_blocks.resize(p->ncol());
432 block = tmp_blocks.data();
433 }
434
435 differential_analysis::PerRowFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, group, ngroups, block, nblocks, threshold);
436 tatami::apply<0>(p, fact);
437 }
438
439 if (do_cd) {
440#ifdef SCRAN_LOGGER
441 SCRAN_LOGGER("scran::ScoreMarkers", "Summarizing Cohen's D across comparisons");
442#endif
443 auto& min_rank_cohen = cohen[scran::differential_analysis::MIN_RANK];
444 if (min_rank_cohen.size()) {
445 differential_analysis::compute_min_rank(p->nrow(), ngroups, cohens_d.data(), min_rank_cohen);
446 }
447 differential_analysis::summarize_comparisons(p->nrow(), ngroups, cohens_d.data(), cohen); // non-const w.r.t. cohens_d, so done after min-rank calculations.
448 }
449 if (do_wilcox) {
450#ifdef SCRAN_LOGGER
451 SCRAN_LOGGER("scran::ScoreMarkers", "Summarizing the AUC across comparisons");
452#endif
453 auto& min_rank_auc = auc[scran::differential_analysis::MIN_RANK];
454 if (min_rank_auc.size()) {
455 differential_analysis::compute_min_rank(p->nrow(), ngroups, wilcox_auc.data(), min_rank_auc);
456 }
457 differential_analysis::summarize_comparisons(p->nrow(), ngroups, wilcox_auc.data(), auc); // non-const w.r.t. wilcox_auc, so done after min-rank calculations.
458 }
459
460 return;
461 }

474 auto fill_inner = [&](int N, auto& type) {
475 type.reserve(N);
476 for (int n = 0; n < N; ++n) {
477 type.emplace_back(ngenes);
478 }
479 };
480
481 means.resize(ngroups);
482 detected.resize(ngroups);
483 for (int g = 0; g < ngroups; ++g) {
484 fill_inner(nblocks, means[g]);
485 fill_inner(nblocks, detected[g]);
486 }
487
488 auto fill_effect = [&](auto& effect) {
489 effect.resize(differential_analysis::n_summaries);
490 if (do_min) {
491 fill_inner(ngroups, effect[differential_analysis::MIN]);
492 }
493 if (do_mean) {
494 fill_inner(ngroups, effect[differential_analysis::MEAN]);
495 }

499 if (do_max) {

501 }
502 if (do_rank) {
503 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
504 }
505 return;
506 };
507
508 if (do_cohen) {
509 fill_effect(cohen);
510 }
511 if (do_auc) {
512 fill_effect(auc);
