290 template<class Matrix, typename G, typename B, typename Stat>
291 void run_blocked(
292 const Matrix* p,
293 const G* group,
294 const B* block,
295 std::vector<std::vector<Stat*> > means,
296 std::vector<std::vector<Stat*> > detected,
297 Stat* cohen,
298 Stat* auc,
299 Stat* lfc,
300 Stat* delta_detected)
301 const {
302 if (block == NULL) {
303 run(p, group, fetch_first(means), fetch_first(detected), cohen, auc, lfc, delta_detected);
304 return;
305 }
306
307 size_t ngroups = means.size();
308 size_t nblocks = (ngroups ? means[0].size() : 0); // if means.size() == 0, then there are no groups, so there are no blocks, either.
309 size_t ncombos = ngroups * nblocks;
310 std::vector<int> combos(p->ncol());
311 std::vector<int> combo_size(ncombos);
312
313 for (size_t i = 0; i < combos.size(); ++i) {
314 combos[i] = group[i] * nblocks + block[i];
315 ++(combo_size[combos[i]]);
316 }
317
318 std::vector<Stat*> means2(ncombos), detected2(ncombos);
319 auto mIt = means2.begin(), dIt = detected2.begin();
320 for (int g = 0; g < ngroups; ++g) {
321 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
322 *mIt = means[g][b];
323 *dIt = detected[g][b];
324 }
325 }
326
327 core(p, combos.data(), combo_size, group, ngroups, block, nblocks, std::move(means2), std::move(detected2), cohen, auc, lfc, delta_detected);
328 }
