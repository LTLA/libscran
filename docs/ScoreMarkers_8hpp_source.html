557 void run_internal(const MAT* p, const G* group, int ngroups,
558 std::vector<Stat*>& means,
559 std::vector<Stat*>& detected,

572 template<class MAT, typename G, typename B, typename Stat>
573 void run_blocked_internal(const MAT* p, const G* group, const B* block, int ngroups, int nblocks,
574 std::vector<std::vector<Stat*> >& means,
575 std::vector<std::vector<Stat*> >& detected,
576 std::vector<std::vector<Stat*> >& cohen,
577 std::vector<std::vector<Stat*> >& auc,
578 std::vector<std::vector<Stat*> >& lfc,
579 std::vector<std::vector<Stat*> >& delta_detected)
580 {
581 int ncombos = ngroups * nblocks;
582 std::vector<int> combos(p->ncol());
583 std::vector<int> combo_size(ncombos);
584
585 for (size_t i = 0; i < combos.size(); ++i) {
586 combos[i] = group[i] * nblocks + block[i];
587 ++(combo_size[combos[i]]);
588 }
589
590 std::vector<Stat*> means2(ncombos), detected2(ncombos);
591 auto mIt = means2.begin(), dIt = detected2.begin();
592 for (int g = 0; g < ngroups; ++g) {
593 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
594 *mIt = means[g][b];
595 *dIt = detected[g][b];
596 }
597 }
598
599 core(p, combos.data(), combo_size, group, ngroups, block, nblocks, means2, detected2, cohen, auc, lfc, delta_detected);
600 }

602 template<class MAT, typename L, class Ls, typename G, typename B, typename Stat>
603 void core(const MAT* p,
604 const L* level, const Ls& level_size,
605 const G* group, int ngroups,
606 const B* block, int nblocks,
607 std::vector<Stat*>& means,
608 std::vector<Stat*>& detected,
609 std::vector<std::vector<Stat*> >& cohen,
610 std::vector<std::vector<Stat*> >& auc,
611 std::vector<std::vector<Stat*> >& lfc,
612 std::vector<std::vector<Stat*> >& delta_detected)
613 {
614 size_t buffer_size = p->nrow() * ngroups * ngroups;
615
616 const bool do_cohen = !cohen.empty();
617 std::vector<Stat> cohen_buffer(do_cohen ? buffer_size : 0);
618 Stat* cohen_ptr = do_cohen ? cohen_buffer.data() : NULL;
619
620 const bool do_auc = !auc.empty();
621 std::vector<Stat> auc_buffer(do_auc ? buffer_size : 0);
622 Stat* auc_ptr = do_auc ? auc_buffer.data() : NULL;
623
624 const bool do_lfc = !lfc.empty();
625 std::vector<Stat> lfc_buffer(do_lfc ? buffer_size : 0);
626 Stat* lfc_ptr = do_lfc ? lfc_buffer.data() : NULL;
627
628 const bool do_delta = !delta_detected.empty();
629 std::vector<Stat> delta_buffer(do_delta ? buffer_size : 0);
630 Stat* delta_ptr = do_delta ? delta_buffer.data() : NULL;
631
632 std::vector<Stat*> effects { cohen_ptr, auc_ptr, lfc_ptr, delta_ptr };
633
634#ifdef SCRAN_LOGGER
635 SCRAN_LOGGER("scran::ScoreMarkers", "Performing pairwise comparisons between groups of cells");
636#endif
637 if (!do_auc) {
638 differential_analysis::BidimensionalFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, ngroups, nblocks, threshold);
639 tatami::apply<0>(p, fact, num_threads);
640
641 } else {
642 // Need to remake this, as there's no guarantee that 'blocks' exists.
643 std::vector<B> tmp_blocks;
644 if (!block) {
645 tmp_blocks.resize(p->ncol());
646 block = tmp_blocks.data();
647 }
648
649 differential_analysis::PerRowFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, group, ngroups, block, nblocks, threshold);
650 tatami::apply<0>(p, fact, num_threads);
651 }
652
653 auto summarize = [&](Stat* ptr, std::vector<std::vector<Stat*> >& output) -> void {
654 auto& min_rank = output[scran::differential_analysis::MIN_RANK];
655 if (min_rank.size()) {
656 differential_analysis::compute_min_rank(p->nrow(), ngroups, ptr, min_rank, num_threads);
657 }
658 differential_analysis::summarize_comparisons(p->nrow(), ngroups, ptr, output, num_threads);
659 };
660
661 if (do_cohen) {
662 summarize(cohen_ptr, cohen);
663 }
664 if (do_auc) {
665 summarize(auc_ptr, auc);
666 }
667 if (do_lfc) {
668 summarize(lfc_ptr, lfc);
669 }
670 if (do_delta) {
671 summarize(delta_ptr, delta_detected);
672 }
673
674 return;
675 }

688 struct Results {
692 Results() {}
693
694 Results(
695 size_t ngenes,
696 int ngroups,
697 int nblocks,
698 const ComputeSummaries& do_cohen,
699 const ComputeSummaries& do_auc,
700 const ComputeSummaries& do_lfc,
701 const ComputeSummaries& do_delta_detected)
702 {
703 auto fill_inner = [&](int N, auto& type) {
704 type.reserve(N);
705 for (int n = 0; n < N; ++n) {
706 type.emplace_back(ngenes);
707 }
708 };
709
710 means.resize(ngroups);
711 detected.resize(ngroups);
712 for (int g = 0; g < ngroups; ++g) {
713 fill_inner(nblocks, means[g]);
714 fill_inner(nblocks, detected[g]);
715 }
716
717 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
718 bool has_any = false;
719 for (size_t i = 0; i < do_this.size(); ++i) {
720 if (do_this[i]) {
721 has_any = true;
722 break;
723 }
724 }
725
726 if (has_any) {
727 effect.resize(differential_analysis::n_summaries);
728 if (do_this[differential_analysis::MIN]) {
729 fill_inner(ngroups, effect[differential_analysis::MIN]);
730 }
731 if (do_this[differential_analysis::MEAN]) {
732 fill_inner(ngroups, effect[differential_analysis::MEAN]);
733 }
734 if (do_this[differential_analysis::MEDIAN]) {
735 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
736 }
737 if (do_this[differential_analysis::MAX]) {
738 fill_inner(ngroups, effect[differential_analysis::MAX]);
739 }
740 if (do_this[differential_analysis::MIN_RANK]) {
741 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
742 }
743 }
744 return;
745 };
746
747 fill_effect(do_cohen, cohen);
748 fill_effect(do_auc, auc);
749 fill_effect(do_lfc, lfc);
750 fill_effect(do_delta_detected, delta_detected);
751 return;
752 }
