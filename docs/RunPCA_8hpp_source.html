40  void run(const tatami::Matrix<T, IDX>* mat, Eigen::MatrixXd& pcs, Eigen::VectorXd& variance_explained, double& total_var) {
41  irlba::NormalSampler norm(42);
42  irb.set_number(rank);
43 
44  if (mat->sparse()) {

95  void clean_up(size_t NC, const Eigen::MatrixXd& U, const Eigen::VectorXd& D, Eigen::MatrixXd& pcs, Eigen::VectorXd& variance_explained) {
96  pcs = U;
97  for (int i = 0; i < U.cols(); ++i) {
98  for (size_t j = 0; j < NC; ++j) {
99  pcs(j, i) *= D[i];
100  }
101  }
102 
103  variance_explained.resize(D.size());
104  for (int i = 0; i < D.size(); ++i) {
105  variance_explained[i] = D[i] * D[i] / static_cast<double>(NC - 1);
106  }
107 
108  return;
109  }

112  template<typename T, typename IDX>
113  Eigen::SparseMatrix<double> create_eigen_matrix_sparse(const tatami::Matrix<T, IDX>* mat, Eigen::VectorXd& center_v, Eigen::VectorXd& scale_v, double& total_var) {
114  size_t NR = mat->nrow(), NC = mat->ncol();
115  total_var = 0;
116 
117  // Filling up the vector of triplets. We pre-allocate at an assumed 10%
118  // density, so as to avoid unnecessary movements.
119  typedef Eigen::Triplet<double> TRIPLET;
120  std::vector<TRIPLET> triplets;
121  triplets.reserve(static_cast<double>(NR * NC) * sparsity);
122 
123  if (mat->prefer_rows()) {
124  std::vector<double> xbuffer(NC);
125  std::vector<int> ibuffer(NC);
126 
127  for (size_t r = 0; r < NR; ++r) {
128  auto range = mat->sparse_row(r, xbuffer.data(), ibuffer.data());
129 
130  auto stats = tatami::stats::VarianceHelper::compute_with_mean(range, NC);
131  center_v[r] = stats.first;
132  if (scale) {
133  total_var += stats.second;
134  scale_v[r] = std::sqrt(stats.second);
135  } else {
136  ++total_var;
137  scale_v[r] = 1;
138  }
139 
140  for (size_t i = 0; i < range.number; ++i) {
141  triplets.push_back(TRIPLET(range.index[i], r, range.value[i])); // transposing.
142  }
143  }
144 
145  } else {
146  std::vector<double> xbuffer(NR);
147  std::vector<int> ibuffer(NR);
148 
149  tatami::stats::VarianceHelper::Sparse running(NR);
150  for (size_t c = 0; c < NC; ++c) {
151  auto range = mat->sparse_column(c, xbuffer.data(), ibuffer.data());
152  running.add(range);
153 
154  for (size_t i = 0; i < range.number; ++i) {
155  triplets.push_back(TRIPLET(c, range.index[i], range.value[i])); // transposing.
156  }
157  }
158 
159  running.finish();
160  std::copy(running.means().begin(), running.means().end(), center_v.begin());
161 
162  if (scale) {
163  total_var = std::accumulate(running.statistics().begin(), running.statistics().end(), 0.0);
164  std::copy(running.statistics().begin(), running.statistics().end(), scale_v.begin());
165  for (auto& s : scale_v) { s = std::sqrt(s); }
166  } else {
167  total_var = NR;
168  std::fill(scale_v.begin(), scale_v.end(), 1);
169  }
170  }
171 
172  Eigen::SparseMatrix<double> A(NC, NR); // transposed
173  A.setFromTriplets(triplets.begin(), triplets.end());
174  A.makeCompressed();
175  return A;
176  }

179  template<typename T, typename IDX>
180  Eigen::MatrixXd create_eigen_matrix_dense(const tatami::Matrix<T, IDX>* mat, double& total_var) {
181  size_t NR = mat->nrow(), NC = mat->ncol();
182  total_var = 0;
183 
184  Eigen::MatrixXd output(NC, NR); // transposed.
185  std::vector<double> xbuffer(NC);
186  double* outIt = output.data();
187 
188  for (size_t r = 0; r < NR; ++r, outIt += NC) {
189  auto ptr = mat->row_copy(r, outIt);
190  auto stats = tatami::stats::VarianceHelper::compute_with_mean(outIt, NC);
191 
192  auto copy = outIt;
193  for (size_t c = 0; c < NC; ++c, ++copy) {
194  *copy -= stats.first;
195  }
196 
197  if (scale) {
198  double sd = std::sqrt(stats.second);
199  auto copy = outIt;
200  for (size_t c = 0; c < NC; ++c, ++copy) {
201  *copy /= sd;
202  }
203  total_var += stats.second;
204  } else {
205  ++total_var;
206  }
207  }
208 
209  return output;
210  }
