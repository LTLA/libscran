186 template<class MAT, typename G, typename B, typename Stat>
187 void run_blocked_internal(const MAT* p, const G* group, const B* block, int ngroups, int nblocks,
188 std::vector<std::vector<Stat*> >& means,
189 std::vector<std::vector<Stat*> >& detected,
190 std::vector<std::vector<Stat*> >& cohen,
191 std::vector<std::vector<Stat*> >& auc)
192 {
193 int ncombos = ngroups * nblocks;
194 std::vector<int> combos(p->ncol());
195 std::vector<int> combo_size(ncombos);
196
197 for (size_t i = 0; i < combos.size(); ++i) {
198 combos[i] = group[i] * nblocks + block[i];
199 ++(combo_size[combos[i]]);
200 }
201
202 std::vector<Stat*> means2(ncombos), detected2(ncombos);
203 auto mIt = means2.begin(), dIt = detected2.begin();
204 for (int g = 0; g < ngroups; ++g) {
205 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
206 *mIt = means[g][b];
207 *dIt = detected[g][b];
208 }
209 }
210
211 core(p, combos.data(), combo_size, group, ngroups, block, nblocks, means2, detected2, cohen, auc);
212 }

214 template<class MAT, typename L, class Ls, typename G, typename B, typename Stat>
215 void core(const MAT* p,
216 const L* level, const Ls& level_size,
217 const G* group, int ngroups,
218 const B* block, int nblocks,
219 std::vector<Stat*>& means,
220 std::vector<Stat*>& detected,
221 std::vector<std::vector<Stat*> >& cohen,
222 std::vector<std::vector<Stat*> >& auc)
223 {
224 const bool do_cohen = !cohen.empty();
225 std::vector<Stat> cohens_d(do_cohen ? p->nrow() * ngroups * ngroups : 0);
226 Stat* cohens_ptr = do_cohen ? cohens_d.data() : NULL;
227
228 const bool do_wilcox = !auc.empty();
229 std::vector<Stat> wilcox_auc(do_wilcox ? p->nrow() * ngroups * ngroups : 0);
230
231#ifdef SCRAN_LOGGER
232 SCRAN_LOGGER("scran::ScoreMarkers", "Performing pairwise comparisons between groups of cells");
233#endif
234 if (!do_wilcox) {
235 std::vector<Stat*> effects{cohens_ptr};
236 differential_analysis::BidimensionalFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, ngroups, nblocks, threshold);
237 tatami::apply<0>(p, fact);
238 } else {
239 std::vector<Stat*> effects{cohens_ptr, wilcox_auc.data()};
240
241 // Need to remake this, as there's no guarantee that 'blocks' exists.
242 std::vector<B> tmp_blocks;
243 if (!block) {
244 tmp_blocks.resize(p->ncol());
245 block = tmp_blocks.data();
246 }
247
248 differential_analysis::PerRowFactory fact(p->nrow(), p->ncol(), means, detected, effects, level, &level_size, group, ngroups, block, nblocks, threshold);
249 tatami::apply<0>(p, fact);
250 }
251
252 if (do_cohen) {
253#ifdef SCRAN_LOGGER
254 SCRAN_LOGGER("scran::ScoreMarkers", "Summarizing Cohen's D across comparisons");
255#endif
256 auto& min_rank_cohen = cohen[scran::differential_analysis::MIN_RANK];
257 if (min_rank_cohen.size()) {
258 differential_analysis::compute_min_rank(p->nrow(), ngroups, cohens_d.data(), min_rank_cohen);
259 }
260 differential_analysis::summarize_comparisons(p->nrow(), ngroups, cohens_d.data(), cohen); // non-const w.r.t. cohens_d, so done after min-rank calculations.
261 }
262 if (do_wilcox) {
263#ifdef SCRAN_LOGGER
264 SCRAN_LOGGER("scran::ScoreMarkers", "Summarizing the AUC across comparisons");
265#endif
266 auto& min_rank_auc = auc[scran::differential_analysis::MIN_RANK];
267 if (min_rank_auc.size()) {
268 differential_analysis::compute_min_rank(p->nrow(), ngroups, wilcox_auc.data(), min_rank_auc);
269 }
270 differential_analysis::summarize_comparisons(p->nrow(), ngroups, wilcox_auc.data(), auc); // non-const w.r.t. wilcox_auc, so done after min-rank calculations.
271 }
272
273 return;
274 }

279 Results(size_t ngenes, int ngroups, int nblocks, bool do_cohen, bool do_auc, bool do_min, bool do_mean, bool do_median, bool do_max, bool do_rank) {
280 auto fill_inner = [&](int N, auto& type) {
281 type.reserve(N);
282 for (int n = 0; n < N; ++n) {
283 type.emplace_back(ngenes);
284 }
285 };
286
287 means.resize(ngroups);
288 detected.resize(ngroups);
289 for (int g = 0; g < ngroups; ++g) {
290 fill_inner(nblocks, means[g]);
291 fill_inner(nblocks, detected[g]);
292 }
293
294 auto fill_effect = [&](auto& effect) {
295 effect.resize(differential_analysis::n_summaries);
296 if (do_min) {
297 fill_inner(ngroups, effect[differential_analysis::MIN]);
298 }
299 if (do_mean) {
300 fill_inner(ngroups, effect[differential_analysis::MEAN]);
301 }
302 if (do_median) {
303 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
304 }
305 if (do_max) {
306 fill_inner(ngroups, effect[differential_analysis::MAX]);
307 }
308 if (do_rank) {
309 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
310 }
311 return;
312 };
313
314 if (do_cohen) {
315 fill_effect(cohen);
316 }
317 if (do_auc) {
318 fill_effect(auc);
319 }
320 return;
321 }
