195 std::deque<WeightedEdge> run(const std::vector<std::vector<int> >& indices) const {
196 size_t ncells = indices.size();
197
198 // Not parallel-frendly, so we don't construct this with the neighbor search
199 std::vector<std::vector<std::pair<int, size_t> > > hosts(ncells);
200 for (size_t i = 0; i < ncells; ++i) {
201 hosts[i].push_back(std::make_pair(0, i)); // each point is its own 0-th nearest neighbor
202 const auto& current = indices[i];
203 int counter = 1;
204 for (auto x : current) {
205 hosts[x].push_back(std::make_pair(counter, i));
206 ++counter;
207 }
208 }
209
210 // Constructing the shared neighbor graph.
211#ifdef SCRAN_LOGGER
212 SCRAN_LOGGER("scran::BuildSNNGraph", "Computing shared neighbor weights");
213#endif
214 std::vector<size_t> current_added;
215 current_added.reserve(ncells);
216 std::vector<int> current_score(ncells);
217 std::deque<WeightedEdge> store;
218
219 for (size_t j = 0; j < ncells; ++j) {
220 const auto& current_neighbors = indices[j];
221
222 for (int i = 0; i <= current_neighbors.size(); ++i) {
223 // First iteration treats 'j' as the zero-th neighbor.
224 // Remaining iterations go through the neighbors of 'j'.
225 const int cur_neighbor = (i==0 ? j : current_neighbors[i-1]);
226
227 // Going through all observations 'h' for which 'cur_neighbor'
228 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
229 // neighbor of both 'h' and 'j'.
230 for (auto& h : hosts[cur_neighbor]) {
231 const auto& othernode = h.second;
232
233 if (othernode < j) { // avoid duplicates from symmetry in the SNN calculations.
234 int& existing_other = current_score[othernode];
235 if (weight_scheme == RANKED) {
236 // Recording the lowest combined rank per neighbor.
237 int currank = h.first + i;
238 if (existing_other == 0) {
239 existing_other = currank;
240 current_added.push_back(othernode);
241 } else if (existing_other > currank) {
242 existing_other = currank;
243 }
244 } else {
245 // Recording the number of shared neighbors.
246 if (existing_other==0) {
247 current_added.push_back(othernode);
248 }
249 ++existing_other;
250 }
251 }
252 }
253 }
254
255 // Converting to edges.
256 for (auto othernode : current_added) {
257 int& otherscore=current_score[othernode];
258 double finalscore;
259 if (weight_scheme == RANKED) {
260 finalscore = static_cast<double>(current_neighbors.size()) - 0.5 * static_cast<double>(otherscore);
261 } else {
262 finalscore = otherscore;
263 if (weight_scheme == JACCARD) {
264 finalscore = finalscore / (2 * (current_neighbors.size() + 1) - finalscore);
265 }
266 }
267
268 // Ensuring that an edge with a positive weight is always reported.
269 store.push_back(WeightedEdge(j, othernode, std::max(finalscore, 1e-6)));
270
271 // Resetting all those added to zero.
272 otherscore=0;
273 }
274 current_added.clear();
275 }
276
277 return store;
278 }
