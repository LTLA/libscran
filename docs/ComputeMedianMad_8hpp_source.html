244 void compute(size_t n, Buffer* ptr, double& median, double& mad) const {
245 if (!n) {
246 median = std::numeric_limits<double>::quiet_NaN();
247 mad = std::numeric_limits<double>::quiet_NaN();
248 return;
249 }
250
251 if (log) {
252 auto copy = ptr;
253 for (size_t i = 0; i < n; ++i, ++copy) {
254 if (*copy > 0) {
255 *copy = std::log(*copy);
256 } else if (*copy == 0) {
257 *copy = -std::numeric_limits<double>::infinity();
258 } else {
259 throw std::runtime_error("cannot log-transform negative values");
260 }
261 }
262 }
263
264 // First, getting the median.
265 size_t halfway = n / 2;
266 median = compute_median(ptr, halfway, n);
267
268 if (std::isnan(median)) {
269 // Giving up.
270 mad = std::numeric_limits<double>::quiet_NaN();
271 return;
272 } else if (std::isinf(median)) {
273 // MADs should be no-ops when added/subtracted from infinity. Any
274 // finite value will do here, so might as well keep it simple.
275 mad = 0;
276 return;
277 }
278
279 // Now getting the MAD.
280 auto copy = ptr;
281 for (size_t i = 0; i < n; ++i, ++copy) {
282 *copy = std::abs(*copy - median);
283 }
284 mad = compute_median(ptr, halfway, n);
285 mad *= 1.4826; // for equivalence with the standard deviation under normality.
286 return;
287 }

289 static double compute_median (double* ptr, size_t halfway, size_t n) {
290 std::nth_element(ptr, ptr + halfway, ptr + n);
291 double medtmp = *(ptr + halfway);
292
293 if (n % 2 == 0) {
294 std::nth_element(ptr, ptr + halfway - 1, ptr + n);
295 double left = *(ptr + halfway - 1);
296
297 bool inf_left = std::isinf(left);
298 bool inf_right = std::isinf(medtmp);
299 if (inf_left && inf_right) {
300 if ((inf_left < 0) != (inf_right > 0)) {
301 return std::numeric_limits<double>::quiet_NaN();
302 } else {
303 return inf_left;
304 }
305 }
306
307 medtmp = (medtmp + left)/2;
308 }
309
310 return medtmp;
311 }
