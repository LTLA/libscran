175 template<typename T, typename X>
176 Thresholds run(size_t n, const T* metrics, X* outliers, std::vector<double>& buffer) const {
177 Thresholds output(1);
178 std::copy(metrics, metrics + n, buffer.data());
179
180 double& lthresh = output.lower[0];
181 double& uthresh = output.upper[0];
182 compute_thresholds(n, buffer.data(), lthresh, uthresh);
183
184 auto copy = metrics;
185 for (size_t i = 0; i < n; ++i, ++copy, ++outliers) {
186 const double val = *copy;

188 }

342 void compute_thresholds(size_t n, double* ptr, double& lower_threshold, double& upper_threshold) const {
343 if (!n) {
344 lower_threshold = std::numeric_limits<double>::quiet_NaN();
345 upper_threshold = std::numeric_limits<double>::quiet_NaN();
346 return;
347 }
348
349 if (log) {
350 auto copy = ptr;
351 for (size_t i = 0; i < n; ++i, ++copy) {
352 if (*copy > 0) {
353 *copy = std::log(*copy);
354 } else if (*copy == 0) {
355 *copy = -std::numeric_limits<double>::infinity();
356 } else {
357 throw std::runtime_error("cannot log-transform negative values");
358 }
359 }
360 }
361
362 // First, getting the median.
363 size_t halfway = n / 2;
364 double medtmp = compute_median(ptr, halfway, n);
365
366 if (log && std::isinf(medtmp)) {
367 if (lower) {
368 lower_threshold = 0;
369 } else {
370 lower_threshold = -std::numeric_limits<double>::infinity();
371 }
372
373 if (upper) {
374 upper_threshold = 0;
375 } else {
376 upper_threshold = std::numeric_limits<double>::infinity();
377 }
378
379 return;
380 }
381
382 // Now getting the MAD. No need to protect against -inf here;
383 // if the median is finite, the MAD must also be finite.
384 auto copy = ptr;
385 for (size_t i = 0; i < n; ++i, ++copy) {
386 *copy = std::abs(*copy - medtmp);
387 }
388 double madtmp = compute_median(ptr, halfway, n);
389
390 madtmp *= 1.4826; // for equivalence with the standard deviation under normality.
391 double delta = std::max(min_diff, nmads * madtmp);
392
393 // Computing the outputs.
394 if (lower) {
395 lower_threshold = medtmp - delta;
396 if (log) {
397 lower_threshold = std::exp(lower_threshold);
398 }
399 } else {
400 lower_threshold = -std::numeric_limits<double>::infinity();
401 }
402
403 if (upper) {
404 upper_threshold = medtmp + delta;
405 if (log) {
406 upper_threshold = std::exp(upper_threshold);
407 }
408 } else {
409 upper_threshold = std::numeric_limits<double>::infinity();
410 }
411
412 return;
413 }
414
415 static double compute_median (double* ptr, size_t halfway, size_t n) {
416 std::nth_element(ptr, ptr + halfway, ptr + n);
417 double medtmp = *(ptr + halfway);
418
419 if (n % 2 == 0) {
420 std::nth_element(ptr, ptr + halfway - 1, ptr + n);
421 medtmp = (medtmp + *(ptr + halfway - 1))/2;
422 }
423
424 return medtmp;
425 }
