22template<class IT>
23double median (IT start, size_t n) {
24 int halfway = n / 2;
25 std::nth_element(start, start + halfway, start + n);
26
27 if (n % 2 == 1) {
28 return start[halfway];
29 }
30
31 double med = start[halfway];
32 std::nth_element(start, start + halfway - 1, start + halfway);
33 return (med + start[halfway - 1])/ 2;
34}

36template<typename Stat>
37void summarize_comparisons(size_t ngenes, int ngroups, Stat* effects, std::vector<std::vector<Stat*> >& output) {
38 #pragma omp parallel for
39 for (size_t gene = 0; gene < ngenes; ++gene) {
40 auto base = effects + gene * ngroups * ngroups;
41 for (int l = 0; l < ngroups; ++l) {
42 auto start = base + l * ngroups;
43
44 // Ignoring the self comparison.
45 int restart = 1;
46 std::swap(start[0], start[l]);
47
48 // Pruning out NAs.
49 for (int r = restart; r < ngroups; ++r) {
50 if (std::isnan(start[r])) {
51 if (r != restart) {
52 std::swap(start[restart], start[r]);
53 }
54 ++restart;
55 }
56 }
57
58 if (restart == ngroups) {
59 for (size_t i = 0; i < MIN_RANK; ++i) {
60 if (output[i].size()) {
61 output[i][l][gene] = std::numeric_limits<double>::quiet_NaN();
62 }
63 }
64 } else {
65 int ncomps = ngroups - restart;
66 if (ncomps > 1) {
67 if (output[MIN].size()) {
68 output[MIN][l][gene] = *std::min_element(start + restart, start + ngroups);
69 }
70 if (output[MEAN].size()) {
71 output[MEAN][l][gene] = std::accumulate(start + restart, start + ngroups, 0.0) / ncomps; // Mean
72 }
73 if (output[MEDIAN].size()) {
74 output[MEDIAN][l][gene] = median(start + restart, ncomps); // Median
75 }
76 if (output[MAX].size()) {
77 output[MAX][l][gene] = *std::max_element(start + restart, start + ngroups); // Maximum
78 }
79 } else {
80 for (size_t i = 0; i < MIN_RANK; ++i) {
81 if (output[i].size()) {
82 output[i][l][gene] = start[restart];
83 }
84 }
85 }
86 }
87 }
88 }
89 return;
90}

92template<typename Stat>
93void compute_min_rank(size_t ngenes, int ngroups, const Stat* effects, std::vector<Stat*>& output) {
94 auto shift = ngroups * ngroups;
95
96 #pragma omp parallel
97 {
98 std::vector<std::pair<Stat, int> > buffer(ngenes);
99
100 #pragma omp for
101 for (int g = 0; g < ngroups; ++g) {
102 auto target = output[g];
103 std::fill(target, target + ngenes, ngenes + 1);
104
105 for (int g2 = 0; g2 < ngroups; ++g2) {
106 if (g == g2) {
107 continue;
108 }
109 auto copy = effects + g * ngroups + g2;
110 for (size_t i = 0; i < ngenes; ++i, copy += shift) {
111 buffer[i].first = -*copy; // negative to sort by decreasing value.
112 buffer[i].second = i;
113 }
114 std::sort(buffer.begin(), buffer.end());
115
116 double counter = 1;
117 for (const auto& x : buffer) {
118 target[x.second] = std::min(counter, target[x.second]);
119 ++counter;
120 }
121 }
122 }
123 }
124}
