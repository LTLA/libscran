196 template<class MAT, class V, typename B>
197 std::shared_ptr<MAT> run_blocked(std::shared_ptr<MAT> mat, V size_factors, const B* block) const {
198 // One might ask why we don't require a pointer for size_factors here.
199 // It's because size_factors need to be moved into the Delayed operation
200 // anyway, so we might as well ask the user to construct a vector for us.
201 if (size_factors.size() != mat->ncol()) {
202 throw std::runtime_error("number of size factors and columns are not equal");
203 }
204
205 bool has_zero = false;
206 if (center) {
207 // Falls back to centerer.run() if block=NULL.
208 has_zero = centerer.run_blocked(size_factors.size(), size_factors.data(), block);
209 } else {
210 // CenterSizeFactors do their own validity checks,
211 // so we don't need to call it again in that case.
212 has_zero = CenterSizeFactors::validate(size_factors.size(), size_factors.data());
213 }
214
215 if (has_zero) {
216 if (!handle_zeros) {
217 throw std::runtime_error("all size factors should be positive");
218 } else if (size_factors.size()) {
219 // Replacing them with the smallest non-zero size factor, or 1.
220 auto smallest = std::numeric_limits<double>::infinity();
221 for (auto s : size_factors) {
222 if (s && smallest > s) {
223 smallest = s;
224 }
225 }
226
227 if (std::isinf(smallest)) {
228 smallest = 1;
229 }
230
231 for (auto& s : size_factors) {
232 if (s == 0) {
233 s = smallest;
234 }
235 }
236 }
237 }
238
239 if (sparse_addition && pseudo_count != 1) {
240 for (auto& d : size_factors) {
241 d *= pseudo_count;
242 }
243 }
244
245 auto div = tatami::make_DelayedIsometricOp(mat, tatami::make_DelayedDivideVectorHelper<true, 1>(std::move(size_factors)));
246
247 if (pseudo_count == 1 || sparse_addition) {
248 return tatami::make_DelayedIsometricOp(div, tatami::DelayedLog1pHelper(2.0));
249 } else {
250 auto add = tatami::make_DelayedIsometricOp(div, tatami::DelayedAddScalarHelper<double>(pseudo_count));
251 return tatami::make_DelayedIsometricOp(add, tatami::DelayedLogHelper(2.0));
252 }
253 }
