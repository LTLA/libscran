119 Graph build(size_t ncells, const std::deque<BuildSNNGraph::WeightedEdge>& store) const {
120#ifdef SCRAN_LOGGER
121 SCRAN_LOGGER("scran::ClusterSNNGraph", "Filling the graph object");
122#endif
123 Graph output;
124 igraph_vector_t edges;
125 igraph_vector_init(&edges, store.size() * 2);
126
127 igraph_vector_t& weights = output.weights;
128 igraph_vector_init(&weights, store.size());
129
130 size_t counter = 0;
131 for (size_t i = 0; i < store.size(); ++i, counter += 2) { // not entirely sure it's safe to use std::copy here.
132 const auto& edge = store[i];
133 VECTOR(edges)[counter] = std::get<0>(edge);
134 VECTOR(edges)[counter + 1] = std::get<1>(edge);
135 VECTOR(weights)[i] = std::get<2>(edge);
136 }
137
138 igraph_create(&(output.graph), &edges, ncells, 0);
139 igraph_vector_destroy(&edges);
140
141 return output;
142 }

768 Results run(const Graph& graph_info) {
769 igraph_vector_t membership;
770 igraph_integer_t nb_clusters;
771 igraph_real_t quality;
772
773 igraph_vector_init(&membership, 0);
774
775 IgraphRNGScope rngs(seed);
776
777 Results output;
778
779 if (!modularity) {
780 for (int i = 0; i < iterations; ++i) {
781 output.status = igraph_community_leiden(&graph_info.graph, &graph_info.weights, NULL, resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
782 if (output.status) {
783 break;
784 }
785 }
786 } else {
787 // Based on https://igraph.org/c/doc/igraph-Community.html#igraph_community_leiden.
788 igraph_vector_t degree;
789 igraph_vector_init(&degree, igraph_vcount(&graph_info.graph));
790 igraph_degree(&graph_info.graph, &degree, igraph_vss_all(), IGRAPH_ALL, 1);
791
792 // This assumes that resolution = 1 in the example in the C documentation.
793 // igraph::cluster_leiden in the R package does the same thing.
794 double mod_resolution = resolution / (2 * igraph_ecount(&graph_info.graph));
795
796 for (int i = 0; i < iterations; ++i) {
797 output.status = igraph_community_leiden(&graph_info.graph, &graph_info.weights, &degree, mod_resolution, beta, (i > 0), &membership, &nb_clusters, &quality);
798 if (output.status) {
799 break;
800 }
801 }
802
803 igraph_vector_destroy(&degree);
804 }
805
806 if (!output.status) {
807 size_t ncells = igraph_vcount(&graph_info.graph);
808 output.membership.resize(ncells);
809 for (size_t i = 0; i < ncells; ++i) {
810 output.membership[i] = VECTOR(membership)[i];
811 }
812 output.quality = quality;
813 }
814
815 igraph_vector_destroy(&membership);
