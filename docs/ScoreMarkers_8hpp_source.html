541 means.resize(ngroups);
542 detected.resize(ngroups);
543 for (int g = 0; g < ngroups; ++g) {
544 fill_inner(nblocks, means[g]);
545 fill_inner(nblocks, detected[g]);
546 }
547
548 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
549 bool has_any = false;
550 for (size_t i = 0; i < do_this.size(); ++i) {
551 if (do_this[i]) {
552 has_any = true;
553 break;
554 }
555 }
556
557 if (has_any) {
558 effect.resize(differential_analysis::n_summaries);
559 if (do_this[differential_analysis::MIN]) {
560 fill_inner(ngroups, effect[differential_analysis::MIN]);
561 }
562 if (do_this[differential_analysis::MEAN]) {
563 fill_inner(ngroups, effect[differential_analysis::MEAN]);
564 }
565 if (do_this[differential_analysis::MEDIAN]) {
566 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
567 }
568 if (do_this[differential_analysis::MAX]) {
569 fill_inner(ngroups, effect[differential_analysis::MAX]);
570 }
571 if (do_this[differential_analysis::MIN_RANK]) {
572 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
573 }
574 }
575 return;
576 };

578 fill_effect(do_cohen, cohen);
579 fill_effect(do_auc, auc);
580 fill_effect(do_lfc, lfc);
