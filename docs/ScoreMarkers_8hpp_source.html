31// This cache tries to store as many of the reverse effects as possible before
32// it starts evicting. Evictions are based on the principle that it is better
33// to store effects that will be re-used quickly, thus freeing up the cache for
34// future stores. The 'speed' of reusability of each cache entry depends on the
35// first group in the comparison corresponding to each cached effect size; the
36// smaller the first group, the sooner it will be reached when iterating across
37// groups in the ScoreMarkers function.
38//
39// So, the policy is to evict cache entries when the identity of the first
40// group in the cached entry is larger than the corresponding value for the
41// incoming entry. Given that, if the cache is full, we have to throw away
42// one of these effects anyway, I'd prefer to hold onto the one we're using
43// soon, because at least it'll be freed up rapidly.
44template<typename Stat>
45struct EffectsCacher {
46 EffectsCacher(size_t ngenes_, int ngroups_, int cache_size_) :
47 ngenes(ngenes_),
48 ngroups(ngroups_),
49 cache_size(cache_size_),
50 full_set(ngroups * ngenes),
51 actions(ngroups),
52 staging_cache(ngroups)
53 {
54 vector_pool.reserve(cache_size);
55 }
56
57public:
58 size_t ngenes;
59 int ngroups;
60 int cache_size;
61
62 std::vector<double> full_set;
63 std::vector<unsigned char> actions;
64 std::vector<std::vector<Stat> > staging_cache;
65 std::vector<std::vector<Stat> > vector_pool;
66
67 std::map<std::pair<int, int>, std::vector<Stat> > cached;
68
69public:
70 void clear() {
71 cached.clear();
72 }
73
74public:
75 void configure(int group) {
76 for (int other = 0; other < ngroups; ++other) {
77 if (other == group) {
78 actions[other] = 0;
79 continue;
80 }
81
82 if (cache_size == 0) {
83 actions[other] = 1;
84 continue;
85 }
86
87 // Impossible to be cached by a previous run, so we can skip a
88 // look-up - however, it is a candidate to _be_ cached, possibly
89 // evicting other existing cache entries.
90 if (other > group) {
91 actions[other] = 2;
92 continue;
93 }
94
95 // Need to recompute cache entries that were previously evicted.
96 if (cached.empty()) {
97 actions[other] = 1;
98 continue;
99 }
100
101 const auto& front = cached.begin()->first;
102 if (front.first > group || front.second > other) { // less-thans should be impossible.
103 actions[other] = 1;
104 continue;
105 }
106
107 // Transferring the cached vector to the full_set.
108 actions[other] = 0;
109 auto& x = cached.begin()->second;
110 for (size_t i = 0, end = x.size(); i < end; ++i) {
111 full_set[other + i * ngroups] = x[i];
112 }
113
114 vector_pool.emplace_back(std::move(x)); // recycle memory to avoid heap reallocations.
115 cached.erase(cached.begin());
116 }
117
118 // Refining our choice by doing a dummy run and seeing
119 // whether eviction actually happens. If it doesn't, we won't
120 // bother caching, beause that would be a waste of memory accesses.
121 for (int other = 0; other < ngroups; ++other) {
122 if (actions[other] != 2) {
123 continue;
124 }
125
126 std::pair<int, int> key(other, group);
127 if (cached.size() < static_cast<size_t>(cache_size)) {
128 cached[key] = std::vector<Stat>();
129 prepare_staging_cache(other);
130 continue;
131 }
132
133 auto it = cached.end();
134 --it;
135 if ((it->first).first > other) {
136 // Evict the existing entry. This is safe to do so as the
137 // evicted entry won't be used in this round anyway.
138 auto& evicted = it->second;
139 if (evicted.size()) {
140 vector_pool.emplace_back(std::move(evicted));
141 }
142 cached.erase(it);
143 prepare_staging_cache(other);
144 cached[key] = std::vector<Stat>();
145 } else {
146 // No evictions, so we don't even bother computing the reverse.
147 actions[other] = 1;
148 }
149 }
150 }
151
152 void transfer(int group) {
153 for (int other = 0; other < ngroups; ++other) {
154 if (actions[other] != 2) {
155 continue;
156 }
157
158 // All the to-be-staged logic is already implemented in 'configure()'.
159 auto& staged = staging_cache[other];
160 std::pair<int, int> key(other, group);
161 cached[key] = std::move(staged);
162 }
163 }
164
165private:
166 void prepare_staging_cache(int other) {
167 if (vector_pool.empty()) {
168 staging_cache[other].resize(ngenes);
169 } else {
170 // Recycling existing heap allocations.
171 staging_cache[other] = std::move(vector_pool.back());
172 vector_pool.pop_back();
173 }
174 }
175};

640 std::vector<std::vector<Stat*> > means,
641 std::vector<std::vector<Stat*> > detected,
642 std::vector<std::vector<Stat*> > cohen,

651 Overlord<Stat> overlord(p->nrow(), ngroups, auc.empty());

654 int ncombos = ngroups * nblocks;

656 auto mIt = means2.begin(), dIt = detected2.begin();

659 *mIt = means[g][b];
660 *dIt = detected[g][b];
661 }
662 }
663
664 process_simple_effects(state, means2, detected2, cohen, lfc, delta_detected);
665 summarize_auc(state, auc, overlord.auc_buffer);

698 template<typename Stat>
699 void process_simple_effects(
700 const differential_analysis::EffectsCalculator::State& state,
701 std::vector<Stat*>& means,
702 std::vector<Stat*>& detected,
703 std::vector<std::vector<Stat*> >& cohen,
704 std::vector<std::vector<Stat*> >& lfc,
705 std::vector<std::vector<Stat*> >& delta_detected)
706 const {
707 auto ngenes = state.ngenes;
708 const auto& level_size = state.level_size;
709 auto nlevels = level_size.size();
710 auto ngroups = state.ngroups;
711 auto nblocks = state.nblocks;
712 const auto* tmp_means = state.means.data();
713 const auto* tmp_variances = state.variances.data();
714 const auto* tmp_detected = state.detected.data();
715
716 // Transferring the block-wise statistics over.
717 {
718 auto my_means = tmp_means;
719 auto my_detected = tmp_detected;
720 for (size_t gene = 0; gene < ngenes; ++gene) {
721 for (size_t l = 0; l < nlevels; ++l, ++my_means, ++my_detected) {
722 means[l][gene] = *my_means;
723 detected[l][gene] = *my_detected;
724 }
725 }
726 }
727
728 // Looping over each group and computing the various summaries. We do this on
729 // a per-group basis to avoid having to store the full group-by-group matrix of
730 // effect sizes that we would otherwise need as input to SummarizeEffects.
731 differential_analysis::EffectsCacher<Stat> cache(ngenes, ngroups, cache_size);
732 auto& full_set = cache.full_set;
733 auto& actions = cache.actions;
734 auto& staging_cache = cache.staging_cache;
735
736 if (cohen.size()) {
737 cache.clear();
738
739 for (int group = 0; group < ngroups; ++group) {
740 cache.configure(group);
741
742#ifndef SCRAN_CUSTOM_PARALLEL
743 #pragma omp parallel num_threads(nthreads)
744 {
745 std::vector<double> effect_buffer(ngroups);
746 #pragma omp for
747 for (size_t gene = 0; gene < ngenes; ++gene) {
748#else
749 SCRAN_CUSTOM_PARALLEL(ngenes, [&](size_t start, size_t end) -> void {
750 std::vector<double> effect_buffer(ngroups);
751 for (size_t gene = start; gene < end; ++gene) {
752#endif
753
754 size_t in_offset = nlevels * gene;
755 auto my_means = tmp_means + in_offset;
756 auto my_variances = tmp_variances + in_offset;
757
758 auto cohen_ptr = full_set.data() + gene * ngroups;
759 for (int other = 0; other < ngroups; ++other) {
760 if (actions[other] == 0) {
761 continue;
762 }
763
764 if (actions[other] == 1) {
765 cohen_ptr[other] = differential_analysis::compute_pairwise_cohens_d<false>(group, other, my_means, my_variances, level_size, ngroups, nblocks, threshold);
766 continue;
767 }
768
769 auto tmp = differential_analysis::compute_pairwise_cohens_d<true>(group, other, my_means, my_variances, level_size, ngroups, nblocks, threshold);
770 cohen_ptr[other] = tmp.first;
771 staging_cache[other][gene] = tmp.second;
772 }
773
774 differential_analysis::summarize_comparisons(ngroups, cohen_ptr, group, gene, cohen, effect_buffer);
775
776#ifndef SCRAN_CUSTOM_PARALLEL
777 }
778 }
779#else
780 }
781 }, nthreads);
782#endif
783
784 if (cohen[differential_analysis::summary::MIN_RANK].size()) {
785 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_set.data(), cohen[differential_analysis::summary::MIN_RANK][group], nthreads);
786 }
787
788 cache.transfer(group);
789 }
790 }
791
792 if (lfc.size()) {
793 cache.clear();
794
795 for (int group = 0; group < ngroups; ++group) {
796 cache.configure(group);
797
798#ifndef SCRAN_CUSTOM_PARALLEL
799 #pragma omp parallel num_threads(nthreads)
800 {
801 std::vector<double> effect_buffer(ngroups);
802 #pragma omp for
803 for (size_t gene = 0; gene < ngenes; ++gene) {
804#else
805 SCRAN_CUSTOM_PARALLEL(ngenes, [&](size_t start, size_t end) -> void {
806 std::vector<double> effect_buffer(ngroups);
807 for (size_t gene = start; gene < end; ++gene) {
808#endif
809
810 auto my_means = tmp_means + nlevels * gene;
811 auto lfc_ptr = full_set.data() + gene * ngroups;
812 for (int other = 0; other < ngroups; ++other) {
813 if (actions[other] == 0) {
814 continue;
815 }
816
817 auto val = differential_analysis::compute_pairwise_simple_diff(group, other, my_means, level_size, ngroups, nblocks);
818 lfc_ptr[other] = val;
819 if (actions[other] == 2) {
820 staging_cache[other][gene] = -val;
821 }
822 }
823
824 differential_analysis::summarize_comparisons(ngroups, lfc_ptr, group, gene, lfc, effect_buffer);
825
826#ifndef SCRAN_CUSTOM_PARALLEL
827 }
828 }
829#else
830 }
831 }, nthreads);
832#endif
833
834 if (lfc[differential_analysis::summary::MIN_RANK].size()) {
835 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_set.data(), lfc[differential_analysis::summary::MIN_RANK][group], nthreads);
836 }
837
838 cache.transfer(group);
839 }
840 }
841
842 if (delta_detected.size()) {
843 cache.clear();
844
845 for (int group = 0; group < ngroups; ++group) {
846 cache.configure(group);
847
848#ifndef SCRAN_CUSTOM_PARALLEL
849 #pragma omp parallel num_threads(nthreads)
850 {
851 std::vector<double> effect_buffer(ngroups);
852 #pragma omp for
853 for (size_t gene = 0; gene < ngenes; ++gene) {
854#else
855 SCRAN_CUSTOM_PARALLEL(ngenes, [&](size_t start, size_t end) -> void {
856 std::vector<double> effect_buffer(ngroups);
857 for (size_t gene = start; gene < end; ++gene) {
858#endif
859
860 auto my_detected = tmp_detected + gene * nlevels;
861 auto delta_detected_ptr = full_set.data() + gene * ngroups;
862
863 for (int other = 0; other < ngroups; ++other) {
864 if (actions[other] == 0) {
865 continue;
866 }
867
868 auto val = differential_analysis::compute_pairwise_simple_diff(group, other, my_detected, level_size, ngroups, nblocks);
869 delta_detected_ptr[other] = val;
870 if (actions[other] == 2) {
871 staging_cache[other][gene] = -val;
872 }
873 }
874
875 differential_analysis::summarize_comparisons(ngroups, delta_detected_ptr, group, gene, delta_detected, effect_buffer);
876
877#ifndef SCRAN_CUSTOM_PARALLEL
878 }
879 }
880#else
881 }
882 }, nthreads);
883#endif
884
885 if (delta_detected[differential_analysis::summary::MIN_RANK].size()) {
886 differential_analysis::compute_min_rank(ngenes, ngroups, group, full_set.data(), delta_detected[differential_analysis::summary::MIN_RANK][group], nthreads);
887 }
888
889 cache.transfer(group);
890 }

928 Results(
929 size_t ngenes,
930 int ngroups,

932 const ComputeSummaries& do_cohen,
933 const ComputeSummaries& do_auc,
934 const ComputeSummaries& do_lfc,

936 {
937 auto fill_inner = [&](int N, auto& type) {
938 type.reserve(N);
939 for (int n = 0; n < N; ++n) {
940 type.emplace_back(ngenes);
941 }
942 };
943
944 means.resize(ngroups);
945 detected.resize(ngroups);
946 for (int g = 0; g < ngroups; ++g) {
947 fill_inner(nblocks, means[g]);
948 fill_inner(nblocks, detected[g]);
949 }
950
951 auto fill_effect = [&](const ComputeSummaries& do_this, auto& effect) {
952 bool has_any = false;
953 for (size_t i = 0; i < do_this.size(); ++i) {
954 if (do_this[i]) {
955 has_any = true;
956 break;
957 }
958 }
959
960 if (has_any) {
961 effect.resize(differential_analysis::n_summaries);
962 if (do_this[differential_analysis::MIN]) {
963 fill_inner(ngroups, effect[differential_analysis::MIN]);
964 }
965 if (do_this[differential_analysis::MEAN]) {
966 fill_inner(ngroups, effect[differential_analysis::MEAN]);
967 }
968 if (do_this[differential_analysis::MEDIAN]) {
969 fill_inner(ngroups, effect[differential_analysis::MEDIAN]);
970 }
971 if (do_this[differential_analysis::MAX]) {
972 fill_inner(ngroups, effect[differential_analysis::MAX]);
973 }
974 if (do_this[differential_analysis::MIN_RANK]) {
975 fill_inner(ngroups, effect[differential_analysis::MIN_RANK]);
976 }
977 }
978 return;
979 };
