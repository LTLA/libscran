152 template<typename Index, typename Float>
153 void project_location(
154 const std::vector<std::pair<Index, Float> >& neighbors,
155 const std::vector<std::vector<Index> >& embedded_neighbors,
156 int ndim,
157 const Float* ref,
158 const Float* coord,
159 int nembed,
160 const Float* embedding,
161 std::unordered_map<Index, Float>& cache,
162 Float* output)
163 const {
164 if (neighbors.empty()) {
165 std::fill(output, output + nembed, std::numeric_limits<Float>::quiet_NaN());
166 return;
167 }
168
169 Float bandwidth = neighbors.front().second;
170 if (bandwidth == 0) {
171 auto src = embedding + nembed * neighbors.front().first;
172 std::copy(src, src + nembed, output);
173 return;
174 }
175
176 // Compute the weight using a gaussian kernel.
177 Float denom = bandwidth * bandwidth * 2;
178 auto compute_weight = [&](Float dist2) -> Float {
179 // Protect against float overflow.
180 if (denom > 1 || std::numeric_limits<Float>::max() * denom > dist2) {
181 return std::exp(-dist2 / denom);
182 } else {
183 return 0;
184 }
185 };
186
187 // Prefilling the immediate neighbors to avoid unnecessary compute.
188 cache.clear();
189 for (const auto& neighbor : neighbors) {
190 cache[neighbor.first] = compute_weight(neighbor.second * neighbor.second);
191 }
192
193 // Finding the neighbor with the max total weight amongst its own neighbor set.
194 Float max_weight = 0;
195 Index best_neighbor = neighbors.front().first;
196
197 for (const auto& neighbor : neighbors) {
198 const auto& candidates = embedded_neighbors[neighbor.first];
199 Float current_weight = 0;
200
201 for (auto x : candidates) {
202 Float w;
203 auto it = cache.find(x);
204 if (it == cache.end()) {
205 Float dist2 = 0;
206 auto pos = ref + x * ndim;
207 for (int d = 0; d < ndim; ++d) {
208 dist2 += (pos[d] - coord[d]) * (pos[d] - coord[d]);
209 }
210 w = compute_weight(dist2);
211 cache[x] = w;
212 } else {
213 w = it->second;
214 }
215 current_weight += w;
216 }
217
218 if (current_weight > max_weight) {
219 max_weight = current_weight;
220 best_neighbor = neighbor.first;
221 }
222 }
223
224 // Computing the average of the neighbor set with the largest weights.
225 // The maximum weight better be positive!
226 std::fill(output, output + nembed, 0);
227 const auto& candidates = embedded_neighbors[best_neighbor];
228 for (auto x : candidates) {
229 auto w = cache.at(x) / max_weight;
230 auto src = embedding + x * nembed;
231 for (int e = 0; e < nembed; ++e) {
232 output[e] += w * src[e];
233 }
234 }
235 }
