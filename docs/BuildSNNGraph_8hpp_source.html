114#ifdef SCRAN_LOGGER
115 SCRAN_LOGGER("scran::BuildSNNGraph", "Identifying nearest neighbors");
116#endif
117 std::vector<std::vector<int> > indices(ncells);
118 std::vector<std::vector<std::pair<int, size_t> > > hosts(ncells);
119
120 for (size_t i = 0; i < ncells; ++i) {
121 auto neighbors = ptr->find_nearest_neighbors(i, num_neighbors);
122 auto& current = indices[i];
123
124 hosts[i].push_back(std::make_pair(0, i)); // each point is its own 0-th nearest neighbor
125 int counter = 1;
126 for (auto x : neighbors) {
127 current.push_back(x.first);
128 hosts[x.first].push_back(std::make_pair(counter, i));
129 ++counter;
130 }
131 }
132
133 // Constructing the shared neighbor graph.
134#ifdef SCRAN_LOGGER
135 SCRAN_LOGGER("scran::BuildSNNGraph", "Computing shared neighbor weights");
136#endif
137 std::vector<size_t> current_added;
138 current_added.reserve(ncells);
139 std::vector<int> current_score(ncells);
140
141 for (size_t j = 0; j < ncells; ++j) {
142 const auto& current_neighbors = indices[j];
143
144 for (int i = 0; i <= current_neighbors.size(); ++i) {
145 // First iteration treats 'j' as the zero-th neighbor.
146 // Remaining iterations go through the neighbors of 'j'.
147 const int cur_neighbor = (i==0 ? j : current_neighbors[i-1]);
148
149 // Going through all observations 'h' for which 'cur_neighbor'
150 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
151 // neighbor of both 'h' and 'j'.
152 for (auto& h : hosts[cur_neighbor]) {
153 const auto& othernode = h.second;
154
155 if (othernode < j) { // avoid duplicates from symmetry in the SNN calculations.
156 int& existing_other = current_score[othernode];
157 if (weight_scheme == RANKED) {
158 // Recording the lowest combined rank per neighbor.
159 int currank = h.first + i;
160 if (existing_other == 0) {
161 existing_other = currank;
162 current_added.push_back(othernode);
163 } else if (existing_other > currank) {
164 existing_other = currank;
165 }
166 } else {
167 // Recording the number of shared neighbors.
168 if (existing_other==0) {
169 current_added.push_back(othernode);
170 }
171 ++existing_other;
172 }
173 }
174 }
175 }
176
177 // Converting to edges.
178 for (auto othernode : current_added) {
179 int& otherscore=current_score[othernode];
180 double finalscore;
181 if (weight_scheme == RANKED) {
182 finalscore = static_cast<double>(current_neighbors.size()) - 0.5 * static_cast<double>(otherscore);
183 } else {
184 finalscore = otherscore;
185 if (weight_scheme == JACCARD) {
186 finalscore = finalscore / (2 * (current_neighbors.size() + 1) - finalscore);
187 }
188 }
189
190 // Ensuring that an edge with a positive weight is always reported.
191 store.push_back(WeightedEdge(j, othernode, std::max(finalscore, 1e-6)));
192
193 // Resetting all those added to zero.
194 otherscore=0;
195 }
196 current_added.clear();
197 }
198
199 return store;
