306
307 int ngroups = means.size();
308 int nblocks = (ngroups ? means[0].size() : 0); // no blocks = no groups.

311 std::vector<Stat*> means2(ncombos), detected2(ncombos);
312 {
313 auto mIt = means2.begin(), dIt = detected2.begin();
314 for (int g = 0; g < ngroups; ++g) {
315 for (int b = 0; b < nblocks; ++b, ++mIt, ++dIt) {
316 *mIt = means[g][b];
317 *dIt = detected[g][b];
318 }
319 }
320 }

323 Overlord overlord(std::move(means2), std::move(detected2), cohen, auc, lfc, delta_detected, threshold);
